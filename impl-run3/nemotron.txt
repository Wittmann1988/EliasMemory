"""
Event Bus Module for Self-Improving Agent Framework

Provides an asynchronous publish-subscribe system with:
- Priority queue support
- Dead-letter queue for failed events
- Metrics collection (pub/sub latency, queue depth)
- SQLite persistence with WAL mode
- Event replay functionality
- Backpressure handling
- Graceful shutdown protocol
"""

import asyncio
import logging
import sqlite3
import time
from collections import deque, defaultdict
from dataclasses import dataclass, field
from enum import Enum, auto
from pathlib import Path
from typing import Any, Callable, Dict, List, Optional, Tuple, Union
from typing_extensions import TypeAlias

logger = logging.getLogger(__name__)

class EventPriority(Enum):
    """Priority levels for event queuing."""
    CRITICAL = auto()
    HIGH = auto()
    NORMAL = auto()
    LOW = auto()
    BACKGROUND = auto()
    
    def __lt__(self, other):
        return self.value < other.value

EventPayload: TypeAlias = Dict[str, Any]
EventSource: TypeAlias = str
EventKey: TypeAlias = str

class EventBus:
    """Asynchronous event bus with priority-based routing and persistence."""
    
    def __init__(self, db_path: Union[str, Path] = "event_bus.db"):
        self._queues: Dict[EventPriority, asyncio.PriorityQueue] = {
            priority: asyncio.PriorityQueue() 
            for priority in EventPriority
        }
        self._dead_letter: asyncio.Queue = asyncio.Queue()
        self._subscribers: Dict[str, List[Callable[[EventPayload], None]]]] = defaultdict(list)
        self._db_path = Path(db_path)
        self._metrics = {
            'publish_latency_ms': defaultdict(lambda: []),
            'queue_depth': defaultdict(lambda: 0),
            'dead_letter_count': 0
        }
        self._shutdown_event = asyncio.Event()
        self._setup_database()
        
    def _setup_database(self) -> None:
        """Initialize SQLite database with WAL mode."""
        with sqlite3.connect(self._db_path) as conn:
            conn.execute("PRAGMA journal_mode=WAL;")
            conn.execute("""
                CREATE TABLE IF NOT EXISTS events (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    priority INTEGER NOT NULL,
                    source TEXT,
                    key TEXT,
                    payload TEXT,
                    timestamp REAL,
                    processed BOOLEAN DEFAULT 0
                )
            """)
            conn.execute("""
                CREATE INDEX IF NOT EXISTS idx_events_processed 
                ON events(processed)
            """)
            
    async def publish(
        self, 
        payload: EventPayload, 
        priority: EventPriority = EventPriority.NORMAL,
        source: EventSource = "system",
        key: Optional[EventKey] = None
    ) -> None:
        """Publish an event with specified priority."""
        start_time = time.perf_counter()
        event = {
            'priority': priority.value,
            'source': source,
            'key': key or '',
            'payload': payload,
            'timestamp': time.time()
        }
        
        # Add to priority queue
        queue = self._queues[priority]
        await queue.put(event)
        self._metrics['queue_depth'][priority] += 1
        
        # Persist to database
        self._persist_event(event)
        
        # Notify subscribers
        for callback in self._subscribers[key or '']:
            try:
                asyncio.create_task(callback(event))
            except Exception as e:
                logger.error(f"Subscriber callback failed: {e}")
                
        latency = (time.perf_counter() - start_time) * 1000
        self._metrics['publish_latency_ms'][priority].append(latency)
    
    async def subscribe(
        self, 
        key: EventKey, 
        callback: Callable[[EventPayload], None],
        priority: EventPriority = EventPriority.NORMAL
    ) -> None:
        """Subscribe to events with specific key."""
        self._subscribers[key].append(callback)
        logger.debug(f"Subscribed to key '{key}' with priority {priority}")
    
    async def consume(self, timeout: Optional[float] = None) -> EventPayload:
        """Consume next available event."""
        try:
            event = await asyncio.wait_for(
                self._get_next_event(), 
                timeout=timeout
            )
            return event
        except asyncio.TimeoutError:
            return {}
    
    async def _get_next_event(self) -> EventPayload:
        """Internal method to get next event from queues."""
        # Check dead-letter first
        if not self._dead_letter.empty():
            event = await self._dead_letter.get()
            self._metrics['dead_letter_count'] += 1
            return event
            
        # Find highest priority non-empty queue
        for priority in EventPriority:
            if not self._queues[priority].empty():
                event = await self._queues[priority].get()
                self._metrics['queue_depth'][priority] -= 1
                return event
                
        # Wait for new event
        event = await self._queues[EventPriority.NORMAL].get()
        self._metrics['queue_depth'][EventPriority.NORMAL] -= 1
        return event
    
    async def acknowledge(self, event_id: int) -> None:
        """Mark event as processed in database."""
        with sqlite3.connect(self._db_path) as conn:
            conn.execute(
                "UPDATE events SET processed = 1 WHERE id = ?",
                (event_id,)
            )
    
    async def replay_events(self, processed_since: int) -> List[EventPayload]:
        """Replay unprocessed events after given timestamp."""
        events = []
        with sqlite3.connect(self._db_path) as conn:
            cursor = conn.execute(
                "SELECT id, payload FROM events WHERE processed = 0"
            )
            for row in cursor:
                event_id, payload = row
                events.append(payload)
        return events
    
    async def handle_failure(self, event: EventPayload) -> None:
        """Move failed event to dead-letter queue."""
        await self._dead_letter.put(event)
    
    def get_metrics(self) -> Dict[str, Any]:
        """Get current metrics snapshot."""
        return {
            'queue_depth': dict(self._metrics['queue_depth']),
            'publish_latency_ms': {
                str(k): v[-10:] if v else []  # Keep last 10 samples
                for k, v in self._metrics['publish_latency_ms'].items()
            },
            'dead_letter_count': self._metrics['dead_letter_count']
        }
    
    async def shutdown(self) -> None:
        """Gracefully shutdown the event bus."""
        logger.info("Shutting down event bus...")
        self._shutdown_event.set()
        # Drain queues
        while not any(q.empty() for q in self._queues.values()):
            await asyncio.sleep(0.01)
        logger.info("Event bus shutdown complete")

        candidates = self._embedding_index.get(category, []) if category else list(self._memories.keys())

            memory = self._memories[memory_id]
            distance = np.linalg.norm(query_embedding - memory['embedding'])
            results.append((memory_id, distance))
        
        results.sort(key=lambda x: x[1])

        """Apply memory decay based on half-life."""
        current_time = time.time()
        decay_factor = 0.5 ** (current_time / (half_life_hours * 3600))
        
        for memory_id in list(self._memories.keys()):
            last_access = self._last_access.get(memory_id, current_time)
            self._last_access[memory_id] = current_time
            
            if current_time - last_access > (half_life_hours * 3600):
                with sqlite3.connect(self._db_path) as conn:
                    conn.execute("DELETE FROM memories WHERE id = ?", (memory_id,))
                del self._memories[memory_id]
    
    async def get_diverse_memories(self, k: int = 5) -> List[Dict[str, Any]]:
        """Get diverse memories using Maximal Marginal Relevance."""
        if not self._memories:
            return []
        
        memories = list(self._memories.values())
        diverse = []
        available = set(memories.keys())
        
        while len(diverse) < k and available:
            if not diverse:
                best = max(available, key=lambda id: memories[id]['timestamp'])
            else:
                best = min(available, key=lambda id: 
                    min(np.linalg.norm(
                        memories[id]['embedding'] - mem['embedding'] 
                        for mem in diverse
                    ))
                )
            diverse.append(best)
            available.remove(best)
        
        return [self._memories[id] for id in diverse]

class OrchestratorState(Dict[str, Any]):
    """State schema for orchestrator."""
    agent_weights: Dict[str, float] = field(default_factory=dict)
    memory: MemoryStore = field(default_factory=MemoryStore)
    event_bus: EventBus = field(default_factory=EventBus)
    current_plan: Optional[str] = None
    execution_log: List[Dict[str, Any]] = field(default_factory=list)

class Orchestrator:
    """Main orchestrator managing agent workflow and self-improvement."""
    
    def __init__(self, 
                 memory: MemoryStore, 
                 event_bus: EventBus):
        self.state = OrchestratorState(
            memory=memory,
            event_bus=event_bus,
            agent_weights={"planner": 1.0, "executor": 1.0, "reviewer": 1.0}
        )
        self.graph = self._build_graph()
        self.checkpoint_saver = SqliteCheckpointSaver.from_conn_str(
            f"file:memory.db?mode=memory&cache=shared"
        )
        
    def _build_graph(self) -> StateGraph:
        """Build LangGraph with state transitions."""
        builder = StateGraph(OrchestratorState)
        
        # Add nodes
        builder.add_node("plan", self._plan)
        builder.add_node("act", self._act)
        builder.add_node("observe", self._observe)
        builder.add_node("improve", self._improve)
        
        # Define edges
        builder.add_edge("plan", "act")
        builder.add_edge("act", "observe")
        builder.add_edge("observe", "improve")
        builder.add_edge("improve", "plan")
        builder.add_edge("improve", END)
        
        return builder.compile()
    
    async def _plan(self, state: OrchestratorState) -> OrchestratorState:
        """Generate plan based on current state."""
        # In production, this would use LLM planning
        state.current_plan = "execute_task_1"
        logger.info(f"Generated plan: {state.current_plan}")
        return state
    
    async def _act(self, state: OrchestratorState) -> OrchestratorState:
        """Execute planned action."""
        # Execute task using executor agent
        logger.info("Executing action...")
        state.execution_log.append({
            "step": "act",
            "status": "success",
            "timestamp": time.time()
        })
        return state
    
    async def _observe(self, state: OrchestratorState) -> OrchestratorState:
        """Observe results of execution."""
        # Collect feedback and observations
        logger.info("Collecting observations...")
        state.execution_log.append({
            "step": "observe", 
            "feedback": "Task completed with 95% success",
            "timestamp": time.time()
        })
        return state
    
    async def _improve(self, state: OrchestratorState) -> OrchestratorState:
        """Trigger self-improvement based on feedback."""
        # Analyze performance and adjust weights
        await self._trigger_self_improvement()
        logger.info("Self-improvement cycle completed")
        return state
    
    async def _trigger_self_improvement(self) -> None:
        """Coordinate self-improvement process."""
        from .self_improver import SelfImprover
        improver = SelfImprover(self.state)
        await improver.optimize()
        
        # Update agent weights in state
        self.state.agent_weights = improver.optimized_weights
    
    async def run(self, task: str) -> Dict[str, Any]:
        """Run complete orchestration cycle for a task."""
        config = {"configurable": {"task": task}}
        
        # Initialize with checkpoint
        config_checkpoint = {"configurable": {"checkpoint_id": "initial"}}
        
        async with self.graph.with_checkpointing(self.checkpoint_saver) as graph:
            while True:
                # Check for termination condition
                if state.get("terminate", False):
                    break
                    
                # Execute one cycle
                result = await graph.invoke(state, config)
                state = result
                
                # Persist checkpoint
                await self.checkpoint_saver.save_checkpoint(config_checkpoint)
        
        return {
            "final_state": state,
            "metrics": self.state.memory.get_metrics(),
            "agent_weights": state.agent_weights
        }

"""
Self-Improver Module for Self-Improving Agent Framework

Analyzes agent performance, memory usage, and system metrics 
to drive continuous improvement through weight adjustment 
and memory optimization.
"""

import asyncio
import logging
from collections import defaultdict
from typing import Dict, List
from typing_extensions import TypeAlias

from .memory_store import MemoryStore

class ImprovementStrategy:
    """Strategies for system improvement."""
    PRUNE_LOW_PERFORMANCE = "prune_low_performance"
    OPTIMIZE_BATCHING = "optimize_batching"
    ADJUST_AGENT_WEIGHTS = "adjust_agent_weights"
    ENHANCE_EMBEDDING = "enhance_embedding"

class SelfImprover:
    """Core self-improvement engine."""
    
    def __init__(self, state: 'OrchestratorState'):
        self.state = state
        self.memory: MemoryStore = state.memory
        self.event_bus: 'EventBus' = state.event_bus
        self.agent_metrics: Dict[str, Dict] = defaultdict(lambda: {
            'success_rate': 0.0,
            'accuracy': 0.0,
            'latency': 0.0,
            'error_count': 0,
            'score': 0.0
        })
        
    async def analyze_performance(self) -> Dict[str, Any]:
        """Analyze system performance across all agents."""
        # Collect metrics from event bus and memory
        metrics = self.memory.get_metrics()
        
        # Calculate aggregate scores
        total_score = sum(
            metrics.get('agent_scores', {}).values(), default=0
        ) / len(metrics.get('agent_scores', {}))
        
        return {
            'total_score': total_score,
            'memory_hits': metrics.get('memory_hits', 0),
            'queue_depth': sum(metrics.get('queue_depth', {}).values()),
            'dead_letter_ratio': metrics.get('dead_letter_count', 0) / 
                               (metrics.get('total_events', 1) or 1)
        }
    
    async def optimize(self) -> Dict[str, Any]:
        """Optimize system based on performance analysis."""
        analysis = await self.analyze_performance()
        logger.info(f"Performance analysis: {analysis}")
        
        # Apply improvement strategies
        strategies = {
            ImprovementStrategy.ADJUST_AGENT_WEIGHTS: self._adjust_weights,
            ImprovementStrategy.PRUNE_LOW_PERFORMANCE: self._prune_memories,
            ImprovementStrategy.OPTIMIZE_BATCHING: self._optimize_batching,
            ImprovementStrategy.ENHANCE_EMBEDDING: self._enhance_embeddings
        }
        
        applied_strategies = []
        for strategy_name, strategy_func in strategies.items():
            if await self._should_apply_strategy(strategy_name, analysis):
                await strategy_func(analysis)
                applied_strategies.append(strategy_name)
                
        return {
            'applied_strategies': applied_strategies,
            'optimization_score': len(applied_strategies) / len(strategies)
        }
    
    async def _should_apply_strategy(self, strategy: str, analysis: Dict) -> bool:
        """Determine if strategy should be applied."""
        thresholds = {
            ImprovementStrategy.ADJUST_AGENT_WEIGHTS: analysis['total_score'] < 0.7,
            ImprovementStrategy.PRUNE_LOW_PERFORMANCE: analysis['memory_hits'] > 1000,
            ImprovementStrategy.OPTIMIZE_BATCHING: analysis['queue_depth'] > 50,
            ImprovementStrategy.ENHANCE_EMBEDDING: analysis['dead_letter_ratio'] > 0.1
        }
        return thresholds.get(strategy, False)
    
    async def _adjust_weights(self, analysis: Dict) -> None:
        """Adjust agent weights based on performance."""
        # Implementation would modify self.state.agent_weights
        logger.info("Adjusting agent weights based on performance")
        # Example weight adjustment logic
        self.state.agent_weights = {
            k: max(0.1, v - 0.1 if analysis['total_score'] < 0.6 else v + 0.05)
            for k, v in self.state.agent_weights.items()
        }
    
    async def _prune_memories(self, analysis: Dict) -> None:
        """Prune low-value memories."""
        logger.info("Pruning low-value memories")
        await self.memory.decay_memories(half_life_hours=12.0)
        # Could also remove memories with low access frequency
    
    async def _optimize_batching(self, analysis: Dict) -> None:
        """Optimize batch processing parameters."""
        logger.info("Optimizing batch processing")
        # Adjust batch size based on queue depth
        optimal_batch = max(1, min(100, analysis['queue_depth'] // 10))
        self.memory._batch_size = optimal_batch
    
    async def _enhance_embeddings(self, analysis: Dict) -> None:
        """Enhance embedding quality."""
        logger.info("Enhancing embedding quality")
        # Could upgrade to larger embedding models or adjust dimensionality
        self.memory._embedding_dim = max(128, min(1024, self.memory._embedding_dim + 128))

Demonstrates complete workflow with 3 agents, 5 tasks,

"""

import asyncio
import logging
import time
from typing import List

from .event_bus import EventBus
from .memory_store import MemoryStore
from .orchestrator import Orchestrator
from .agent_base import (
    PlannerAgent, 
    ExecutorAgent, 
    ReviewerAgent,
    AgentRole
)

# Configure logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger("main")

async def run_demo() -> None:
    """Run the complete self-improving agent demonstration."""
    logger.info("Starting self-improving agent framework demo...")
    
    # Initialize core components
    memory = MemoryStore(db_path="demo_memory.db")
    event_bus = EventBus(db_path="demo_event_bus.db")
    
    # Create specialized agents
    planner = PlannerAgent(
        role=AgentRole.PLANNER,
        event_bus=event_bus,
        memory=memory,
        name="Planner-1"
    )
    
    executor = ExecutorAgent(
        role=AgentRole.EXECUTOR,
        event_bus=event_bus,
        memory=memory,
        name="Executor-1"
    )
    
    reviewer = ReviewerAgent(
        role=AgentRole.REVIEWER,
        event_bus=event_bus,
        memory=memory,
        name="Reviewer-1"
    )
    
    # Initialize orchestrator
    orchestrator = Orchestrator(
        memory=memory,
        event_bus=event_bus
    )
    
    # Register agents with orchestrator
    orchestrator.state.agents = {
        AgentRole.PLANNER: planner,
        AgentRole.EXECUTOR: executor,
        AgentRole.REVIEWER: reviewer
    }
    
    # Define test tasks
    tasks = [
        "Plan and execute user onboarding workflow",
        "Optimize database query performance by 30%",
        "Implement new user analytics dashboard",
        "Handle payment processing edge cases",
        "Deploy security patch for authentication module"
    ]
    
    # Track improvement metrics
    initial_metrics = {}
    improvement_scores = []
    
    # Run 5 tasks with progressive improvement
    for i, task in enumerate(tasks):
        logger.info(f"\n=== Running Task {i+1}: {task[:30]}... ===")
        
        # Start timing
        start_time = time.time()
        
        # Execute orchestration cycle
        result = await orchestrator.run(task)
        
        # Calculate performance metrics
        cycle_time = time.time() - start_time
        success = result['final_state'].get('terminate', False)
        
        # Record improvement
        improvement = {
            'task': task,
            'success': success,
            'duration_s': cycle_time,
            'final_score': result['final_state'].get('agent_weights', {}).get('planner', 1.0)
        }
        improvement_scores.append(improvement)
        
        # Log results
        logger.info(f"Task {i+1} completed in {cycle_time:.2f}s - Success: {success}")
        logger.info(f"Final agent weights: {result['final_state'].get('agent_weights')}")
        
        # Update metrics store
        await memory.store(f"Completed task {i+1}: {success}", category="performance")
    
    # Final analysis
    final_metrics = await orchestrator.run("final_analysis")
    logger.info(f"\n=== FINAL ANALYSIS ===")
    logger.info(f"Total tasks completed: {len(tasks)}")
    logger.info(f"Average success rate: {sum(1 for t in improvement_scores if t['success']) / len(tasks):.2f}")
    logger.info(f"Average improvement score: {sum(t['final_score'] for t in improvement_scores) / len(tasks):.2f}")
    
    # Display memory store metrics
    mem_metrics = await memory.get_metrics()
    logger.info(f"Memory store metrics: {mem_metrics}")
    
    # Show event bus metrics
    bus_metrics = event_bus.get_metrics()
    logger.info(f"Event bus metrics: {bus_metrics}")
    
    logger.info("\nDemo completed successfully!")

if __name__ == "__main__":
    asyncio.run(run_demo())
