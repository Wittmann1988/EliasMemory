@dataclass(order=True)
class Event:
    """Priority event for async processing."""
    priority: int
    timestamp: float = field(compare=False)
    event_type: str = field(compare=False)
    payload: Dict[str, Any] = field(compare=False)
    event_id: str = field(compare=False, default_factory=lambda: str(uuid.uuid4()))
    source: Optional[str] = field(compare=False, default=None)
    
    def to_dict(self) -> Dict[str, Any]:

            'source': self.source,
            'priority': self.priority

class MetricsCollector:
    """Collects and aggregates event metrics."""
    
    def __init__(self):
        self._counters: Dict[str, int] = {}
        self._latencies: Dict[str, List[float]] = {}
        self._lock = asyncio.Lock()
    
    async def increment(self, metric_name: str, value: int = 1):
        async with self._lock:
            self._counters[metric_name] = self._counters.get(metric_name, 0) + value
    
    async def record_latency(self, metric_name: str, latency: float):
        async with self._lock:
            if metric_name not in self._latencies:
                self._latencies[metric_name] = []
            self._latencies[metric_name].append(latency)
    
    async def get_stats(self) -> Dict[str, Any]:
        async with self._lock:
            stats = {}
            for name, count in self._counters.items():
                stats[f"{name}_count"] = count
            
            for name, latencies in self._latencies.items():
                if latencies:
                    stats[f"{name}_avg_latency"] = sum(latencies) / len(latencies)
                    stats[f"{name}_max_latency"] = max(latencies)
            return stats

        try:
            if self._wal_mode:
                conn.execute("PRAGMA journal_mode=WAL")
            conn.execute("PRAGMA synchronous=NORMAL")
            
            conn.execute("""
                CREATE TABLE IF NOT EXISTS events (
                    event_id TEXT PRIMARY KEY,
                    timestamp REAL,
                    event_type TEXT,
                    payload TEXT,
                    status TEXT DEFAULT 'pending',

                    error_msg TEXT,

                CREATE INDEX IF NOT EXISTS idx_status ON events(status)
            """)
            
            conn.commit()
        finally:
            conn.close()
    
    @contextmanager
    def _get_connection(self):
        """Connection pooling context manager."""
        conn = sqlite3.connect(self._db_path, check_same_thread=False)
        try:
            if self._wal_mode:
                conn.execute("PRAGMA journal_mode=WAL")
            yield conn
        finally:
            conn.close()
    
    async def start(self):
        """Start the event processing worker."""
        self._worker_task = asyncio.create_task(self._process_events())
        logger.info("EventBus started")
    
    async def stop(self, timeout: float = 30.0):
        """Graceful shutdown with timeout."""
        logger.info("EventBus stopping...")
        self._shutdown_event.set()
        
        # Drain queue with timeout
        try:
            await asyncio.wait_for(self._worker_task, timeout=timeout)
        except asyncio.TimeoutError:
            logger.warning("EventBus shutdown timed out")
            self._worker_task.cancel()

        await self._persist_remaining()
        logger.info("EventBus stopped")

                remaining.append(event)

                break

                    )

        """
        Publish event with backpressure handling.
        Returns False if queue is full (backpressure).
        """
        if self._shutdown_event.is_set():
            return False
        
        # Persist first
        with self._get_connection() as conn:
            conn.execute(
                "INSERT OR REPLACE INTO events VALUES (?, ?, ?, ?, ?, ?, ?, ?)",

            )
            conn.commit()

            return False

            self._subscribers[event_type] = set()
        self._subscribers[event_type].add(handler)
        logger.info(f"Handler subscribed to {event_type}")
    
    def unsubscribe(self, event_type: str, handler: Callable[[Event], Any]):
        """Unsubscribe handler from event type."""
        if event_type in self._subscribers:
            self._subscribers[event_type].discard(handler)

        """Main processing loop."""
        while not self._shutdown_event.is_set():

    async def _handle_event(self, event: Event, max_retries: int = 3):
        """Handle single event with retry logic."""
        start_time = time.time()
        handlers = self._subscribers.get(event.event_type, set())
        
        if not handlers:
            await self._metrics.increment('events_no_handler')
            return
        
        success = False
        last_error = None
        
        for attempt in range(max_retries):
            try:
                for handler in handlers:
                    if asyncio.iscoroutinefunction(handler):
                        await asyncio.wait_for(handler(event), timeout=10.0)
                    else:
                        handler(event)
                success = True
                break
            except Exception as e:
                last_error = str(e)
                await asyncio.sleep(0.5 * (2 ** attempt))  # Exponential backoff
        
        latency = time.time() - start_time
        
        if success:
            await self._metrics.increment('events_processed')
            await self._metrics.record_latency('event_processing', latency)
            self._update_event_status(event.event_id, 'completed')
        else:
            await self._metrics.increment('events_failed')
            event.payload['error'] = last_error
            event.payload['retry_count'] = max_retries
            
            if event.payload.get('retry_count', 0) >= max_retries:
                self._dead_letter_queue.append(event)
                self._update_event_status(event.event_id, 'dead_letter', last_error)
                await self._metrics.increment('events_dead_lettered')
            else:
                # Re-queue with lower priority
                event.priority += 1
                await self._queue.put(event)
    
    def _update_event_status(self, event_id: str, status: str, error_msg: Optional[str] = None):

    async def _apply_decay(self):
        """Apply decay to all memories based on half-life."""
        current_time = time.time()
        
        with self._get_connection() as conn:
            cursor = conn.execute(
                "SELECT id, timestamp, half_life FROM memories"
            )
            rows = cursor.fetchall()

                mem_id, timestamp, half_life = row
                age = current_time - timestamp
                # Decay factor: 0.5^(age/half_life)
                relevance = 0.5 ** (age / half_life) if half_life > 0 else 1.0
                
                conn.execute(

    @retry(stop=stop_after_attempt(4),

        """

    async def search(self, query: str, top_k: int = 5,

        """

                JOIN vec_memories v ON m.id = v.rowid

                decay_factor = meta_dict.get('relevance', 1.0)
                adjusted_relevance = relevance * decay_factor
                
                candidates.append({
                    'id': mem_id,
                    'content': content,
                    'relevance': adjusted_relevance,
                    'vector': emb,
                    'metadata': meta_dict,
                    'timestamp': timestamp,
                    'access_count': access_count
                })
            
            # MMR selection
            selected = []
            remaining = candidates.copy()
            
            while len(selected) < top_k and remaining:
                if not selected:
                    # First pick: highest relevance
                    best = max(remaining, key=lambda x: x['relevance'])
                else:
                    # MMR score: lambda * relevance - (1-lambda) * max_sim_to_selected
                    best_score = -float('inf')
                    best = None
                    
                    for cand in remaining:
                        max_sim = max(np.dot(cand['vector'], s['vector']) 
                                    for s in selected)
                        mmr_score = (mmr_lambda * cand['relevance'] - 
                                   (1 - mmr_lambda) * max_sim)
                        
                        if mmr_score > best_score:
                            best_score = mmr_score
                            best = cand
                
                selected.append(best)
                remaining.remove(best)
                
                # Update access count
                conn.execute(
                    "UPDATE memories SET access_count = access_count + 1, last_access = ? WHERE id = ?",
                    (time.time(), best['id'])
                )
            
            conn.commit()
        
        if self._event_bus:
            await self._event_bus.publish(Event(
                priority=2,
                timestamp=time.time(),
                event_type="memory_searched",
                payload={"query": query, "results": len(selected)}
            ))
        
        return selected
