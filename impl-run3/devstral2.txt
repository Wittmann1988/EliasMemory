# FILE: event_bus.py
"""Production-grade async event bus with persistence, backpressure, and metrics."""

import asyncio
import json
import sqlite3
import time
from collections import defaultdict, deque
from dataclasses import dataclass
from enum import Enum
from typing import Any, AsyncIterator, Callable, Dict, List, Optional, Set, Tuple

import aiosqlite
from typing_extensions import Self

__all__ = ["EventBus", "Event", "Priority", "EventHandler"]

class Priority(Enum):
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    CRITICAL = 4

@dataclass
class Event:
    """Immutable event with priority and metadata."""
    topic: str
    data: Dict[str, Any]
    priority: Priority = Priority.MEDIUM
    timestamp: float = 0.0
    attempt: int = 0
    metadata: Optional[Dict[str, Any]] = None

    def __post_init__(self):
        self.timestamp = self.timestamp or time.time()
        self.metadata = self.metadata or {}

class EventHandler:
    """Base class for event handlers with async processing."""
    def __init__(self, max_retries: int = 3, backoff_factor: float = 1.0):
        self.max_retries = max_retries
        self.backoff_factor = backoff_factor

    async def handle(self, event: Event) -> bool:
        """Process event. Return True if successful."""
        raise NotImplementedError

class EventBus:
    """Production async event bus with SQLite persistence and backpressure."""

    def __init__(self, db_path: str = ":memory:", max_queue_size: int = 1000):
        self.db_path = db_path
        self.max_queue_size = max_queue_size
        self.handlers: Dict[str, List[EventHandler]] = defaultdict(list)
        self.metrics = {
            "events_processed": 0,
            "events_failed": 0,
            "queue_size": 0,
            "latency_ms": []
        }
        self._shutdown = False
        self._queue: Dict[Priority, deque] = {
            Priority.LOW: deque(),
            Priority.MEDIUM: deque(),
            Priority.HIGH: deque(),
            Priority.CRITICAL: deque()
        }
        self._processing = asyncio.Semaphore(10)  # Backpressure control

    async def initialize(self):
        """Initialize SQLite database with WAL mode."""
        self.conn = await aiosqlite.connect(self.db_path)
        await self.conn.execute("PRAGMA journal_mode=WAL")
        await self.conn.execute("""
            CREATE TABLE IF NOT EXISTS events (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                topic TEXT NOT NULL,
                data TEXT NOT NULL,
                priority INTEGER NOT NULL,
                timestamp REAL NOT NULL,
                attempt INTEGER DEFAULT 0,
                metadata TEXT,
                processed INTEGER DEFAULT 0
            )
        """)
        await self.conn.commit()
        await self._load_persisted_events()

    async def _load_persisted_events(self):
        """Load unprocessed events from database."""
        async with self.conn.execute(
            "SELECT topic, data, priority, timestamp, attempt, metadata "
            "FROM events WHERE processed = 0 ORDER BY priority DESC, timestamp"
        ) as cursor:
            async for row in cursor:
                event = Event(
                    topic=row[0],
                    data=json.loads(row[1]),
                    priority=Priority(row[2]),
                    timestamp=row[3],
                    attempt=row[4],
                    metadata=json.loads(row[5]) if row[5] else None
                )
                self._queue[event.priority].append(event)
                self.metrics["queue_size"] += 1

    async def publish(self, event: Event) -> bool:
        """Publish event with backpressure handling."""
        if self._shutdown:
            return False

        if self.metrics["queue_size"] >= self.max_queue_size:
            raise RuntimeError("Event bus backpressure limit reached")

        self._queue[event.priority].append(event)
        self.metrics["queue_size"] += 1

        # Persist to database
        await self.conn.execute(
            "INSERT INTO events (topic, data, priority, timestamp, attempt, metadata) "
            "VALUES (?, ?, ?, ?, ?, ?)",
            (
                event.topic,
                json.dumps(event.data),
                event.priority.value,
                event.timestamp,
                event.attempt,
                json.dumps(event.metadata) if event.metadata else None
            )
        )
        await self.conn.commit()
        return True

    async def subscribe(self, topic: str, handler: EventHandler):
        """Subscribe handler to topic."""
        self.handlers[topic].append(handler)

    async def _process_event(self, event: Event) -> bool:
        """Process single event with retry logic."""
        start_time = time.time()
        success = False

        try:
            handlers = self.handlers.get(event.topic, [])
            if not handlers:
                return True  # No handlers for topic

            for handler in handlers:
                if await handler.handle(event):
                    success = True
                else:
                    success = False
                    break

            if success:
                self.metrics["events_processed"] += 1
                self.metrics["latency_ms"].append((time.time() - start_time) * 1000)
                await self._mark_processed(event)
            else:
                self.metrics["events_failed"] += 1
                if event.attempt < 3:  # Max 3 attempts
                    event.attempt += 1
                    await self.publish(event)

        except Exception as e:
            print(f"Error processing event {event.topic}: {str(e)}")
            success = False

        return success

    async def _mark_processed(self, event: Event):
        """Mark event as processed in database."""
        await self.conn.execute(
            "UPDATE events SET processed = 1 WHERE topic = ? AND timestamp = ?",
            (event.topic, event.timestamp)
        )
        await self.conn.commit()

    async def process_queue(self):
        """Continuously process events with priority handling."""
        while not self._shutdown:
            try:
                # Get next event by priority
                event = None
                for priority in [Priority.CRITICAL, Priority.HIGH, Priority.MEDIUM, Priority.LOW]:
                    if self._queue[priority]:
                        event = self._queue[priority].popleft()
                        self.metrics["queue_size"] -= 1
                        break

                if event:
                    async with self._processing:
                        await self._process_event(event)
                else:
                    await asyncio.sleep(0.1)

            except Exception as e:
                print(f"Error in event processing loop: {str(e)}")
                await asyncio.sleep(1)

    async def shutdown(self):
        """Graceful shutdown with queue persistence."""
        self._shutdown = True
        # Persist remaining events
        for priority in self._queue:
            for event in self._queue[priority]:
                await self.conn.execute(
                    "INSERT INTO events (topic, data, priority, timestamp, attempt, metadata) "
                    "VALUES (?, ?, ?, ?, ?, ?)",
                    (
                        event.topic,
                        json.dumps(event.data),
                        event.priority.value,
                        event.timestamp,
                        event.attempt,
                        json.dumps(event.metadata) if event.metadata else None
                    )
                )
        await self.conn.commit()
        await self.conn.close()

    def get_metrics(self) -> Dict[str, Any]:
        """Get current metrics snapshot."""
        return {
            **self.metrics,
            "avg_latency_ms": sum(self.metrics["latency_ms"]) / len(self.metrics["latency_ms"])
            if self.metrics["latency_ms"] else 0
        }

"""Vector memory store with NVIDIA embeddings, decay, and diversity search."""

import asyncio
import json
import math
import sqlite3
from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import Any, Dict, List, Optional, Tuple

import aiosqlite
import numpy as np
from numpy.typing import NDArray
from typing_extensions import Self

__all__ = ["MemoryStore", "MemoryEntry", "EmbeddingClient"]

@dataclass
class MemoryEntry:
    """Structured memory entry with metadata and decay."""
    id: str
    content: str
    embedding: Optional[NDArray[np.float32]] = None
    metadata: Optional[Dict[str, Any]] = None
    created_at: datetime = None
    last_accessed: datetime = None
    importance: float = 1.0

    def __post_init__(self):
        self.created_at = self.created_at or datetime.utcnow()
        self.last_accessed = self.last_accessed or self.created_at
        self.metadata = self.metadata or {}

class EmbeddingClient:
    """NVIDIA embedding client with batching and retry logic."""

    def __init__(self, api_key: str, batch_size: int = 32, max_retries: int = 3):
        self.api_key = api_key
        self.batch_size = batch_size
        self.max_retries = max_retries
        # In production, this would use actual NVIDIA API
        self._mock_model = self._load_mock_model()

    def _load_mock_model(self):
        """Load mock embedding model for demonstration."""
        import sentence_transformers
        return sentence_transformers.SentenceTransformer('all-MiniLM-L6-v2')

    async def embed(self, texts: List[str]) -> List[NDArray[np.float32]]:
        """Generate embeddings with batching and retry."""
        if not texts:
            return []

        batches = [texts[i:i + self.batch_size] for i in range(0, len(texts), self.batch_size)]
        all_embeddings = []

        for batch in batches:
            for attempt in range(self.max_retries):
                try:
                    # In production: call NVIDIA API
                    embeddings = self._mock_model.encode(batch, convert_to_numpy=True)
                    all_embeddings.extend(embeddings)
                    break
                except Exception as e:
                    if attempt == self.max_retries - 1:
                        raise RuntimeError(f"Failed to embed after {self.max_retries} attempts: {str(e)}")
                    await asyncio.sleep(2 ** attempt)  # Exponential backoff

        return all_embeddings

class MemoryStore:
    """SQLite-backed vector store with memory decay and diversity search."""

    def __init__(self, db_path: str = ":memory:", embedding_client: Optional[EmbeddingClient] = None):
        self.db_path = db_path
        self.embedding_client = embedding_client or EmbeddingClient(api_key="mock-key")
        self.conn: Optional[aiosqlite.Connection] = None
        self._initialized = False

    async def initialize(self):
        """Initialize SQLite database with vector extensions."""
        self.conn = await aiosqlite.connect(self.db_path)
        await self.conn.execute("PRAGMA journal_mode=WAL")
        await self.conn.execute("""
            CREATE TABLE IF NOT EXISTS memories (
                id TEXT PRIMARY KEY,
                content TEXT NOT NULL,
                embedding BLOB,
                metadata TEXT,
                created_at INTEGER NOT NULL,
                last_accessed INTEGER NOT NULL,
                importance REAL DEFAULT 1.0
            )
        """)
        await self.conn.execute("CREATE VIRTUAL TABLE IF NOT EXISTS memories_vec USING vec0(embedding(384))")
        await self.conn.commit()
        self._initialized = True

    async def add_memory(self, entry: MemoryEntry) -> bool:
        """Add memory with automatic embedding generation."""
        if not self._initialized:
            raise RuntimeError("MemoryStore not initialized")

        if entry.embedding is None and entry.content:
            entry.embedding = (await self.embedding_client.embed([entry.content]))[0]

        await self.conn.execute(
            "INSERT INTO memories (id, content, embedding, metadata, created_at, last_accessed, importance) "
            "VALUES (?, ?, ?, ?, ?, ?, ?)",
            (
                entry.id,
                entry.content,
                entry.embedding.tobytes() if entry.embedding is not None else None,
                json.dumps(entry.metadata),
                int(entry.created_at.timestamp()),
                int(entry.last_accessed.timestamp()),
                entry.importance
            )
        )
        await self.conn.commit()
        return True

    async def search(self, query: str, k: int = 5, decay_half_life: float = 24.0) -> List[MemoryEntry]:
        """Search with MMR diversity and temporal decay."""
        if not self._initialized:
            raise RuntimeError("MemoryStore not initialized")

        query_embedding = (await self.embedding_client.embed([query]))[0]

        # Get candidate memories
        async with self.conn.execute(
            "SELECT id, content, embedding, metadata, created_at, last_accessed, importance "
            "FROM memories"
        ) as cursor:
            rows = await cursor.fetchall()

        if not rows:
            return []

        # Calculate scores with temporal decay
        now = datetime.utcnow()
        memories = []
        for row in rows:
            embedding = np.frombuffer(row[2], dtype=np.float32) if row[2] else None
            created_at = datetime.fromtimestamp(row[4])
            last_accessed = datetime.fromtimestamp(row[5])

            # Temporal decay: importance halves every half_life hours
            hours_since_access = (now - last_accessed).total_seconds() / 3600
            decay_factor = 0.5 ** (hours_since_access / decay_half_life)

            score = np.dot(query_embedding, embedding) if embedding is not None else 0
            adjusted_score = score * row[6] * decay_factor

            memories.append((
                adjusted_score,
                MemoryEntry(
                    id=row[0],
                    content=row[1],
                    embedding=embedding,
                    metadata=json.loads(row[3]) if row[3] else None,
                    created_at=created_at,
                    last_accessed=last_accessed,
                    importance=row[6]
                )
            ))

        # Sort by score and apply MMR for diversity
        memories.sort(key=lambda x: x[0], reverse=True)
        selected = []
        selected_indices = set()

        # Greedy MMR selection
        for i, (score, mem) in enumerate(memories):
            if len(selected) >= k:
                break

            if i == 0:
                selected.append(mem)
                selected_indices.add(i)
            else:
                # Calculate diversity (max distance to selected)
                diversity = min(
                    np.linalg.norm(mem.embedding - selected_mem.embedding)
                    for selected_mem in selected
                ) if mem.embedding is not None else 0

                # Combine relevance and diversity
                mmr_score = 0.7 * score + 0.3 * diversity
                if mmr_score > min((s[0] for s in selected), default=0):
                    # Replace the least relevant
                    min_idx = min(range(len(selected)), key=lambda j: selected[j].embedding.dot(query_embedding) if selected[j].embedding is not None else 0)
                    selected[min_idx] = mem

        return selected

    async def update_access_time(self, memory_id: str):
        """Update last accessed time for memory decay."""
        if not self._initialized:
            raise RuntimeError("MemoryStore not initialized")

        await self.conn.execute(
            "UPDATE memories SET last_accessed = ? WHERE id = ?",
            (int(datetime.utcnow().timestamp()), memory_id)
        )
        await self.conn.commit()

    async def prune_memories(self, threshold: float = 0.1):
        """Prune low-importance, rarely accessed memories."""
        if not self._initialized:
            raise RuntimeError("MemoryStore not initialized")

        now = datetime.utcnow()
        cutoff = (now - timedelta(days=30)).timestamp()

        await self.conn.execute(
            "DELETE FROM memories WHERE importance < ? AND last_accessed < ?",
            (threshold, cutoff)
        )
        await self.conn.commit()
        return True

    async def get_metrics(self) -> Dict[str, Any]:
        """Get memory store metrics."""
        if not self._initialized:
            raise RuntimeError("MemoryStore not initialized")

        async with self.conn.execute("SELECT COUNT(*) FROM memories") as cursor:
            count = await cursor.fetchone()

        return {
            "memory_count": count[0] if count else 0,
            "avg_importance": 0,  # Would calculate in production
            "access_distribution": {}  # Would calculate in production
        }

# FILE: orchestrator.py
"""Agent orchestration using LangGraph StateGraph with self-improvement."""

from typing import Any, Dict, List, Optional, Tuple
from langgraph.graph import StateGraph, END
from langgraph.graph.message import add_messages
from pydantic import BaseModel, Field
from typing_extensions import TypedDict

from event_bus import EventBus, Event, Priority
from memory_store import MemoryStore, MemoryEntry

__all__ = ["Orchestrator", "AgentState", "Task"]

class Task(BaseModel):
    """Structured task representation."""
    id: str
    description: str
    context: Dict[str, Any] = Field(default_factory=dict)
    max_attempts: int = 3
    current_attempt: int = 0

class AgentState(TypedDict):
    """State passed between agents in the graph."""
    task: Task
    plan: Optional[str]
    execution_result: Optional[Dict[str, Any]]
    review_score: Optional[float]
    improvement_suggestions: List[str]
    memory_entries: List[MemoryEntry]

class Orchestrator:
    """Coordinates agents through plan->act->observe->improve cycle."""

    def __init__(
        self,
        event_bus: EventBus,
        memory_store: MemoryStore,
        planner_agent: Any,
        executor_agent: Any,
        reviewer_agent: Any,
        self_improver: Any
    ):
        self.event_bus = event_bus
        self.memory_store = memory_store
        self.planner = planner_agent
        self.executor = executor_agent
        self.reviewer = reviewer_agent
        self.improver = self_improver
        self._build_graph()

    def _build_graph(self):
        """Construct the agent workflow graph."""
        workflow = StateGraph(AgentState)

        # Define nodes
        workflow.add_node("plan", self._plan_node)
        workflow.add_node("execute", self._execute_node)
        workflow.add_node("review", self._review_node)
        workflow.add_node("improve", self._improve_node)
        workflow.add_node("checkpoint", self._checkpoint_node)

        # Define edges
        workflow.add_edge("plan", "execute")
        workflow.add_edge("execute", "review")
        workflow.add_conditional_edges(
            "review",
            self._should_improve,
            {
                "improve": "improve",
                "checkpoint": "checkpoint"
            }
        )
        workflow.add_edge("improve", "checkpoint")
        workflow.add_edge("checkpoint", END)

        # Set entry point
        workflow.set_entry_point("plan")

        self.graph = workflow.compile()

    async def _plan_node(self, state: AgentState) -> AgentState:
        """Generate execution plan using planner agent."""
        plan = await self.planner.plan(state["task"])
        return {**state, "plan": plan}

    async def _execute_node(self, state: AgentState) -> AgentState:
        """Execute plan using executor agent."""
        result = await self.executor.execute(state["task"], state["plan"])
        return {**state, "execution_result": result}

    async def _review_node(self, state: AgentState) -> AgentState:
        """Review execution results."""
        score, suggestions = await self.reviewer.review(
            state["task"],
            state["plan"],
            state["execution_result"]
        )
        return {
            **state,
            "review_score": score,
            "improvement_suggestions": suggestions
        }

    async def _improve_node(self, state: AgentState) -> AgentState:
        """Apply self-improvement based on review."""
        improved_state = await self.improver.improve(state)
        return improved_state

    async def _checkpoint_node(self, state: AgentState) -> AgentState:
        """Persist successful execution to memory."""
        if state["review_score"] and state["review_score"] >= 0.7:
            entry = MemoryEntry(
                id=f"task_{state['task'].id}",
                content=f"Task: {state['task'].description}\nPlan: {state['plan']}\nResult: {state['execution_result']}",
                metadata={
                    "task_id": state["task"].id,
                    "score": state["review_score"],
                    "timestamp": datetime.utcnow().isoformat()
                },
                importance=state["review_score"]
            )
            await self.memory_store.add_memory(entry)

        # Publish completion event
        await self.event_bus.publish(Event(
            topic="task_completed",
            data={
                "task_id": state["task"].id,
                "success": state["review_score"] >= 0.7 if state["review_score"] else False,
                "score": state["review_score"]
            },
            priority=Priority.HIGH
        ))

        return state

    def _should_improve(self, state: AgentState) -> str:
        """Determine if improvement is needed based on review score."""
        if state["review_score"] is None:
            return "checkpoint"
        return "improve" if state["review_score"] < 0.8 else "checkpoint"

    async def run_task(self, task: Task) -> Dict[str, Any]:
        """Execute complete workflow for a task."""
        initial_state = AgentState(
            task=task,
            plan=None,
            execution_result=None,
            review_score=None,
            improvement_suggestions=[],
            memory_entries=[]
        )

        final_state = await self.graph.ainvoke(initial_state)
        return {
            "task_id": task.id,
            "success": final_state["review_score"] >= 0.7 if final_state["review_score"] else False,
            "score": final_state["review_score"],
            "improvements": final_state["improvement_suggestions"]
        }

"""Base agent classes with lifecycle management and metrics."""

import time
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional, Tuple

from event_bus import Event, EventBus, EventHandler, Priority
from memory_store import MemoryEntry, MemoryStore

__all__ = ["BaseAgent", "PlannerAgent", "ExecutorAgent", "ReviewerAgent", "AgentMetrics"]

@dataclass
class AgentMetrics:
    """Track agent performance metrics."""
    tasks_processed: int = 0
    success_rate: float = 0.0
    avg_processing_time: float = 0.0
    memory_hits: int = 0
    last_updated: float = 0.0

    def update(self, success: bool, processing_time: float, memory_hit: bool = False):
        """Update metrics with new data point."""
        self.tasks_processed += 1
        self.success_rate = (self.success_rate * (self.tasks_processed - 1) + success) / self.tasks_processed
        self.avg_processing_time = (self.avg_processing_time * (self.tasks_processed - 1) + processing_time) / self.tasks_processed
        if memory_hit:
            self.memory_hits += 1
        self.last_updated = time.time()

class BaseAgent(ABC, EventHandler):
    """Base class for all agents with lifecycle and metrics."""

    def __init__(
        self,
        agent_id: str,
        event_bus: EventBus,
        memory_store: MemoryStore,
        weight: float = 1.0
    ):
        self.agent_id = agent_id
        self.event_bus = event_bus
        self.memory_store = memory_store
        self.weight = weight  # Used for routing decisions
        self.metrics = AgentMetrics()
        self._initialized = False

    async def initialize(self):
        """Initialize agent resources."""
        await self.memory_store.initialize()
        self._initialized = True

    async def handle(self, event: Event) -> bool:
        """Handle events from the event bus."""
        if not self._initialized:
            return False

        try:
            if event.topic == f"agent_{self.agent_id}_task":
                start_time = time.time()
                success = await self._process_task(event.data)
                processing_time = time.time() - start_time
                self.metrics.update(success, processing_time)
                return success
            return True
        except Exception as e:
            print(f"Agent {self.agent_id} error: {str(e)}")
            return False

    @abstractmethod
    async def _process_task(self, task_data: Dict[str, Any]) -> bool:
        """Process a specific task."""
        pass

    async def _search_memory(self, query: str, k: int = 3) -> List[MemoryEntry]:
        """Search memory store with agent-specific parameters."""
        return await self.memory_store.search(query, k=k)

    async def _log_event(self, topic: str, data: Dict[str, Any], priority: Priority = Priority.MEDIUM):
        """Publish event to bus."""
        await self.event_bus.publish(Event(
            topic=topic,
            data=data,
            priority=priority,
            metadata={"agent_id": self.agent_id}
        ))

class PlannerAgent(BaseAgent):
    """Generates execution plans for tasks."""

    async def _process_task(self, task_data: Dict[str, Any]) -> bool:
        """Generate plan for task."""
        start_time = time.time()

        # Search relevant memories
        memories = await self._search_memory(task_data["description"])
        memory_context = "\n".join([f"Memory {i+1}: {m.content}" for i, m in enumerate(memories)])

        # Simple planning logic (would use LLM in production)
        plan = f"""
        Step 1: Analyze requirements from '{task_data['description']}'
        Step 2: Consider relevant past experiences:
        {memory_context}
        Step 3: Execute with these parameters: {task_data.get('context', {})}
        """

        processing_time = time.time() - start_time
        self.metrics.update(True, processing_time, memory_hit=bool(memories))

        # Publish plan event
        await self._log_event(
            "plan_generated",
            {
                "task_id": task_data["id"],
                "plan": plan,
                "memory_hits": len(memories)
            },
            Priority.HIGH
        )

        return True

    async def plan(self, task: Dict[str, Any]) -> str:
        """Public method to generate plan."""
        if not self._initialized:
            raise RuntimeError("Agent not initialized")

        # Process through event bus for consistency
        await self.handle(Event(
            topic=f"agent_{self.agent_id}_task",
            data=task,
            priority=Priority.HIGH
        ))

        # In production, would return the actual plan from memory/event
        return "Generated plan"

class ExecutorAgent(BaseAgent):
    """Executes plans and returns results."""

    async def _process_task(self, task_data: Dict[str, Any]) -> bool:
        """Execute task plan."""
        start_time = time.time()

        # Simple execution simulation
        result = {
            "status": "completed",
            "output": f"Executed: {task_data['description']}",
            "metrics": {
                "duration": 0.5,
                "resources_used": 2
            }
        }

        processing_time = time.time() - start_time
        self.metrics.update(True, processing_time)

        await self._log_event(
            "execution_completed",
            {
                "task_id": task_data["id"],
                "result": result
            }
        )

        return True

    async def execute(self, task: Dict[str, Any], plan: str) -> Dict[str, Any]:
        """Public method to execute task."""
        if not self._initialized:
            raise RuntimeError("Agent not initialized")

        await self.handle(Event(
            topic=f"agent_{self.agent_id}_task",
            data={"id": task["id"], "description": task["description"], "plan": plan},
            priority=Priority.HIGH
