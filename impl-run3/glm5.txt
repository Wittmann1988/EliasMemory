import asyncio
import json
import sqlite3
import time
import uuid
from dataclasses import dataclass, field, asdict
from datetime import datetime, timedelta
from enum import Enum
from typing import Any, Callable, Dict, List, Optional, AsyncIterator
import logging

logger = logging.getLogger(__name__)

# Ensure WAL mode is supported
sqlite3.enable_callback_tracebacks(True)

class EventPriority(int, Enum):
    LOW = 1
    NORMAL = 5
    HIGH = 10
    CRITICAL = 20

@dataclass(order=True)
class Event:
    sort_index: int = field(init=False)
    id: str = field(default_factory=lambda: str(uuid.uuid4()))
    topic: str = ""
    payload: Dict[str, Any] = field(default_factory=dict)
    priority: EventPriority = EventPriority.NORMAL
    timestamp: str = field(default_factory=lambda: datetime.utcnow().isoformat())
    retries: int = 0
    max_retries: int = 3
    error: Optional[str] = None

    def __post_init__(self):
        self.sort_index = -self.priority.value

    def to_json(self) -> str:
        return json.dumps(asdict(self))

    @staticmethod
    def from_json(data: str) -> 'Event':
        d = json.loads(data)
        d['priority'] = EventPriority(d['priority'])
        return Event(**d)

class EventBus:
    """
    Production-ready Async Pub/Sub with Priority Queue, SQLite Persistence,
    Dead-Letter Queue, Backpressure, and Graceful Shutdown.
    """
    
    def __init__(self, db_path: str = "event_bus.db", max_queue_size: int = 1000):
        self.db_path = db_path
        self._queue: asyncio.PriorityQueue[Event] = asyncio.PriorityQueue(maxsize=max_queue_size)
        self._subscribers: Dict[str, List[Callable[[Event], asyncio.coroutine]]] = {}
        self._running = False
        self._worker_task: Optional[asyncio.Task] = None
        self._metrics = {
            "published": 0,
            "processed": 0,
            "failed": 0,
            "retried": 0,
            "dlq_count": 0
        }
        self._init_db()

    def _init_db(self):
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("PRAGMA journal_mode=WAL;")
            conn.execute("""
                CREATE TABLE IF NOT EXISTS event_log (
                    id TEXT PRIMARY KEY,
                    topic TEXT,
                    payload JSON,
                    priority INTEGER,
                    timestamp TEXT,
                    status TEXT DEFAULT 'pending'
                )
            """)
            conn.execute("""
                CREATE TABLE IF NOT EXISTS dead_letter_queue (
                    id TEXT PRIMARY KEY,
                    topic TEXT,
                    payload JSON,
                    error TEXT,
                    timestamp TEXT
                )
            """)
            conn.commit()

    async def start(self):
        self._running = True
        self._worker_task = asyncio.create_task(self._process_loop())
        logger.info("EventBus started.")

    async def stop(self):
        logger.info("Stopping EventBus...")
        self._running = False
        if self._worker_task:
            self._worker_task.cancel()
            try:
                await self._worker_task
            except asyncio.CancelledError:
                pass
        
        # Persist remaining queue items
        while not self._queue.empty():
            event = await self._queue.get()
            await self._persist_event(event, status='queued_on_shutdown')
        logger.info("EventBus stopped.")

    async def _persist_event(self, event: Event, status: str = 'published'):
        def _sync_persist():
            with sqlite3.connect(self.db_path) as conn:
                conn.execute("PRAGMA journal_mode=WAL;")
                conn.execute(
                    "INSERT OR REPLACE INTO event_log (id, topic, payload, priority, timestamp, status) VALUES (?, ?, ?, ?, ?, ?)",
                    (event.id, event.topic, json.dumps(event.payload), event.priority.value, event.timestamp, status)
                )
                conn.commit()
        await asyncio.to_thread(_sync_persist)

    async def _persist_dlq(self, event: Event, error: str):
        def _sync_dlq():
            with sqlite3.connect(self.db_path) as conn:
                conn.execute("PRAGMA journal_mode=WAL;")
                conn.execute(
                    "INSERT INTO dead_letter_queue (id, topic, payload, error, timestamp) VALUES (?, ?, ?, ?, ?)",
                    (event.id, event.topic, json.dumps(event.payload), error, event.timestamp)
                )
                conn.commit()
        await asyncio.to_thread(_sync_dlq)
        self._metrics["dlq_count"] += 1

    async def publish(self, topic: str, payload: Dict[str, Any], priority: EventPriority = EventPriority.NORMAL) -> str:
        event = Event(topic=topic, payload=payload, priority=priority)
        
        # Backpressure handling: wait if queue is full
        await self._queue.put(event)
        
        self._metrics["published"] += 1
        await self._persist_event(event)
        return event.id

    def subscribe(self, topic: str, handler: Callable[[Event], asyncio.coroutine]):
        if topic not in self._subscribers:
            self._subscribers[topic] = []
        self._subscribers[topic].append(handler)

    async def _process_loop(self):
        while self._running:
            try:
                # Timeout allows checking self._running periodically
                event = await asyncio.wait_for(self._queue.get(), timeout=0.1)
                asyncio.create_task(self._dispatch(event))
            except asyncio.TimeoutError:
                continue
            except Exception as e:
                logger.error(f"Error in process loop: {e}")

    async def _dispatch(self, event: Event):
        handlers = self._subscribers.get(event.topic, [])
        if not handlers:
            # No handlers, consider processed if not critical
            return

        for handler in handlers:
            try:
                await handler(event)
                self._metrics["processed"] += 1
            except Exception as e:
                logger.warning(f"Handler failed for {event.topic}: {e}")
                event.retries += 1
                event.error = str(e)
                
                if event.retries < event.max_retries:
                    # Exponential backoff handled by re-queueing (simple version)
                    await asyncio.sleep(2 ** event.retries)
                    await self._queue.put(event)
                    self._metrics["retried"] += 1
                else:
                    # Move to DLQ
                    await self._persist_dlq(event, str(e))
                    self._metrics["failed"] += 1
                    logger.error(f"Event {event.id} moved to DLQ after {event.retries} retries.")

    async def replay_dlq(self, topic: Optional[str] = None):
        """Replay events from Dead Letter Queue."""
        def _fetch():
            with sqlite3.connect(self.db_path) as conn:
                conn.row_factory = sqlite3.Row
                cur = conn.cursor()
                if topic:
                    cur.execute("SELECT * FROM dead_letter_queue WHERE topic=?", (topic,))
                else:
                    cur.execute("SELECT * FROM dead_letter_queue")
                return [dict(r) for r in cur.fetchall()]
        
        rows = await asyncio.to_thread(_fetch)
        for row in rows:
            # Resubmit
            await self.publish(row['topic'], json.loads(row['payload']), EventPriority.HIGH)
            # Remove from DLQ (simple logic, should be transactional in prod)
            def _delete():
                with sqlite3.connect(self.db_path) as conn:
                    conn.execute("DELETE FROM dead_letter_queue WHERE id=?", (row['id'],))
                    conn.commit()
            await asyncio.to_thread(_delete)
            
    def get_metrics(self) -> Dict[str, int]:
        return self._metrics.copy()

import asyncio
import time
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Dict, Any, List, Optional
import logging

logger = logging.getLogger(__name__)

@dataclass
class AgentMetrics:
    tasks_completed: int = 0
    tasks_failed: int = 0
    total_processing_time: float = 0.0
    avg_score: float = 0.0
    
    def update(self, success: bool, score: float, duration: float):
        if success:
            self.tasks_completed += 1
            # Rolling average for score
            self.avg_score = ((self.avg_score * (self.tasks_completed - 1)) + score) / self.tasks_completed
        else:
            self.tasks_failed += 1
        self.total_processing_time += duration

class AgentBase(ABC):
    def __init__(self, name: str, event_bus: 'EventBus'):
        self.name = name
        self.event_bus = event_bus
        self.metrics = AgentMetrics()
        self._running = False
        self.logger = logging.getLogger(f"Agent.{name}")

    async def start(self):
        self._running = True
        self.event_bus.subscribe(f"task.assign.{self.name}", self._handle_task)
        self.logger.info(f"Agent {self.name} started.")

    async def stop(self):
        self._running = False
        self.logger.info(f"Agent {self.name} stopped.")

    async def _handle_task(self, event: 'Event'):
        task = event.payload
        start_time = time.time()
        try:
            result, score = await self.process(task)
            duration = time.time() - start_time
            self.metrics.update(success=True, score=score, duration=duration)
            
            await self.event_bus.publish(
                f"task.complete.{self.name}",
                {"task_id": task.get("id"), "result": result, "score": score, "agent": self.name}
            )
        except Exception as e:
            duration = time.time() - start_time
            self.metrics.update(success=False, score=0.0, duration=duration)
            self.logger.error(f"Task {task.get('id')} failed: {e}")
            await self.event_bus.publish(
                f"task.fail.{self.name}",
                {"task_id": task.get("id"), "error": str(e)}
            )

    @abstractmethod
    async def process(self, task: Dict[str, Any]) -> tuple[Dict[str, Any], float]:
        """Process task and return (result_dict, score_0_to_1)"""
        pass

class PlannerAgent(AgentBase):
    def __init__(self, event_bus: 'EventBus'):
        super().__init__("Planner", event_bus)

    async def process(self, task: Dict[str, Any]) -> tuple[Dict[str, Any], float]:
        self.logger.info(f"Planning task: {task.get('description')}")
        # Simulate planning logic
        await asyncio.sleep(0.2)
        plan = {
            "steps": [
                {"action": "retrieve_context", "query": task.get("description")},
                {"action": "execute_code", "code": "print('hello')"},
                {"action": "review_result"}
            ]
        }
        # Score based on complexity handling
        score = 0.9 if len(task.get("description", "")) > 10 else 0.5
        return plan, score

class ExecutorAgent(AgentBase):
    def __init__(self, event_bus: 'EventBus'):
        super().__init__("Executor", event_bus)

    async def process(self, task: Dict[str, Any]) -> tuple[Dict[str, Any], float]:
        plan = task.get("plan", {})
        self.logger.info(f"Executing plan with {len(plan.get('steps', []))} steps")
        
        results = []
        success_count = 0
        for step in plan.get("steps", []):
            # Simulate execution
            await asyncio.sleep(0.1)
            success = True # Random failure simulation could go here
            results.append({"step": step["action"], "status": "success" if success else "fail"})
            if success:
                success_count += 1
        
        score = success_count / len(plan.get("steps", [1]))
        return {"execution_log": results}, score

class ReviewerAgent(AgentBase):
    def __init__(self, event_bus: 'EventBus'):
        super().__init__("Reviewer", event_bus)

    async def process(self, task: Dict[str, Any]) -> tuple[Dict[str, Any], float]:
        execution_log = task.get("execution_log", [])
        self.logger.info(f"Reviewing execution log.")
        
        # Simple heuristic scoring
        failures = sum(1 for r in execution_log if r["status"] == "fail")
        score = max(0.0, 1.0 - (failures * 0.25))
        
        return {"approved": score > 0.7, "feedback": "Looks good" if score > 0.7 else "Needs work"}, score
