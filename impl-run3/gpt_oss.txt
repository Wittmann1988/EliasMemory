"""
event_bus.py
------------

Async publish/subscribe event bus with:

* Priority queue per topic
* Dead‑letter queue for failed events
* Metrics collection
* Event persistence to SQLite (WAL mode) with replay capability
* Back‑pressure handling (configurable max queued events)
* Graceful shutdown
"""

from __future__ import annotations

import asyncio
import json
import logging
import sqlite3
import time
from collections import defaultdict
from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Any, Awaitable, Callable, Coroutine, Dict, List, NamedTuple, Tuple

import aiosqlite

logger = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Data structures
# --------------------------------------------------------------------------- #
@dataclass
class Event:
    """A single event that travels through the bus."""
    topic: str
    priority: int
    payload: Dict[str, Any]
    timestamp: float = time.time()
    id: int | None = None  # Assigned on persistence


# --------------------------------------------------------------------------- #
# Helper – exponential back‑off for SQLite retries
# --------------------------------------------------------------------------- #
async def _sqlite_retry(
    coro: Callable[[], Awaitable[Any]],
    attempts: int = 5,
    base: float = 0.5,
) -> Any:
    delay = base
    for attempt in range(attempts):
        try:
            return await coro()
        except (sqlite3.OperationalError, aiosqlite.Error) as exc:
            if attempt == attempts - 1:
                raise
            logger.debug("SQLite retry %s/%s after %.2fs – %s", attempt + 1, attempts, delay, exc)
            await asyncio.sleep(delay)
            delay = min(delay * 2, 4.0)


# --------------------------------------------------------------------------- #
# EventBus implementation
# --------------------------------------------------------------------------- #
class EventBus:
    """
    Central asynchronous event bus.

    Subscriptions are coroutine callbacks that receive an :class:`Event`.
    """

    def __init__(
        self,
        db_path: Path | str = "event_bus.sqlite",
        max_queue: int = 1000,
        shutdown_timeout: float = 5.0,
    ) -> None:
        self._db_path = Path(db_path)
        self._max_queue = max_queue
        self._shutdown_timeout = shutdown_timeout

        self._subscribers: Dict[str, List[Callable[[Event], Awaitable[None]]]] = defaultdict(list)
        self._queues: Dict[str, asyncio.PriorityQueue[Tuple[int, Event]]] = defaultdict(
            lambda: asyncio.PriorityQueue()
        )
        self._dead_letter: List[Event] = []
        self._metrics: Dict[str, int] = defaultdict(int)

        self._running = False
        self._worker_tasks: List[asyncio.Task[None]] = []

    # ------------------------------------------------------------------- #
    # Public API
    # ------------------------------------------------------------------- #
    async def start(self) -> None:
        """Initialise DB and start background workers."""
        await self._init_db()
        self._running = True
        for topic in self._queues:
            self._worker_tasks.append(asyncio.create_task(self._worker(topic)))
        logger.info("EventBus started with %d topics", len(self._queues))

    async def stop(self) -> None:
        """Graceful shutdown – wait for queued events to be processed."""
        self._running = False
        logger.info("Shutting down EventBus, waiting up to %.1fs", self._shutdown_timeout)
        await asyncio.wait_for(
            asyncio.gather(*self._worker_tasks, return_exceptions=True),
            timeout=self._shutdown_timeout,
        )
        await self._persist_dead_letter()
        logger.info("EventBus stopped")

    async def publish(
        self,
        topic: str,
        payload: Dict[str, Any],
        priority: int = 0,
        *,
        timeout: float = 2.0,
    ) -> None:
        """
        Publish an event.

        If the internal queue for the topic exceeds ``max_queue`` the call
        back‑pressures until space becomes available (or ``timeout`` expires).
        """
        if not self._running:
            raise RuntimeError("EventBus not started")

        queue = self._queues[topic]
        if queue.qsize() >= self._max_queue:
            # back‑pressure – wait for space
            await asyncio.wait_for(self._wait_for_queue_space(queue), timeout=timeout)

        ev = Event(topic=topic, priority=priority, payload=payload)
        await self._persist_event(ev)  # store before dispatch
        await queue.put((priority, ev))
        self._metrics["published"] += 1
        logger.debug("Published event %s to %s (prio=%d)", ev.id, topic, priority)

    def subscribe(self, topic: str, callback: Callable[[Event], Awaitable[None]]) -> None:
        """Register a coroutine callback for a specific topic."""
        self._subscribers[topic].append(callback)
        # ensure a queue exists for the topic
        _ = self._queues[topic]

    async def replay(self, from_timestamp: float = 0.0) -> None:
        """
        Replay persisted events from the DB whose timestamp >= ``from_timestamp``.
        Events are re‑published in their original priority order.
        """
        async with aiosqlite.connect(self._db_path) as db:
            await db.execute("PRAGMA journal_mode=WAL;")
            async with db.execute(
                "SELECT id, topic, priority, payload, timestamp FROM events "
                "WHERE timestamp >= ? ORDER BY timestamp ASC;",
                (from_timestamp,),
            ) as cursor:
                async for row in cursor:
                    ev = Event(
                        id=row[0],
                        topic=row[1],
                        priority=row[2],
                        payload=json.loads(row[3]),
                        timestamp=row[4],
                    )
                    await self.publish(ev.topic, ev.payload, ev.priority)

    # ------------------------------------------------------------------- #
    # Internal helpers
    # ------------------------------------------------------------------- #
    async def _wait_for_queue_space(self, queue: asyncio.PriorityQueue[Tuple[int, Event]]) -> None:
        """Yield control until the queue size drops below ``max_queue``."""
        while queue.qsize() >= self._max_queue:
            await asyncio.sleep(0.05)

    async def _worker(self, topic: str) -> None:
        """Consume events for a single topic and dispatch to subscribers."""
        queue = self._queues[topic]
        while self._running or not queue.empty():
            try:
                priority, ev = await asyncio.wait_for(queue.get(), timeout=0.5)
            except asyncio.TimeoutError:
                continue

            callbacks = self._subscribers.get(topic, [])
            if not callbacks:
                # no consumer – move to dead‑letter
                self._dead_letter.append(ev)
                self._metrics["dead_letter"] += 1
                continue

            for cb in callbacks:
                try:
                    await asyncio.wait_for(cb(ev), timeout=2.0)
                except Exception as exc:  # pylint: disable=broad-except
                    logger.exception("Subscriber error, moving event %s to dead‑letter", ev.id)
                    self._dead_letter.append(ev)
                    self._metrics["dead_letter"] += 1
            self._metrics["dispatched"] += 1
            queue.task_done()

    async def _init_db(self) -> None:
        """Create tables if they do not exist; enable WAL mode."""
        async with aiosqlite.connect(self._db_path) as db:
            await db.execute("PRAGMA journal_mode=WAL;")
            await db.execute(
                """
                CREATE TABLE IF NOT EXISTS events (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    topic TEXT NOT NULL,
                    priority INTEGER NOT NULL,
                    payload TEXT NOT NULL,
                    timestamp REAL NOT NULL
                );
                """
            )
            await db.commit()

    async def _persist_event(self, ev: Event) -> None:
        """Insert event into SQLite and assign its autogenerated id."""
        async def _insert() -> int:
            async with aiosqlite.connect(self._db_path) as db:
                cur = await db.execute(
                    "INSERT INTO events (topic, priority, payload, timestamp) "
                    "VALUES (?, ?, ?, ?);",
                    (ev.topic, ev.priority, json.dumps(ev.payload), ev.timestamp),
                )
                await db.commit()
                return cur.lastrowid

        ev.id = await _sqlite_retry(_insert)

    async def _persist_dead_letter(self) -> None:
        """Persist dead‑letter events for later inspection."""
        if not self._dead_letter:
            return
        async with aiosqlite.connect(self._db_path) as db:
            await db.executemany(
                "INSERT INTO events (topic, priority, payload, timestamp) "
                "VALUES (?, ?, ?, ?);",
                [
                    (ev.topic, ev.priority, json.dumps(ev.payload), ev.timestamp)
                    for ev in self._dead_letter
                ],
            )
            await db.commit()
        logger.info("Persisted %d dead‑letter events", len(self._dead_letter))

    # ------------------------------------------------------------------- #
    # Metrics exposure
    # ------------------------------------------------------------------- #
    def get_metrics(self) -> Dict[str, int]:
        """Return a snapshot of internal counters."""
        return dict(self._metrics)


__all__ = ["Event", "EventBus"]

"""
memory_store.py
---------------

Vector‑augmented SQLite memory store using the ``sqlite-vec`` extension
(simulated with a regular table for this demo). Supports:

* Batch embedding via NVIDIA API (mocked here)
* Exponential back‑off + retry on embedding failures
* Configurable memory decay (half‑life)
* Semantic search with Maximal Marginal Relevance (MMR)
* Event emission on add / prune
* Connection pooling via aiosqlite
"""

from __future__ import annotations

import asyncio
import json
import logging
import math
import random
import time
from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Any, Dict, List, Tuple

import aiosqlite
import numpy as np

from .event_bus import EventBus, Event

logger = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Mock NVIDIA embedding API
# --------------------------------------------------------------------------- #
async def _nvidia_embed_batch(texts: List[str]) -> List[List[float]]:
    """
    Mocked async call to NVIDIA embedding service.
    Returns a 768‑dimensional vector per text.
    Random failures are injected to demonstrate retry logic.
    """
    await asyncio.sleep(0.1)  # network latency simulation
    if random.random() < 0.1:  # 10% chance of transient failure
        raise RuntimeError("Transient NVIDIA service error")
    return [np.random.rand(768).tolist() for _ in texts]


# --------------------------------------------------------------------------- #
# Data structures
# --------------------------------------------------------------------------- #
@dataclass
class MemoryEntry:
    """A single memory record."""
    id: int | None
    text: str
    vector: List[float]
    timestamp: float = time.time()


# --------------------------------------------------------------------------- #
# Helper – exponential back‑off for embedding retries
# --------------------------------------------------------------------------- #
async def _embed_with_backoff(texts: List[str]) -> List[List[float]]:
    delay = 0.5
    attempts = 0
    while True:
        try:
            return await _nvidia_embed_batch(texts)
        except Exception as exc:  # pylint: disable=broad-except
            attempts += 1
            if attempts >= 5:
                raise RuntimeError(f"Embedding failed after {attempts} attempts") from exc
            logger.debug("Embedding retry %s after %.2fs – %s", attempts, delay, exc)
            await asyncio.sleep(delay)
            delay = min(delay * 2, 4.0)


# --------------------------------------------------------------------------- #
# MemoryStore implementation
# --------------------------------------------------------------------------- #
class MemoryStore:
    """
    SQLite‑backed vector memory with decay and semantic search.
    Emits ``memory.added`` and ``memory.pruned`` events on the provided bus.
    """

    def __init__(
        self,
        bus: EventBus,
        db_path: Path | str = "memory_store.sqlite",
        decay_half_life: float = 86400.0,  # 1 day
        batch_size: int = 32,
    ) -> None:
        self.bus = bus
        self.db_path = Path(db_path)
        self.decay_half_life = decay_half_life
        self.batch_size = batch_size
        self._pool: asyncio.Queue[aiosqlite.Connection] = asyncio.Queue()
        self._init_task = asyncio.create_task(self._init_pool())

    # ------------------------------------------------------------------- #
    # Public API
    # ------------------------------------------------------------------- #
    async def add_texts(self, texts: List[str]) -> List[int]:
        """
        Embed and store a batch of texts. Returns list of row ids.
        """
        await self._init_task
        vectors = await _embed_with_backoff(texts)
        ids: List[int] = []
        async with self._get_conn() as conn:
            async with conn.executemany(
                "INSERT INTO memories (text, vector, timestamp) VALUES (?, ?, ?);",
                [(t, json.dumps(v), time.time()) for t, v in zip(texts, vectors)],
            ) as cur:
                await conn.commit()
                start_id = cur.lastrowid - len(texts) + 1
                ids = list(range(start_id, start_id + len(texts)))

        for txt, vec, mid in zip(texts, vectors, ids):
            await self.bus.publish(
                "memory.added",
                {"id": mid, "text": txt, "vector": vec},
                priority=1,
            )
        return ids

    async def semantic_search(
        self,
        query: str,
        k: int = 5,
        mmr_lambda: float = 0.6,
    ) -> List[MemoryEntry]:
        """
        Perform MMR‑based semantic search.

        Returns up to ``k`` entries ordered by relevance/diversity.
        """
        await self._init_task
        query_vec = (await _embed_with_backoff([query]))[0]

        async with self._get_conn() as conn:
            async with conn.execute("SELECT id, text, vector, timestamp FROM memories;") as cur:
                rows = await cur.fetchall()

        # Convert vectors
        entries = [
            MemoryEntry(
                id=row[0],
                text=row[1],
                vector=json.loads(row[2]),
                timestamp=row[3],
            )
            for row in rows
        ]

        # Compute cosine similarity
        def cos(a: np.ndarray, b: np.ndarray) -> float:
            return float(np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b) + 1e-10))

        query_np = np.array(query_vec)
        similarities = [(e, cos(query_np, np.array(e.vector))) for e in entries]
        selected: List[Tuple[MemoryEntry, float]] = []

        while similarities and len(selected) < k:
            if not selected:
                # pick highest similarity
                best = max(similarities, key=lambda x: x[1])
                selected.append(best)
                similarities.remove(best)
                continue
            # MMR selection
            mmr_scores = []
            for cand, sim in similarities:
                diversity = max(cos(np.array(cand.vector), np.array(sel[0].vector)) for sel in selected)
                mmr = mmr_lambda * sim - (1 - mmr_lambda) * diversity
                mmr_scores.append((cand, sim, mmr))
            best = max(mmr_scores, key=lambda x: x[2])
            selected.append((best[0], best[1]))
            similarities = [(c, s) for c, s in similarities if c.id != best[0].id]

        return [MemoryEntry(id=e.id, text=e.text, vector=e.vector, timestamp=e.timestamp) for e, _ in selected]

    async def decay(self) -> int:
        """
        Remove memories older than the configured half‑life.
        Returns number of rows deleted.
        """
        await self._init_task
        cutoff = time.time() - self.decay_half_life * math.log(2)  # half‑life decay
        async with self._get_conn() as conn:
            cur = await conn.execute("DELETE FROM memories WHERE timestamp < ?;", (cutoff,))
            await conn.commit()
            deleted = cur.rowcount
        if deleted:
            await self.bus.publish(
                "memory.pruned",
                {"deleted": deleted, "cutoff": cutoff},
                priority=0,
            )
        return deleted

    # ------------------------------------------------------------------- #
    # Internal helpers
    # ------------------------------------------------------------------- #
    async def _init_pool(self) -> None:
        """Create a small pool of persistent connections (size 2)."""
        for _ in range(2):
            conn = await aiosqlite.connect(self.db_path)
            await conn.execute("PRAGMA journal_mode=WAL;")
            await conn.execute(
                """
                CREATE TABLE IF NOT EXISTS memories (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    text TEXT NOT NULL,
                    vector TEXT NOT NULL,
                    timestamp REAL NOT NULL
                );
                """
            )
            await conn.commit()
            await self._pool.put(conn)

    async def _get_conn(self) -> aiosqlite.Connection:
        """Acquire a connection from the pool (async context manager)."""
        conn = await self._pool.get()
        class _ConnCtx:
            async def __aenter__(self_inner):
                return conn
            async def __aexit__(self_inner, exc_type, exc, tb):
                await self._pool.put(conn)
        return _ConnCtx()


__all__ = ["MemoryEntry", "MemoryStore"]

"""
agent_base.py
-------------

Base classes for all agents in the framework.

* ``BaseAgent`` – lifecycle handling, async execution, metrics, bus subscription.
* ``PlannerAgent`` – decomposes a high‑level task into subtasks.
* ``ExecutorAgent`` – runs a subtask (simulated with a sleep & dummy result).
* ``ReviewerAgent`` – validates executor output, returns a score between 0‑1.
"""

from __future__ import annotations

import asyncio
import json
import logging
import time
import uuid
from dataclasses import dataclass, field
from typing import Any, Awaitable, Callable, Dict, List, Tuple

from .event_bus import EventBus, Event

logger = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Helper – simple metric collector
# --------------------------------------------------------------------------- #
class Metrics:
    def __init__(self) -> None:
        self.counters: Dict[str, int] = {}

    def inc(self, name: str, amount: int = 1) -> None:
        self.counters[name] = self.counters.get(name, 0) + amount

    def snapshot(self) -> Dict[str, int]:
        return dict(self.counters)


# --------------------------------------------------------------------------- #
# BaseAgent
# --------------------------------------------------------------------------- #
@dataclass
class BaseAgent:
    """Common functionality for all agents."""
    name: str
    bus: EventBus
    state: str = field(default="init", init=False)
    metrics: Metrics = field(default_factory=Metrics, init=False)

    async def start(self) -> None:
        """Transition to ready state and subscribe to relevant topics."""
        self.state = "ready"
        self.bus.subscribe(f"{self.name}.request", self._handle_request)
        logger.info("%s started and listening", self.name)

    async def stop(self) -> None:
        """Mark agent as stopped."""
        self.state = "stopped"
        logger.info("%s stopped", self.name)

    async def _handle_request(self, ev: Event) -> None:
        """Dispatch incoming request events to the concrete implementation."""
        if self.state != "ready":
            logger.warning("%s received request while not ready", self.name)
            return
        self.state = "busy"
        try:
            result = await asyncio.wait_for(self.process(ev.payload), timeout=10.0)
            await self.bus.publish(
                f"{self.name}.response",
                {"request_id": ev.payload.get("request_id"), "result": result},
                priority=1,
            )
            self.metrics.inc("processed")
        except Exception as exc:  # pylint: disable=broad-except
            logger.exception("%s failed to process request", self.name)
            await self.bus.publish(
                f"{self.name}.error",
                {"request_id": ev.payload.get("request_id"), "error": str(exc)},
                priority=0,
            )
            self.metrics.inc("errors")
        finally:
            self.state = "ready"

    async def process(self, payload: Dict[str, Any]) -> Any:
        """Override in subclasses – perform the core work."""
        raise NotImplementedError


# --------------------------------------------------------------------------- #
# PlannerAgent
# --------------------------------------------------------------------------- #
class PlannerAgent(BaseAgent):
    """Splits a high‑level task into a list of sub‑tasks."""

    async def process(self, payload: Dict[str, Any]) -> List[Dict[str, Any]]:
        task_description = payload.get("task")
        if not task_description:
            raise ValueError("Missing 'task' in payload")
        # Very naive decomposition: split by sentences.
        subtasks = [
            {"subtask_id": str(uuid.uuid4()), "instruction": s.strip()}
            for s in task_description.split(".")
            if s.strip()
        ]
        logger.debug("%s decomposed task into %d subtasks", self.name, len(subtasks))
        return subtasks


# --------------------------------------------------------------------------- #
# ExecutorAgent
# --------------------------------------------------------------------------- #
class ExecutorAgent(BaseAgent):
    """Executes a sub‑task (simulated)."""

    async def process(self, payload: Dict[str, Any]) -> Dict[str, Any]:
        instruction = payload.get("instruction")
        if not instruction:
            raise ValueError("Missing 'instruction'")
        # Simulate work
        await asyncio.sleep(0.5)
        result = {"output": f"Executed: {instruction}", "timestamp": time.time()}
        logger.debug("%s executed subtask %s", self.name, payload.get("subtask_id"))
        return result


# --------------------------------------------------------------------------- #
# ReviewerAgent
# --------------------------------------------------------------------------- #
class ReviewerAgent(BaseAgent):
    """Validates executor output and returns a quality score."""

    async def process(self, payload: Dict[str, Any]) -> Dict[str, Any]:
        result = payload.get("result")
        if not result:
            raise ValueError("Missing 'result'")
        # Very naive scoring: length of output string normalized.
        output = result.get("output", "")
        score = min(len(output) / 100.0, 1.0)
        logger.debug("%s scored output %.2f", self.name, score)
        return {"score": score, "details": "Length‑based heuristic"}

"""
orchestrator.py
---------------

Coordinates the overall workflow using LangGraph's ``StateGraph``.
The graph follows the pattern:

plan → act → observe → improve → checkpoint

The orchestrator also:

* Routes events via the EventBus.
* Adjusts routing weights based on agent success rates (self‑improvement).
* Persists checkpoints to SQLite.
"""

from __future__ import annotations

import asyncio
import json
import logging
import sqlite3
import time
import uuid
from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Any, Dict, List, Tuple

import aiosqlite
from langgraph.graph import StateGraph, START, END

from .event_bus import EventBus, Event
from .agent_base import PlannerAgent, ExecutorAgent, ReviewerAgent

logger = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# State definition
# --------------------------------------------------------------------------- #
@dataclass
class OrchestratorState:
    """State carried through the LangGraph execution."""
    task_id: str
    task_description: str
    subtasks: List[Dict[str, Any]] | None = None
    results: List[Dict[str, Any]] | None = None
    scores: List[float] | None = None
    checkpoint: int = 0  # increments after each major step


# --------------------------------------------------------------------------- #
# Orchestrator implementation
# --------------------------------------------------------------------------- #
class Orchestrator:
    """
    Core orchestrator that wires agents together, persists checkpoints,
    and adapts routing weights based on observed success rates.
    """

    def __init__(
        self,
        bus: EventBus,
        planner: PlannerAgent,
        executor: ExecutorAgent,
        reviewer: ReviewerAgent,
        db_path: Path | str = "orchestrator.sqlite",
    ) -> None:
        self.bus = bus
        self.planner = planner
        self.executor = executor
        self.reviewer = reviewer
        self.db_path = Path(db_path)

        # routing weights (simple dict, sum to 1)
        self.routing_weights: Dict[str, float] = {
            "planner": 0.3,
            "executor": 0.5,
            "reviewer": 0.2,
        }

        self._graph = self._build_graph()
        self._checkpoint_lock = asyncio.Lock()

    # ------------------------------------------------------------------- #
    # Public API
    # ------------------------------------------------------------------- #
    async def start(self) -> None:
        """Start agents and load latest checkpoint (if any)."""
        await asyncio.gather(self.planner.start(), self.executor.start(), self.reviewer.start())
        await self._load_latest_checkpoint()
        logger.info("Orchestrator ready")

    async def stop(self) -> None:
        """Stop all agents."""
        await asyncio.gather(self.planner.stop(), self.executor.stop(), self.reviewer.stop())

    async def submit_task(self, description: str) -> str:
        """Create a new task and run it through the graph."""
        task_id = str(uuid.uuid4())
        init_state = OrchestratorState(task_id=task_id, task_description=description)
        await self._graph.ainvoke(init_state)
        return task_id

    # ------------------------------------------------------------------- #
    # Graph construction
    # ------------------------------------------------------------------- #
    def _build_graph(self) -> StateGraph[OrchestratorState]:
        graph = StateGraph(OrchestratorState)

        @graph.add_node("plan")
        async def plan(state: OrchestratorState) -> OrchestratorState:
            payload = {"request_id": state.task_id, "task": state.task_description}
            await self.bus.publish("PlannerAgent.request", payload, priority=1)
            # wait for response
            resp = await self._wait_for_event(f"PlannerAgent.response", state.task_id)
            state.subtasks = resp["result"]
            await self._checkpoint(state, "plan")
            return state

        @graph.add_node("act")
        async def act(state: OrchestratorState) -> OrchestratorState:
            results: List[Dict[str, Any]] = []
            for sub in state.subtasks or []:
                payload = {"request_id": sub["subtask_id"], "instruction": sub["instruction"]}
                await self.bus.publish("ExecutorAgent.request", payload, priority=1)
                resp = await self._wait_for_event("ExecutorAgent.response", sub["subtask_id"])
                results.append(resp["result"])
            state.results = results
            await self._checkpoint(state, "act")
            return state

        @graph.add_node("observe")
        async def observe(state: OrchestratorState) -> OrchestratorState:
            scores: List[float] = []
            for sub, res in zip(state.subtasks or [], state.results or []):
                payload = {"request_id": sub["subtask_id"], "result": res}
                await self.bus.publish("ReviewerAgent.request", payload, priority=1)
                resp = await self._wait_for_event("ReviewerAgent.response", sub["subtask_id"])
                scores.append(resp["result"]["score"])
            state.scores = scores
            await self._checkpoint(state, "observe")
            return state

        @graph.add_node("improve")
        async def improve(state: OrchestratorState) -> OrchestratorState:
            # Simple self‑improvement: boost executor weight if avg score > 0.7
            avg_score = sum(state.scores or []) / max(len(state.scores or []), 1)
            if avg_score > 0.7:
                self.routing_weights["executor"] = min(self.routing_weights["executor"] + 0.05, 0.7)
                self.routing_weights["planner"] = max(self.routing_weights["planner"] - 0.025, 0.1)
                self.routing_weights["reviewer"] = max(self.routing_weights["reviewer"] - 0.025, 0.1)
                logger.info("Improved routing weights: %s", self.routing_weights)
            await self._checkpoint(state, "improve")
            return state

        @graph.add_node("checkpoint")
        async def checkpoint(state: OrchestratorState) -> OrchestratorState:
            # final checkpoint (no‑op)
            await self._checkpoint(state, "final")
            return state

        # Define edges
        graph.add_edge(START, "plan")
        graph.add_edge("plan", "act")
        graph.add_edge("act", "observe")
        graph.add_edge("observe", "improve")
        graph.add_edge("improve", "checkpoint")
        graph.add_edge("checkpoint", END)

        return graph

    # ------------------------------------------------------------------- #
    # Helpers
    # ------------------------------------------------------------------- #
    async def _wait_for_event(self, topic: str, request_id: str, timeout: float = 5.0) -> Dict[str, Any]:
        """
        Wait for a single event on ``topic`` whose payload contains ``request_id``.
        """
        fut: asyncio.Future[Dict[str, Any]] = asyncio.get_event_loop().create_future()

        async def _listener(ev: Event) -> None:
            if ev.payload.get("request_id") == request_id:
                fut.set_result(ev.payload)

        self.bus.subscribe(topic, _listener)
        try:
            return await asyncio.wait_for(fut, timeout=timeout)
        finally:
            # remove listener to avoid memory leak
            self.bus._subscribers[topic].remove(_listener)  # type: ignore

    async def _checkpoint(self, state: OrchestratorState, stage: str) -> None:
        """Persist a checkpoint row."""
        async with self._checkpoint_lock:
            async with aiosqlite.connect(self.db_path) as db:
                await db.execute(
                    """
                    CREATE TABLE IF NOT EXISTS checkpoints (
                        id INTEGER PRIMARY KEY AUTOINCREMENT,
                        task_id TEXT NOT NULL,
                        stage TEXT NOT NULL,
                        state TEXT NOT NULL,
                        ts REAL NOT NULL
                    );
                    """
                )
                await db.execute(
                    "INSERT INTO checkpoints (task_id, stage, state, ts) VALUES (?, ?, ?, ?);",
                    (state.task_id, stage, json.dumps(asdict(state)), time.time()),
                )
                await db.commit()
                state.checkpoint += 1
                logger.debug("Checkpoint saved for %s at stage %s", state.task_id, stage)

    async def _load_latest_checkpoint(self) -> None:
        """Placeholder – in a real system we would restore in‑flight tasks."""
        async with aiosqlite.connect(self.db_path) as db:
            await db.execute("PRAGMA journal_mode=WAL;")
            await db.commit()
        logger.debug("Orchestrator DB initialised (no restoration logic needed for demo)")

    # ------------------------------------------------------------------- #
    # Inspection
    # ------------------------------------------------------------------- #
    def get_routing_weights(self) -> Dict[str, float]:
        """Return a copy of the current routing weights."""
        return dict(self.routing_weights)

"""
self_improver.py
----------------

Background engine that analyses system metrics and performs concrete
adjustments:

* Adjusts agent routing weights (via Orchestrator)
* Prunes stale memories (via MemoryStore)
* Tweaks embedding batch size based on throughput
* Emits ``self_improve`` events for observability
"""

from __future__ import annotations

import asyncio
import logging
import statistics
from typing import Dict, List

from .event_bus import EventBus
from .memory_store import MemoryStore
from .orchestrator import Orchestrator

logger = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# SelfImprover implementation
# --------------------------------------------------------------------------- #
class SelfImprover:
    """
    Periodic self‑improvement loop.

    The ``run`` coroutine should be scheduled as a background task.
    """

    def __init__(
        self,
        bus: EventBus,
        orchestrator: Orchestrator,
        memory: MemoryStore,
        interval: float = 10.0,
    ) -> None:
        self.bus = bus
        self.orchestrator = orchestrator
        self.memory = memory
        self.interval = interval
        self._task: asyncio.Task[None] | None = None
        self._running = False

    async def start(self) -> None:
        """Launch the periodic loop."""
        if self._running:
            return
        self._running = True
        self._task = asyncio.create_task(self._loop())
        logger.info("SelfImprover started (interval %.1fs)", self.interval)

    async def stop(self) -> None:
        """Terminate the loop."""
        self._running = False
        if self._task:
            await self._task
        logger.info("SelfImprover stopped")

    # ------------------------------------------------------------------- #
    # Core loop
    # ------------------------------------------------------------------- #
    async def _loop(self) -> None:
        while self._running:
            try:
                await self._analyze_and_improve()
            except Exception:  # pylint: disable=broad-except
                logger.exception("SelfImprover encountered an error")
            await asyncio.sleep(self.interval)

    async def _analyze_and_improve(self) -> None:
        """
        Gather metrics from the bus and agents, then apply concrete changes.
        """
        # 1. Agent success rates (from orchestrator routing weights)
        weights = self.orchestrator.get_routing_weights()
        avg_weight = sum(weights.values()) / len(weights)

        # Simple heuristic: if any weight deviates >0.1 from average, rebalance.
        rebalance = False
        for name, w in weights.items():
            if abs(w - avg_weight) > 0.1:
                rebalance = True
                break

        if rebalance:
            # equalise weights
            equal = 1.0 / len(weights)
            for name in weights:
                weights[name] = equal
            logger.info("SelfImprover rebalanced routing weights: %s", weights)

        # 2. Memory decay – prune if total rows > 500
        async with self.memory._get_conn() as conn:  # type: ignore
            cur = await conn.execute("SELECT COUNT(*) FROM memories;")
            total = (await cur.fetchone())[0]
        if total > 500:
            deleted = await self.memory.decay()
            logger.info("SelfImprover pruned %d stale memories (total before: %d)", deleted, total)

        # 3. Embedding batch size adaptation based on recent latency
        # (Here we just simulate a change)
        if self.memory.batch_size < 64:
            self.memory.batch_size = min(self.memory.batch_size + 8, 64)
            logger.info("SelfImprover increased embedding batch size to %d", self.memory.batch_size)

        # Emit a summary event
        await self.bus.publish(
            "self_improve",
            {
                "routing_weights": weights,
                "memory_rows": total,
                "embedding_batch": self.memory.batch_size,
            },
            priority=0,
        )


__all__ = ["SelfImprover"]
