    timestamp: float = field(default_factory=time.time)

    headers: Dict[str, str] = field(default_factory=dict)

class EventBusMetrics:
    def __init__(self):

        self.events_processed = 0
        self.events_dropped = 0

        self.topic_counts: Dict[str, int] = defaultdict(int)
        self.priority_counts: Dict[EventPriority, int] = defaultdict(int)

        self.events_published += 1
        self.topic_counts[event.topic] += 1
        self.priority_counts[event.priority] += 1

    def update_processed(self, event: Event, processing_time: float) -> None:
        self.events_processed += 1
        self.processing_times.append(processing_time)

    def update_dropped(self, event: Event, reason: str) -> None:
        self.events_dropped += 1
        self.error_counts[reason] += 1

    def update_queue_size(self, priority: EventPriority, size: int) -> None:

            "avg_processing_time": (

                if self.processing_times
                else 0

            "topic_counts": dict(self.topic_counts),
            "priority_counts": {k.name: v for k, v in self.priority_counts.items()},

class EventBus:
    def __init__(
        self,
        max_queue_size: int = 1000,
        max_concurrent_processors: int = 10,
        persistence_path: str = ":memory:",
        metrics_enabled: bool = True,
    ):
        self.max_queue_size = max_queue_size
        self.max_concurrent_processors = max_concurrent_processors
        self.persistence_path = persistence_path
        self.metrics_enabled = metrics_enabled

        self._queues: Dict[EventPriority, asyncio.Queue] = {
            priority: asyncio.Queue(maxsize=max_queue_size // 4)
            for priority in EventPriority
        }
        self._subscribers: Dict[str, List[Callable[[Event], Any]]] = defaultdict(list)
        self._semaphore = asyncio.Semaphore(max_concurrent_processors)
        self._running = False
        self._processor_task: Optional[asyncio.Task] = None
        self._persistence_task: Optional[asyncio.Task] = None
        self._metrics = EventBusMetrics() if metrics_enabled else None
        self._dead_letter_queue = DeadLetterQueue(persistence_path)
        self._db_pool: Optional[aiosqlite.Connection] = None

        self._init_db()

    async def _init_db(self) -> None:
        self._db_pool = await aiosqlite.connect(self.persistence_path)
        await self._db_pool.execute("PRAGMA journal_mode=WAL")
        await self._db_pool.execute("""
            CREATE TABLE IF NOT EXISTS events (
                event_id TEXT PRIMARY KEY,
                topic TEXT NOT NULL,
                payload TEXT NOT NULL,
                priority TEXT NOT NULL,
                timestamp REAL NOT NULL,
                headers TEXT,
                processed INTEGER NOT NULL DEFAULT 0
            )
        """)
        await self._db_pool.commit()

    async def start(self) -> None:
        if self._running:
            return

        self._running = True
        self._processor_task = asyncio.create_task(self._process_events())
        self._persistence_task = asyncio.create_task(self._persist_events())

    async def stop(self) -> None:
        if not self._running:
            return

        self._running = False
        if self._processor_task:
            await self._processor_task
        if self._persistence_task:
            await self._persistence_task
        if self._db_pool:
            await self._db_pool.close()

    async def _persist_events(self) -> None:
        while self._running:
            try:
                # Persist events from all queues
                for priority in EventPriority:
                    queue = self._queues[priority]
                    if queue.qsize() > 0:
                        events = []
                        while not queue.empty():
                            try:
                                event = queue.get_nowait()
                                events.append(event)
                            except asyncio.QueueEmpty:
                                break

                        if events:
                            async with self._db_pool as conn:
                                for event in events:
                                    await conn.execute(
                                        """
                                        INSERT OR IGNORE INTO events
                                        VALUES (?, ?, ?, ?, ?, ?, 0)
                                        """,
                                        (
                                            event.event_id,
                                            event.topic,
                                            json.dumps(event.payload),
                                            event.priority.name,
                                            event.timestamp,
                                            json.dumps(event.headers),
                                        ),
                                    )
                                await conn.commit()

                await asyncio.sleep(1)
            except Exception as e:
                logger.error(f"Error in persistence task: {e}")
                await asyncio.sleep(5)

    async def _process_events(self) -> None:
        while self._running:
            try:
                # Process events in priority order
                for priority in [
                    EventPriority.CRITICAL,
                    EventPriority.HIGH,
                    EventPriority.NORMAL,
                    EventPriority.LOW,
                ]:
                    queue = self._queues[priority]
                    if queue.qsize() > 0:
                        event = await queue.get()
                        if self.metrics_enabled:
                            self._metrics.update_queue_size(priority, queue.qsize())

                        async with self._semaphore:
                            start_time = time.time()
                            try:
                                await self._dispatch_event(event)
                                if self.metrics_enabled:
                                    self._metrics.update_processed(
                                        event, time.time() - start_time
                                    )
                            except Exception as e:
                                logger.error(
                                    f"Error processing event {event.event_id}: {e}"
                                )
                                event.retry_count += 1
                                if event.retry_count < 3:
                                    await self._queues[priority].put(event)
                                else:
                                    await self._dead_letter_queue.add_event(
                                        event, str(e)
                                    )
                                    if self.metrics_enabled:
                                        self._metrics.update_dropped(
                                            event, "processing_error"
                                        )
                            finally:
                                await self._mark_event_processed(event)
            except Exception as e:
                logger.error(f"Error in event processor: {e}")
                await asyncio.sleep(0.1)

    async def _dispatch_event(self, event: Event) -> None:
        if event.topic in self._subscribers:
            for subscriber in self._subscribers[event.topic]:
                try:
                    if asyncio.iscoroutinefunction(subscriber):
                        await subscriber(event)
                    else:
                        subscriber(event)
                except Exception as e:
                    logger.error(
                        f"Subscriber error for event {event.event_id}: {e}"
                    )
                    raise

    async def _mark_event_processed(self, event: Event) -> None:
        if self._db_pool:
            async with self._db_pool as conn:
                await conn.execute(
                    "UPDATE events SET processed = 1 WHERE event_id = ?",
                    (event.event_id,),
                )
                await conn.commit()

    async def publish(self, event: Event) -> bool:
        if not self._running:
            raise RuntimeError("EventBus is not running")

        if self.metrics_enabled:
            self._metrics.update_published(event)

        queue = self._queues[event.priority]
        if queue.qsize() >= queue.maxsize:
            if self.metrics_enabled:
                self._metrics.update_dropped(event, "queue_full")
            logger.warning(f"Queue full for priority {event.priority}, dropping event")
            return False

        await queue.put(event)
        if self.metrics_enabled:
            self._metrics.update_queue_size(event.priority, queue.qsize())
        return True

    def subscribe(self, topic: str, callback: Callable[[Event], Any]) -> None:
        self._subscribers[topic].append(callback)

    def unsubscribe(self, topic: str, callback: Callable[[Event], Any]) -> None:
        if topic in self._subscribers:
            self._subscribers[topic].remove(callback)

    async def replay_events(
        self,
        topic: Optional[str] = None,
        since: Optional[float] = None,
        until: Optional[float] = None,
    ) -> None:
        query = "SELECT * FROM events WHERE processed = 0"
        params: List[Any] = []

        if topic:
            query += " AND topic = ?"
            params.append(topic)
        if since:
            query += " AND timestamp >= ?"
            params.append(since)
        if until:
            query += " AND timestamp <= ?"
            params.append(until)

        async with self._db_pool.execute(query, params) as cursor:
            async for row in cursor:
                event = Event(
                    event_id=row[0],
                    topic=row[1],
                    payload=json.loads(row[2]),
                    priority=EventPriority[row[3]],
                    timestamp=row[4],
                    headers=json.loads(row[5]) if row[5] else {},
                )
                await self.publish(event)

    async def get_metrics(self) -> Dict[str, Any]:
        if not self.metrics_enabled:
            return {}
        return self._metrics.get_metrics() if self._metrics else {}

    async def get_dead_letters(
        self,
        limit: int = 100,
        topic: Optional[str] = None,
        min_retry_count: int = 0,
    ) -> List[Tuple[Event, str]]:
        return await self._dead_letter_queue.get_events(
            limit, topic, min_retry_count
        )

    async def __aenter__(self):
        await self.start()
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await self.stop()

class MemoryStore:
    def __init__(
        self,
        event_bus: EventBus,

    ):
        self.config = config or MemoryStoreConfig()
        self.event_bus = event_bus
        self._db_pool: Optional[aiosqlite.Connection] = None
        self._embedding_model: Optional[SentenceTransformer] = None
        self._embedding_lock = asyncio.Lock()
        self._init_tasks: List[asyncio.Task] = []

        self._init_db()
        self._init_embedding_model()

    async def initialize(self) -> None:
        """Initialize the memory store and wait for all async tasks to complete."""
        await asyncio.gather(*self._init_tasks)

    def _init_db(self) -> None:
        async def init_db():
            self._db_pool = await aiosqlite.connect(self.config.db_path)
            await self._db_pool.execute("PRAGMA journal_mode=WAL")
            await self._db_pool.execute("""
                CREATE TABLE IF NOT EXISTS memories (
                    memory_id TEXT PRIMARY KEY,
                    content TEXT NOT NULL,
                    metadata TEXT,
                    embedding BLOB,
                    created_at REAL NOT NULL,
                    last_accessed_at REAL NOT NULL,
                    access_count INTEGER NOT NULL,
                    decay_factor REAL NOT NULL,
                    tags TEXT
                )
            """)
            await self._db_pool.execute("""
                CREATE INDEX IF NOT EXISTS idx_memories_created_at
                ON memories(created_at)

            await self._db_pool.execute("""
                CREATE INDEX IF NOT EXISTS idx_memories_accessed_at
                ON memories(last_accessed_at)

            await self._db_pool.commit()

        self._init_tasks.append(asyncio.create_task(init_db()))

    def _init_embedding_model(self) -> None:
        async def load_model():
            with self._embedding_lock:
                if self._embedding_model is None:
                    try:
                        self._embedding_model = SentenceTransformer(
                            self.config.embedding_model,
                            trust_remote_code=True,
                        )
                        if torch.cuda.is_available():
                            self._embedding_model = self._embedding_model.to("cuda")
                    except Exception as e:
                        logger.error(f"Failed to load embedding model: {e}")
                        raise

        self._init_tasks.append(asyncio.create_task(load_model()))

    @retry(
        stop=stop_after_attempt(4),
        wait=wait_exponential(multiplier=1, min=0.5, max=4),
        retry=retry_if_exception_type(Exception),
    )
    async def _generate_embeddings(
        self,
        texts: List[str],
    ) -> List[NDArray[np.float32]]:
        async with self._embedding_lock:
            if self._embedding_model is None:
                raise RuntimeError("Embedding model not initialized")

            try:
                embeddings = await asyncio.to_thread(
                    self._embedding_model.encode,
                    texts,
                    batch_size=self.config.max_batch_size,
                    convert_to_numpy=True,
                    normalize_embeddings=True,
                )
                return [embedding.astype(np.float32) for embedding in embeddings]
            except Exception as e:
                logger.error(f"Embedding generation failed: {e}")
                raise

    async def store(

                    memory.last_accessed_at,
                    memory.access_count,
                    memory.decay_factor,
                    json.dumps(memory.tags),
                ),
            )
            await conn.commit()

        await self.event_bus.publish(
            Event(
                topic="memory:created",
                payload=memory.to_dict(),
                priority=EventPriority.HIGH,
            )
        )

        return memory

    async def retrieve(self, memory_id: str) -> Optional[Memory]:
        async with self._db_pool.execute(

            )

            # Update access stats
            memory.last_accessed_at = time.time()
            memory.access_count += 1
            memory.decay_factor = self._calculate_decay_factor(memory)

                    """,

            await self.event_bus.publish(

    async def search(
        self,
        query: str,
        limit: int = 5,
        tags: Optional[List[str]] = None,
        use_mmr: bool = True,
    ) -> List[SearchResult]:
        query_embedding = await self._generate_embeddings([query])
        query_embedding = query_embedding[0]

        # Build the base query
        base_query = "SELECT * FROM memories"
        params: List[Any] = []

        if tags:
            placeholders = ",".join(["?"] * len(tags))
            base_query += f" WHERE tags LIKE '%' || ? || '%'"
            params.extend(tags)

        # Get all candidate memories
        async with self._db_pool.execute(base_query, params) as cursor:
            rows = await cursor.fetchall()

        if not rows:
            return []

        # Calculate similarity scores
        memories = []
        for row in rows:
            memory = Memory(
                memory_id=row[0],
                content=row[1],
                metadata=json.loads(row[2]) if row[2] else {},
                embedding=np.frombuffer(row[3], dtype=np.float32),
                created_at=row[4],
                last_accessed_at=row[5],
                access_count=row[6],
                decay_factor=row[7],
                tags=json.loads(row[8]) if row[8] else [],
            )
            similarity = np.dot(query_embedding, memory.embedding)
            memories.append((memory, similarity))

        if not use_mmr:
            # Simple top-k selection
            memories.sort(key=lambda x: x[1], reverse=True)
            results = [
                SearchResult(memory=mem, score=score, rank=i)
                for i, (mem, score) in enumerate(memories[:limit])
            ]
        else:
            # Maximal Marginal Relevance selection
            results = self._mmr_selection(
                query_embedding,
                memories,
                limit,
                self.config.mmr_lambda,
            )

        # Emit search event
        await self.event_bus.publish(
            Event(
                topic="memory:searched",
                payload={
                    "query": query,
                    "results": [r.memory.memory_id for r in results],
                    "count": len(results),
                },
                priority=EventPriority.NORMAL,
            )
        )

        return results

    def _mmr_selection(
        self,
        query_embedding: NDArray[np.float32],
        memories: List[Tuple[Memory, float]],
        limit: int,
        lambda_param: float,
    ) -> List[SearchResult]:
        if not memories:

        remaining = memories.copy()

        # Select first memory (most similar)
        remaining.sort(key=lambda x: x[1], reverse=True)
        first_mem, first_score = remaining.pop(0)
        selected.append(SearchResult(memory=first_mem, score=first_score, rank=0))

        while len(selected) < limit and remaining:
            max_mmr = -1
            best_idx = 0
            best_mem = None
            best_score = 0.0

            for i, (mem, score) in enumerate(remaining):
                # Calculate MMR score
                relevance = score
                diversity = 0.0
                for sel in selected:
                    diversity += 1 - np.dot(mem.embedding, sel.memory.embedding)

                mmr_score = lambda_param * relevance - (1 - lambda_param) * diversity

                if mmr_score > max_mmr:
                    max_mmr = mmr_score
                    best_idx = i
                    best_mem = mem
                    best_score = score

            if best_mem:
                selected.append(
                    SearchResult(
                        memory=best_mem,
                        score=best_score,
                        rank=len(selected),
                    )
                )
                remaining.pop(best_idx)

        return selected

    def _calculate_decay_factor(self, memory: Memory) -> float:
        """Calculate decay factor based on half-life and access patterns."""
        time_elapsed = time.time() - memory.last_accessed_at
        time_factor = 2 ** (-time_elapsed / self.config.decay_half_life)

        # Boost for frequently accessed memories
        access_factor = 1 + (memory.access_count ** 0.5) * 0.1

        return time_factor * access_factor

    async def decay_memories(self) -> None:
        """Apply decay to all memories and remove those below threshold."""
        async with self._db_pool.execute(
            "SELECT memory_id, decay_factor FROM memories"
        ) as cursor:
            rows = await cursor.fetchall()

        to_delete = []
        for row in rows:
            memory_id, decay_factor = row
            if decay_factor < 0.1:  # Threshold for deletion
                to_delete.append(memory_id)

        if to_delete:
            async with self._db_pool as conn:
                await conn.executemany(
                    "DELETE FROM memories WHERE memory_id = ?",
                    [(mem_id,) for mem_id in to_delete],
                )
                await conn.commit()

            await self.event_bus.publish(
                Event(
                    topic="memory:pruned",
                    payload={"count": len(to_delete), "memory_ids": to_delete},
                    priority=EventPriority.NORMAL,
                )
            )

    async def get_memory_stats(self) -> Dict[str, Any]:
        async with self._db_pool.execute(
            "SELECT COUNT(*), AVG(decay_factor) FROM memories"
        ) as cursor:
            count, avg_decay = await cursor.fetchone()

        async with self._db_pool.execute(
            "SELECT MIN(created_at), MAX(created_at) FROM memories"
        ) as cursor:
            min_created, max_created = await cursor.fetchone()

        return {
            "total_memories": count,
            "avg_decay_factor": avg_decay,
            "oldest_memory_age": time.time() - max_created if max_created else 0,
            "newest_memory_age": time.time() - min_created if min_created else 0,
        }

    async def close(self) -> None:
        if self._db_pool:
            await self._db_pool.close()
        if self._embedding_model:
            await asyncio.to_thread(self._embedding_model.to, "cpu")
