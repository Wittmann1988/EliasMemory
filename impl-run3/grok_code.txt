"""
Event Bus module for asynchronous pub/sub with priority queue, dead-letter handling, metrics, persistence, replay, backpressure, and graceful shutdown.
"""
import asyncio
import sqlite3
import atexit
import logging
from typing import Any, Callable, Dict, List, Optional, Tuple, Union
from collections import defaultdict
import threading
from dataclasses import dataclass, field
from datetime import datetime, timedelta

logger = logging.getLogger(__name__)

__all__ = ["EventBus", "Event", "Priority", "DeadLetterQueue", "MetricsCollector", "EventPersistence", "BackpressureManager", "ReplayManager"]

@dataclass
class Event:
    """Represents an event in the system."""
    id: str
    topic: str
    data: Dict[str, Any]
    priority: int = 0
    timestamp: datetime = field(default_factory=datetime.now)
    retries: int = 0

class Priority:
    HIGH = 2
    NORMAL = 1
    LOW = 0

class DeadLetterQueue:
    """Handles events that fail processing after retries."""
    def __init__(self, db_path: str = "event_bus.db"):
        self.db_path = db_path
        self._init_db()

    def _init_db(self):
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("PRAGMA journal_mode=WAL")
            conn.execute("""
                CREATE TABLE IF NOT EXISTS dead_letters (
                    id TEXT PRIMARY KEY,
                    topic TEXT,
                    data TEXT,
                    priority INTEGER,
                    timestamp TEXT,
                    retries INTEGER
                )
            """)

    def add(self, event: Event):
        """Add an event to the dead-letter queue."""
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("""
                INSERT INTO dead_letters VALUES (?, ?, ?, ?, ?, ?)
            """, (event.id, event.topic, str(event.data), event.priority, event.timestamp.isoformat(), event.retries))
        logger.warning(f"Event {event.id} added to dead-letter queue")

class MetricsCollector:
    """Collects metrics for the event bus."""
    def __init__(self):
        self.metrics: Dict[str, int] = defaultdict(int)

    def increment(self, metric: str):
        self.metrics[metric] += 1

    def get(self, metric: str) -> int:
        return self.metrics[metric]

class EventPersistence:
    """Persists events to SQLite for replay."""
    def __init__(self, db_path: str = "event_bus.db"):
        self.db_path = db_path
        self._init_db()

    def _init_db(self):
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("PRAGMA journal_mode=WAL")
            conn.execute("""
                CREATE TABLE IF NOT EXISTS events (
                    id TEXT PRIMARY KEY,
                    topic TEXT,
                    data TEXT,
                    priority INTEGER,
                    timestamp TEXT,
                    processed INTEGER DEFAULT 0
                )
            """)

    def save(self, event: Event):
        """Save an event."""
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("""
                INSERT INTO events VALUES (?, ?, ?, ?, ?, 0)
            """, (event.id, event.topic, str(event.data), event.priority, event.timestamp.isoformat()))

    def mark_processed(self, event_id: str):
        """Mark an event as processed."""
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("UPDATE events SET processed=1 WHERE id=?", (event_id,))

class BackpressureManager:
    """Manages backpressure with configurable limits."""
    def __init__(self, max_queue_size: int = 1000):
        self.max_queue_size = max_queue_size
        self.current_size = 0

    def can_enqueue(self) -> bool:
        return self.current_size < self.max_queue_size

    def enqueue(self):
        if self.can_enqueue():
            self.current_size += 1
            return True
        return False

    def dequeue(self):
        if self.current_size > 0:
            self.current_size -= 1

class ReplayManager:
    """Handles event replay from persistence."""
    def __init__(self, persistence: EventPersistence, bus: "EventBus"):
        self.persistence = persistence
        self.bus = bus

    async def replay_unprocessed(self):
        """Replay unprocessed events."""
        with sqlite3.connect(self.persistence.db_path) as conn:
            rows = conn.execute("SELECT * FROM events WHERE processed=0").fetchall()
            for row in rows:
                event = Event(id=row[0], topic=row[1], data=eval(row[2]), priority=row[3], timestamp=datetime.fromisoformat(row[4]))
                await self.bus.publish(event)
                self.persistence.mark_processed(event.id)

class EventBus:
    """Asynchronous event bus with pub/sub, priorities, dead-letter, metrics, persistence, replay, backpressure, and shutdown."""
    def __init__(self, db_path: str = "event_bus.db", max_queue_size: int = 1000):
        self.subscribers: Dict[str, List[Callable]] = defaultdict(list)
        self.priority_queues: Dict[int, asyncio.PriorityQueue] = {Priority.HIGH: asyncio.PriorityQueue(),
                                                                  Priority.NORMAL: asyncio.PriorityQueue(),
                                                                  Priority.LOW: asyncio.PriorityQueue()}
        self.dead_letter = DeadLetterQueue(db_path)
        self.metrics = MetricsCollector()
        self.persistence = EventPersistence(db_path)
        self.backpressure = BackpressureManager(max_queue_size)
        self.replay = ReplayManager(self.persistence, self)
        self.shutdown_event = asyncio.Event()
        self.tasks: List[asyncio.Task] = []
        atexit.register(self._shutdown)

    async def publish(self, event: Event, persist: bool = True) -> bool:
        """Publish an event."""
        if not self.backpressure.can_enqueue():
            logger.warning("Backpressure: Queue full, dropping event")
            return False
        self.backpressure.enqueue()
        if persist:
            self.persistence.save(event)
        await self.priority_queues[event.priority].put((-event.priority, event))  # Negative for max-heap
        self.metrics.increment("published")
        return True

    def subscribe(self, topic: str, callback: Callable):
        """Subscribe to a topic."""
        self.subscribers[topic].append(callback)

    async def _process_queue(self, priority: int):
        """Process events in a priority queue."""
        while not self.shutdown_event.is_set():
            try:
                _, event = await asyncio.wait_for(self.priority_queues[priority].get(), timeout=1.0)
                self.backpressure.dequeue()
                for callback in self.subscribers.get(event.topic, []):
                    try:
                        await asyncio.wait_for(callback(event), timeout=10.0)
                        self.persistence.mark_processed(event.id)
                        self.metrics.increment("processed")
                    except Exception as e:
                        logger.error(f"Error processing event {event.id}: {e}")
                        event.retries += 1
                        if event.retries >= 3:
                            self.dead_letter.add(event)
                            self.metrics.increment("dead_lettered")
                        else:
                            await self.publish(event, persist=False)
            except asyncio.TimeoutError:
                continue

    async def start(self):
        """Start the event bus."""
        await self.replay.replay_unprocessed()
        for priority in [Priority.HIGH, Priority.NORMAL, Priority.LOW]:
            task = asyncio.create_task(self._process_queue(priority))
            self.tasks.append(task)

    async def shutdown(self):
        """Graceful shutdown."""
        self.shutdown_event.set()
        await asyncio.gather(*self.tasks, return_exceptions=True)
        logger.info("EventBus shut down")

    def _shutdown(self):
        """Sync shutdown hook."""
        asyncio.run(self.shutdown())

"""
Memory Store module with SQLite-vec vectors, NVIDIA embeddings, memory decay, semantic search, MMR, event emission, and connection pooling.

import asyncio
import sqlite3

# Assume sqlite_vec is available
import sqlite_vec

logger = logging.getLogger(__name__)

__all__ = ["MemoryStore", "MemoryEntry"]

@dataclass
class MemoryEntry:
    """A memory entry with vector and metadata."""
    id: str
    content: str
    vector: List[float]
    timestamp: datetime
    importance: float = 1.0

class MemoryStore:
    """Memory store with vectors, embeddings, decay, search, and event emission."""
    def __init__(self, db_path: str = "memory.db", bus: EventBus = None, half_life_days: int = 30):
        self.db_path = db_path
        self.bus = bus
        self.half_life_days = half_life_days
        self.pool = sqlite3.connect(db_path, check_same_thread=False)
        self.pool.execute("PRAGMA journal_mode=WAL")
        self._init_db()

    def _init_db(self):
        sqlite_vec.load(self.pool)
        self.pool.execute("""
            CREATE VIRTUAL TABLE IF NOT EXISTS memories USING vec0(
                id TEXT PRIMARY KEY,
                content TEXT,
                vector FLOAT[768],
                timestamp TEXT,
                importance REAL
            )
        """)

    @retry(stop=stop_after_attempt(5), wait=wait_exponential(multiplier=0.5, min=0.5, max=4))
    async def _get_embeddings(self, texts: List[str]) -> List[List[float]]:
        """Get embeddings from NVIDIA with backoff."""
        # Mock NVIDIA API call
        await asyncio.sleep(0.1)  # Simulate API delay
        return [np.random.rand(768).tolist() for _ in texts]  # Mock embeddings

    async def store(self, content: str, importance: float = 1.0) -> str:
        """Store a memory entry."""
        vectors = await self._get_embeddings([content])
        entry_id = str(datetime.now().timestamp())
        entry = MemoryEntry(id=entry_id, content=content, vector=vectors[0], timestamp=datetime.now(), importance=importance)
        with self.pool:
            self.pool.execute("""
                INSERT INTO memories VALUES (?, ?, ?, ?, ?)
            """, (entry.id, entry.content, entry.vector, entry.timestamp.isoformat(), entry.importance))
        if self.bus:
            await self.bus.publish(Event(id=f"memory_{entry_id}", topic="memory_stored", data={"id": entry_id}))
        return entry_id

    async def search(self, query: str, top_k: int = 5, diversity: float = 0.5) -> List[Dict[str, Any]]:
        """Semantic search with MMR diversity."""
        query_vector = (await self._get_embeddings([query]))[0]
        # Decay importance over time
        decay_factor = np.exp(-np.log(2) * (datetime.now() - datetime.fromisoformat("2020-01-01")).days / self.half_life_days)  # Simplified
        with self.pool:
            rows = self.pool.execute("""
                SELECT id, content, vector, timestamp, importance * ? as decayed_importance
                FROM memories
                ORDER BY vec_distance_cosine(vector, ?) LIMIT ?
            """, (decay_factor, query_vector, top_k * 2)).fetchall()
        # Simple MMR: Select diverse results
        results = []
        selected = set()
        for row in rows:
            if len(results) >= top_k:
                break
            if all(np.dot(row[2], prev['vector']) < diversity for prev in results):
                results.append({"id": row[0], "content": row[1], "vector": row[2], "timestamp": row[3], "importance": row[4]})
                selected.add(row[0])
        return results

    async def prune_stale(self, days_old: int = 90):
        """Prune memories older than days_old."""
        cutoff = (datetime.now() - timedelta(days=days_old)).isoformat()
        with self.pool:
            self.pool.execute("DELETE FROM memories WHERE timestamp < ?", (cutoff,))
        if self.bus:
            await self.bus.publish(Event(id="prune_done", topic="memory_pruned", data={"pruned": True}))

"""
Orchestrator module using LangGraph StateGraph for task execution with self-improvement via event bus.
"""
import asyncio
import json
import logging
from typing import Any, Dict, List, Optional
from langgraph import StateGraph, START, END
from langgraph.graph import StateGraph
from .event_bus import EventBus, Event
from .memory_store import MemoryStore

logger = logging.getLogger(__name__)

__all__ = ["Orchestrator", "TaskState"]

@dataclass
class TaskState:
    """State for the task graph."""
    task_id: str
    description: str
    plan: Optional[str] = None
    execution: Optional[str] = None
    review: Optional[Dict[str, Any]] = None
    checkpoint: Optional[str] = None

class Orchestrator:
    """Orchestrator with LangGraph and self-improvement."""
    def __init__(self, bus: EventBus, memory: MemoryStore):
        self.bus = bus
        self.memory = memory
        self.agent_weights: Dict[str, float] = {"planner": 1.0, "executor": 1.0, "reviewer": 1.0}
        self.success_rates: Dict[str, List[float]] = defaultdict(list)
        self.graph = self._build_graph()
        self.bus.subscribe("task_complete", self._on_task_complete)

    def _build_graph(self) -> StateGraph:
        """Build the state graph."""
        graph = StateGraph(TaskState)
        graph.add_node("plan", self._plan_node)
        graph.add_node("act", self._act_node)
        graph.add_node("observe", self._observe_node)
        graph.add_node("improve", self._improve_node)
        graph.add_node("checkpoint", self._checkpoint_node)
        graph.add_edge(START, "plan")
        graph.add_edge("plan", "act")
        graph.add_edge("act", "observe")
        graph.add_edge("observe", "improve")
        graph.add_edge("improve", "checkpoint")
        graph.add_edge("checkpoint", END)
        return graph.compile()

    async def _plan_node(self, state: TaskState) -> Dict[str, Any]:
        """Plan node."""
        await self.bus.publish(Event(id=f"plan_{state.task_id}", topic="agent_request", data={"agent": "planner", "task": state.description}))
        # Wait for response (simplified)
        state.plan = "Planned steps"
        return {"plan": state.plan}

    async def _act_node(self, state: TaskState) -> Dict[str, Any]:
        """Act node."""
        await self.bus.publish(Event(id=f"act_{state.task_id}", topic="agent_request", data={"agent": "executor", "plan": state.plan}))
        state.execution = "Executed"
        return {"execution": state.execution}

    async def _observe_node(self, state: TaskState) -> Dict[str, Any]:
        """Observe node."""
        await self.bus.publish(Event(id=f"observe_{state.task_id}", topic="agent_request", data={"agent": "reviewer", "execution": state.execution}))
        state.review = {"score": 0.8}
        return {"review": state.review}

    async def _improve_node(self, state: TaskState) -> Dict[str, Any]:
        """Improve node: Adjust weights based on success."""
        for agent, scores in self.success_rates.items():
            if scores:
                avg = sum(scores) / len(scores)
                self.agent_weights[agent] = max(0.1, self.agent_weights[agent] * (1 + (avg - 0.5) * 0.1))  # Actual adjustment
        await self.memory.store(f"Task {state.task_id} improved weights: {self.agent_weights}")
        return {}

    async def _checkpoint_node(self, state: TaskState) -> Dict[str, Any]:
        """Checkpoint node."""
        state.checkpoint = json.dumps({"task_id": state.task_id, "weights": self.agent_weights})
        return {"checkpoint": state.checkpoint}

    async def _on_task_complete(self, event: Event):
        """Handle task completion."""
        data = event.data
        agent = data.get("agent")
        score = data.get("score", 0)
        self.success_rates[agent].append(score)

    async def execute_task(self, task_id: str, description: str):
        """Execute a task."""
        state = TaskState(task_id=task_id, description=description)
        await self.graph.ainvoke(state)

    def serialize_state(self) -> str:
        """Serialize current state."""
        return json.dumps({"weights": self.agent_weights, "success_rates": dict(self.success_rates)})

"""
Agent Base module with lifecycle, specific agents, and metrics.
"""
import asyncio
import logging
from typing import Any, Dict, List, Optional
from abc import ABC, abstractmethod
from .event_bus import EventBus, Event

logger = logging.getLogger(__name__)

__all__ = ["AgentBase", "PlannerAgent", "ExecutorAgent", "ReviewerAgent", "AgentLifecycle"]

class AgentLifecycle:
    INIT = "init"
    READY = "ready"
    BUSY = "busy"
    DONE = "done"

class AgentBase(ABC):
    """Base agent class."""
    def __init__(self, bus: EventBus, name: str):
        self.bus = bus
        self.name = name
        self.lifecycle = AgentLifecycle.INIT
        self.metrics: Dict[str, int] = {"tasks": 0, "successes": 0}
        self.bus.subscribe("agent_request", self._handle_request)

    async def _handle_request(self, event: Event):
        """Handle incoming requests."""
        if event.data.get("agent") == self.name:
            self.lifecycle = AgentLifecycle.BUSY
            result = await self.process(event.data)
            self.lifecycle = AgentLifecycle.DONE
            self.metrics["tasks"] += 1
            if result.get("success"):
                self.metrics["successes"] += 1
            await self.bus.publish(Event(id=f"response_{event.id}", topic="agent_response", data={**result, "agent": self.name}))

    @abstractmethod
    async def process(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """Process the task."""
        pass

class PlannerAgent(AgentBase):
    """Agent for planning tasks."""
    async def process(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """Decompose task."""
        task = data["task"]
        plan = f"Plan for: {task}"
        return {"plan": plan, "success": True}

class ExecutorAgent(AgentBase):
    """Agent for executing tasks."""
    async def process(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """Execute based on plan."""
        plan = data["plan"]
        execution = f"Executed: {plan}"
        return {"execution": execution, "success": True}

class ReviewerAgent(AgentBase):
    """Agent for reviewing outputs."""
    async def process(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """Validate and score."""
        execution = data["execution"]
        score = 0.9 if "Executed" in execution else 0.5
        return {"score": score, "success": True}

"""
Self-Improver module for analyzing and optimizing the system.
"""
import asyncio
import logging
from typing import Dict, List
from .event_bus import EventBus
from .memory_store import MemoryStore
from .orchestrator import Orchestrator

logger = logging.getLogger(__name__)

__all__ = ["SelfImprover"]

class SelfImprover:
    """Dedicated self-improvement engine."""
    def __init__(self, bus: EventBus, memory: MemoryStore, orchestrator: Orchestrator, schedule_seconds: int = 60):
        self.bus = bus
        self.memory = memory
        self.orchestrator = orchestrator
        self.schedule_seconds = schedule_seconds
        self.running = True

    async def analyze(self) -> Dict[str, Any]:
        """Analyze system metrics."""
        success_rates = {agent: (sum(scores)/len(scores) if scores else 0) for agent, scores in self.orchestrator.success_rates.items()}
        hit_rate = len(await self.memory.search("test")) / 10  # Mock
        throughput = self.bus.metrics.get("published")
        errors = self.bus.metrics.get("dead_lettered")
        return {"success_rates": success_rates, "hit_rate": hit_rate, "throughput": throughput, "errors": errors}

    async def act(self, analysis: Dict[str, Any]):
        """Perform improvements."""
        # Adjust weights
        for agent, rate in analysis["success_rates"].items():
            self.orchestrator.agent_weights[agent] *= (1 + (rate - 0.5) * 0.05)
        # Prune memories
        await self.memory.prune_stale(30)
        # Optimize batch sizes (mock)
        logger.info(f"Optimized: weights {self.orchestrator.agent_weights}")
        await self.bus.publish(Event(id="improvement", topic="system_improved", data=analysis))

    async def run(self):
        """Run on schedule."""
        while self.running:
            await asyncio.sleep(self.schedule_seconds)
            analysis = await self.analyze()
            await self.act(analysis)

    def stop(self):
        self.running = False

"""
Main demo script integrating all components.
"""
import asyncio
import logging
from .event_bus import EventBus
from .memory_store import MemoryStore
from .orchestrator import Orchestrator
from .agent_base import PlannerAgent, ExecutorAgent, ReviewerAgent
from .self_improver import SelfImprover

logging.basicConfig(level=logging.INFO)

async def main():
    bus = EventBus()
    memory = MemoryStore(bus=bus)
    orchestrator = Orchestrator(bus, memory)
    agents = [PlannerAgent(bus, "planner"), ExecutorAgent(bus, "executor"), ReviewerAgent(bus, "reviewer")]
    improver = SelfImprover(bus, memory, orchestrator, 10)  # Short schedule for demo

    await bus.start()
    asyncio.create_task(improver.run())

    tasks = [f"Task {i}: Do something {i}" for i in range(1, 6)]
    performances = []

    for i, task in enumerate(tasks, 1):
        print(f"\nRunning {task}")
        start = asyncio.get_event_loop().time()
        await orchestrator.execute_task(f"task_{i}", task)
        end = asyncio.get_event_loop().time()
        perf = end - start
        performances.append(perf)
        print(f"Task {i} completed in {perf:.2f}s. Weights: {orchestrator.agent_weights}")

    improver.stop()
    await bus.shutdown()

    print("\nFinal Report:")
    print(f"Task 1: {performances[0]:.2f}s")
    print(f"Task 5: {performances[4]:.2f}s")
    print(f"Improvement: {((performances[0] - performances[4]) / performances[0] * 100):.1f}%")
    print(f"Final weights: {orchestrator.agent_weights}")

if __name__ == "__main__":
    asyncio.run(main())
