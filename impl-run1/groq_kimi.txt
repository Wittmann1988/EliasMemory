"""
Async event bus with typed events, pub/sub, priority queue, correlation IDs.
"""
from __future__ import annotations

import asyncio
import json
import logging
import os
import time
import uuid
from asyncio import PriorityQueue
from dataclasses import dataclass, field
from enum import IntEnum
from typing import Any, Callable, Dict, List, Optional, Set

import aiohttp

# Configure logging
logging.basicConfig(level=os.getenv("LOG_LEVEL", "INFO"))
logger = logging.getLogger(__name__)


class Priority(IntEnum):
    CRITICAL = 1
    HIGH = 2
    NORMAL = 3
    LOW = 4


@dataclass
class Event:
    """Typed event with correlation tracking."""
    id: str = field(default_factory=lambda: str(uuid.uuid4()))
    type: str
    payload: Dict[str, Any]
    priority: Priority = Priority.NORMAL
    timestamp: float = field(default_factory=time.time)
    correlation_id: Optional[str] = None
    reply_to: Optional[str] = None


class EventBus:
    """
    Async event bus with priority queue and pub/sub.
    Uses Redis if available, else in-memory.
    """
    def __init__(
        self,
        redis_url: Optional[str] = None,
        max_queue_size: int = 10_000
    ) -> None:
        self._redis_url = redis_url or os.getenv("REDIS_URL")
        self._handlers: Dict[str, Set[Callable[[Event], Any]]] = {}
        self._queue: PriorityQueue[tuple[int, float, Event]] = PriorityQueue(
            maxsize=max_queue_size
        )
        self._running = False
        self._redis_sub: Optional[Any] = None
        self._session: Optional[aiohttp.ClientSession] = None

    async def start(self) -> None:
        """Start the event bus worker."""
        if self._running:
            return

        self._session = aiohttp.ClientSession()
        self._running = True
        asyncio.create_task(self._worker())

        if self._redis_url:
            try:
                await self._start_redis()
            except Exception as e:
                logger.warning("Redis unavailable, using in-memory: %s", e)

        logger.info("EventBus started")

    async def stop(self) -> None:
        """Stop the event bus worker."""
        self._running = False
        if self._session:
            await self._session.close()

    async def publish(self, event: Event) -> None:
        """Publish an event to the bus."""
        await self._queue.put((event.priority.value, event.timestamp, event))

    async def subscribe(
        self,
        event_type: str,
        handler: Callable[[Event], Any]
    ) -> None:
        """Subscribe to an event type."""
        if event_type not in self._handlers:
            self._handlers[event_type] = set()
        self._handlers[event_type].add(handler)

    async def _worker(self) -> None:
        """Main worker loop."""
        while self._running:
            try:
                _, _, event = await self._queue.get()
                await self._dispatch(event)
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.exception("Worker error: %s", e)

    async def _dispatch(self, event: Event) -> None:
        """Dispatch event to handlers."""
        handlers = self._handlers.get(event.type, set())
        for handler in handlers:
            try:
                await handler(event)
            except Exception as e:
                logger.exception("Handler error for %s: %s", event.type, e)

    async def _start_redis(self) -> None:
        """Start Redis subscription if available."""
        # Placeholder for Redis integration
        pass


async def demo() -> None:
    """Demo the event bus."""
    bus = EventBus()
    await bus.start()

    async def handle_test(event: Event) -> None:
        logger.info("Received: %s", event)

    await bus.subscribe("test", handle_test)

    event = Event(type="test", payload={"msg": "hello"})
    await bus.publish(event)

    await asyncio.sleep(0.5)
    await bus.stop()


if __name__ == "__main__":
    asyncio.run(demo())
