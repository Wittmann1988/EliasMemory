class Event(BaseModel):

    priority: Priority = Priority.NORMAL
    timestamp: float = Field(default_factory=time.time)

        """Comparison for priority queue ordering."""
        return self.priority.value < other.priority.value


@dataclass(order=True)
class PrioritizedEvent:
    """Wrapper for priority queue sorting."""
    priority: int
    sequence: int = field(compare=True)
    event: Event = field(compare=False)


class EventBus:
    """
    Async event bus implementing pub/sub pattern with priority queuing.
    Designed to be extended with Redis backend for distributed systems.
    """
    
    def __init__(self, name: str = "default_bus"):
        self.name = name
        self._subscribers: Dict[EventType, List[Callable[[Event], Awaitable[None]]]] = {}
        self._priority_queue: List[PrioritizedEvent] = []
        self._sequence_counter = 0
        self._queue_lock = asyncio.Lock()
        self._running = False
        self._worker_task: Optional[asyncio.Task] = None
        self._event_processed = asyncio.Event()
        
    async def publish(self, event: Event) -> str:
        """
        Publish an event to the bus.
        
        Args:
            event: The event to publish
            
        Returns:
            The event ID for tracking
        """
        async with self._queue_lock:
            prioritized = PrioritizedEvent(
                priority=event.priority.value,
                sequence=self._sequence_counter,
                event=event
            )
            heapq.heappush(self._priority_queue, prioritized)
            self._sequence_counter += 1
            
        logger.debug(f"Published event {event.event_id}: {event.event_type.value}")
        return event.event_id
    
    def subscribe(
        self, 
        event_type: EventType, 
        handler: Callable[[Event], Awaitable[None]]
    ) -> None:
        """
        Subscribe a handler to a specific event type.
        
        Args:
            event_type: The type of event to listen for
            handler: Async function to handle the event
        """
        if event_type not in self._subscribers:
            self._subscribers[event_type] = []
        self._subscribers[event_type].append(handler)
        logger.info(f"Subscribed handler to {event_type.value}")
    
    async def _process_event(self, event: Event) -> None:
        """Process a single event by calling all subscribers."""
        handlers = self._subscribers.get(event.event_type, [])
        
        if not handlers:
            logger.debug(f"No handlers for event type {event.event_type.value}")
            return
            
        # Execute all handlers concurrently
        tasks = []
        for handler in handlers:
            try:
                tasks.append(handler(event))
            except Exception as e:
                logger.error(f"Error creating handler task: {e}")
                
        if tasks:
            results = await asyncio.gather(*tasks, return_exceptions=True)
            for i, result in enumerate(results):
                if isinstance(result, Exception):
                    logger.error(f"Handler {i} failed for {event.event_id}: {result}")
    
    async def _worker(self) -> None:
        """Background worker that processes events from the priority queue."""
        while self._running:
            try:
                event = None
                
                async with self._queue_lock:
                    if self._priority_queue:
                        prioritized = heapq.heappop(self._priority_queue)
                        event = prioritized.event
                
                if event:
                    await self._process_event(event)
                else:
                    # Small sleep to prevent busy waiting
                    await asyncio.sleep(0.01)
                    
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Worker error: {e}")
                await asyncio.sleep(0.1)
    
    async def start(self) -> None:
        """Start the event bus worker."""
        if not self._running:
            self._running = True
            self._worker_task = asyncio.create_task(self._worker())
            logger.info(f"Event bus '{self.name}' started")
    
    async def stop(self) -> None:
        """Stop the event bus worker."""
        self._running = False
        if self._worker_task:
            self._worker_task.cancel()
            try:
                await self._worker_task
            except asyncio.CancelledError:
                pass
        logger.info(f"Event bus '{self.name}' stopped")
    
    async def wait_for_event(self, event_id: str, timeout: float = 30.0) -> bool:
        """Wait for a specific event to be processed."""
        start_time = time.time()
        while time.time() - start_time < timeout:
            # Check if event is still in queue
            async with self._queue_lock:
                for pe in self._priority_queue:
                    if pe.event.event_id == event_id:
                        break
                else:
                    # Event not in queue, assume processed
                    return True
            await asyncio.sleep(0.1)
        return False

    def __init__(self, bus: EventBus, agent_id: str):

            return func

# Constants
NVIDIA_EMBED_API_URL = "https://integrate.api.nvidia.com/v1/embeddings"
NVIDIA_EMBED_MODEL = "nvidia/llama-nemotron-embed-1b-v2"
EMBEDDING_DIM = 2048


class EmbeddingError(Exception):
    """Custom exception for embedding API errors."""
    pass


class MemoryType(str):
    """Types of memory stored."""
    EPISODIC = "episodic"
    SEMANTIC = "semantic"
    PROCEDURAL = "procedural"
    WORKING = "working"


@dataclass
class MemoryEntry:
    """A single memory entry."""
    memory_id: str
    content: str
    embedding: Optional[List[float]]
    memory_type: str
    metadata: Dict[str, Any]
    timestamp: float
    importance: float = 1.0

class NVIDIAEmbeddingClient:
    """
    Async client for NVIDIA embedding API.
    Uses the nvidia/llama-nemotron-embed-1b-v2 model (2048 dimensions).
    """
    
    def __init__(self, api_key: Optional[str] = None, timeout: float = 30.0):
        self.api_key = api_key or os.getenv("NVIDIA_EMBED_API_KEY")
        if not self.api_key:
            logger.warning("NVIDIA_EMBED_API_KEY not set. Embeddings will use fallback.")
        self.timeout = timeout
        self._session: Optional[aiohttp.ClientSession] = None
        
    async def __aenter__(self):
        self._session = aiohttp.ClientSession(
            timeout=aiohttp.ClientTimeout(total=self.timeout)
        )
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self._session:
            await self._session.close()
    
    async def embed(
        self, 
        texts: List[str], 
        input_type: str = "passage",
        retry_count: int = 3
    ) -> List[List[float]]:
        """
        Generate embeddings for a list of texts.
        
        Args:
            texts: List of strings to embed
            input_type: "passage" for storage, "query" for search
            retry_count: Number of retries on network failure
            
        Returns:
            List of embedding vectors (2048-dim each)
        """
        if not self.api_key:
            return self._fallback_embed(texts)
            
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        
        payload = {
            "model": NVIDIA_EMBED_MODEL,
            "input": texts,
            "input_type": input_type,
            "encoding_format": "float"
        }
        
        for attempt in range(retry_count):
            try:
                if not self._session:
                    self._session = aiohttp.ClientSession(
                        timeout=aiohttp.ClientTimeout(total=self.timeout)
                    )
                    
                async with self._session.post(
                    NVIDIA_EMBED_API_URL,
                    headers=headers,
                    json=payload
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        embeddings = [item["embedding"] for item in data["data"]]
                        logger.info(f"Generated {len(embeddings)} embeddings via NVIDIA API")
                        return embeddings
                    elif response.status == 429:
                        # Rate limit - exponential backoff
                        wait_time = 2 ** attempt
                        logger.warning(f"Rate limited, waiting {wait_time}s")
                        await asyncio.sleep(wait_time)
                    else:
                        error_text = await response.text()
                        raise EmbeddingError(f"API error {response.status}: {error_text}")
                        
            except aiohttp.ClientError as e:
                logger.error(f"Network error (attempt {attempt + 1}): {e}")
                if attempt < retry_count - 1:
                    await asyncio.sleep(1)
                else:
                    logger.warning("Falling back to random embeddings")
                    return self._fallback_embed(texts)
        
        return self._fallback_embed(texts)
    
    def _fallback_embed(self, texts: List[str]) -> List[List[float]]:
        """
        Fallback embedding for offline/error scenarios.
        Uses deterministic hash-based vectors for reproducibility.
        """
        logger.warning(f"Using fallback embeddings for {len(texts)} texts")
        embeddings = []
        for text in texts:
            # Deterministic pseudo-embedding based on content hash
            np.random.seed(hash(text) % (2**32))
            vec = np.random.randn(EMBEDDING_DIM).astype(np.float32)
            vec = vec / np.linalg.norm(vec)  # Normalize
            embeddings.append(vec.tolist())
        return embeddings


class VectorStore:

            )
        """)

                    embedding FLOAT[{EMBEDDING_DIM}]
                )
            """)
        except sqlite3.OperationalError as e:

                metadata TEXT DEFAULT '{}',

        results = []

            if memory_type:

                    SELECT id, content, embedding, metadata FROM memories
                    WHERE memory_type = ?
                """, (memory_type,))
            else:
                cursor = self._conn.execute("""
                    SELECT id, content, embedding, metadata FROM memories
                """)
                
            query_vec = np.array(query_embedding, dtype=np.float32)
            query_norm = np.linalg.norm(query_vec)
            
            for row in cursor.fetchall():
                memory_id, content, emb_blob, metadata_json = row
                
                if emb_blob is None:
                    continue
                    
                try:
                    stored_vec = np.frombuffer(emb_blob, dtype=np.float32)
                    
                    # Cosine similarity
                    dot_product = np.dot(query_vec, stored_vec)
                    norm_product = query_norm * np.linalg.norm(stored_vec)
                    
                    if norm_product > 0:
                        similarity = dot_product / norm_product
                    else:
                        similarity = 0.0
                        
                    if similarity >= threshold:
                        metadata = json.loads(metadata_json) if metadata_json else {}
                        results.append((memory_id, content, float(similarity), metadata))
                        
                except Exception as e:
                    logger.debug(f"Error processing memory {memory_id}: {e}")
                    continue
                    
            # Sort by similarity descending
            results.sort(key=lambda x: x[2], reverse=True)
            
        except Exception as e:
            logger.error(f"Search error: {e}")
            
        return results[:k]
    
    def get(self, memory_id: str) -> Optional[MemoryEntry]:
        """Retrieve a specific memory by ID."""
        cursor = self._conn.execute("""
            SELECT id, content, embedding, memory_type, metadata, timestamp, importance, access_count
            FROM memories WHERE id = ?
        """, (memory_id,))
        
        row = cursor.fetchone()
        if row:
            embedding = list(np.frombuffer(row[2], dtype=np.float32)) if row[2] else None
            return MemoryEntry(
                memory_id=row[0],
                content=row[1],
                embedding=embedding,
                memory_type=row[3],
                metadata=json.loads(row[4]) if row[4] else {},
                timestamp=row[5],
                importance=row[6],
                access_count=row[7]
            )
        return None
    
    def update_access(self, memory_id: str) -> None:
        """Increment access count for a memory."""
        self._conn.execute("""
            UPDATE memories SET access_count = access_count + 1 WHERE id = ?
        """, (memory_id,))
        self._conn.commit()
        
    def close(self) -> None:
        """Close the database connection."""
        if self._conn:
            self._conn.close()


class MemoryManager:
    """
    Unified memory manager combining:
    - Mem0 for graph-based short-term memory
    - SQLite-vec for long-term vector persistence
    - NVIDIA embeddings for semantic representation
    """
    
    def __init__(
        self, 
        db_path: str = "memory_vectors.db",
        api_key: Optional[str] = None
    ):
        self.vector_store = VectorStore(db_path)
        self.embed_client = NVIDIAEmbeddingClient(api_key=api_key)
        self._initialized = False
        
        # In-memory working memory (Mem0 style)
        self._working_memory: Dict[str, MemoryEntry] = {}
        self._memory_graph: Dict[str, List[str]] = {}  # Memory associations
        
    async def initialize(self) -> None:
        """Initialize all memory systems."""
        self.vector_store.initialize()
        await self.embed_client.__aenter__()
        self._initialized = True
        logger.info("MemoryManager initialized")
        
    async def shutdown(self) -> None:
        """Cleanup resources."""
        await self.embed_client.__aexit__(None, None, None)
        self.vector_store.close()
        self._initialized = False
        
    async def add_memory(
        self,
        content: str,
        memory_type: str = MemoryType.EPISODIC,
        metadata: Optional[Dict[str, Any]] = None,
        importance: float = 1.0
    ) -> str:
        """
        Add a new memory to both working and long-term storage.
        
        Args:
            content: The memory content to store
            memory_type: Type classification
            metadata: Additional metadata
            importance: Importance score (0.0 to 1.0)
            
        Returns:
            The memory ID
        """
        memory_id = str(uuid.uuid4())
        
        # Generate embedding
        embeddings = await self.embed_client.embed([content], input_type="passage")
        embedding = embeddings[0]
        
        # Create memory entry
        entry = MemoryEntry(
            memory_id=memory_id,
            content=content,
            embedding=embedding,
            memory_type=memory_type,
            metadata=metadata or {},
            timestamp=time.time(),
            importance=importance
        )
        
        # Store in working memory
        self._working_memory[memory_id] = entry
        
        # Store in vector store
        self.vector_store.store(
            memory_id=memory_id,
            content=content,
            embedding=embedding,
            memory_type=memory_type,
            metadata=metadata,
            importance=importance
        )
        
        logger.info(f"Added memory {memory_id}: {content[:50]}...")
        return memory_id
    
    async def search(
        self,
        query: str,
        k: int = 5,
        memory_type: Optional[str] = None,
        threshold: float = 0.5
    ) -> List[Tuple[str, str, float, Dict[str, Any]]]:
        """
        Search memories by semantic similarity.
        
        Args:
            query: Search query text
            k: Number of results
            memory_type: Filter by type
            threshold: Minimum similarity score
            
        Returns:
            List of (memory_id, content, score, metadata)
        """
        # Generate query embedding
        embeddings = await self.embed_client.embed([query], input_type="query")
        query_embedding = embeddings[0]
        
        # Search vector store
        results = self.vector_store.search(
            query_embedding=query_embedding,
            k=k,
            memory_type=memory_type,
            threshold=threshold
        )
        
        # Update access counts
        for memory_id, _, _, _ in results:
            self.vector_store.update_access(memory_id)
            
        return results
    
    def get_context(
        self, 
        query: str, 
        max_tokens: int = 2000
    ) -> List[Dict[str, Any]]:
        """
        Get relevant context for a query (working memory style).
        Returns memories sorted by recency and importance.
        """
        # Get from working memory
        working_entries = sorted(
            self._working_memory.values(),
            key=lambda x: (x.importance, x.timestamp),
            reverse=True
        )
        
        context = []
        token_count = 0
        
        for entry in working_entries:
            entry_tokens = len(entry.content.split()) * 1.3  # Rough estimate
            
            if token_count + entry_tokens > max_tokens:
                break
                
            context.append({
                "memory_id": entry.memory_id,
                "content": entry.content,
                "memory_type": entry.memory_type,
                "importance": entry.importance
            })
            token_count += entry_tokens
            
        return context
    
    def associate_memories(
        self, 
        memory_id_1: str, 
        memory_id_2: str
    ) -> None:
        """Create an association between two memories (graph memory)."""
        if memory_id_1 not in self._memory_graph:
            self._memory_graph[memory_id_1] = []
        if memory_id_2 not in self._memory_graph:
            self._memory_graph[memory_id_2] = []
            
        if memory_id_2 not in self._memory_graph[memory_id_1]:
            self._memory_graph[memory_id_1].append(memory_id_2)
        if memory_id_1 not in self._memory_graph[memory_id_2]:
            self._memory_graph[memory_id_2].append(memory_id_1)
            
    def get_associations(self, memory_id: str) -> List[str]:
        """Get all memories associated with a given memory."""
        return self._memory_graph.get(memory_id, [])
    
    def get_stats(self) -> Dict[str, Any]:
        """Get memory system statistics."""
        return {
            "working_memory_count": len(self._working_memory),
            "graph_nodes": len(self._memory_graph),
            "graph_edges": sum(len(v) for v in self._memory_graph.values()) // 2
        }
