# FILE: event_bus.py
import asyncio
import heapq
import uuid
from typing import Any, Dict, List, Optional, Protocol, TypeVar
from pydantic import BaseModel
import redis.asyncio as redis
from aio_pubsub import Hub, Subscriber, Publisher
import os

T = TypeVar('T')

class Event(BaseModel):
    id: str = ""
    correlation_id: str = ""
    type: str
    priority: int = 1
    data: Dict[str, Any] = {}

    def __init__(self, **data):
        super().__init__(**data)
        if not self.id:
            self.id = str(uuid.uuid4())
        if not self.correlation_id:
            self.correlation_id = self.id

class EventHandler(Protocol):
    async def __call__(self, event: Event) -> None:
        ...

class EventBus:
    def __init__(self):
        self.redis = redis.Redis(host='localhost', port=6379, decode_responses=True)
        self.hub = Hub()
        self.publisher = Publisher(self.hub)
        self.subscriber = Subscriber(self.hub, 'events')
        self.priority_queue: List[tuple[int, Event]] = []
        self.handlers: Dict[str, List[EventHandler]] = {}

    async def publish(self, event: Event) -> None:
        heapq.heappush(self.priority_queue, (-event.priority, event))
        await self.publisher.publish('events', event.model_dump_json())

    async def subscribe(self, event_type: str, handler: EventHandler) -> None:
        if event_type not in self.handlers:
            self.handlers[event_type] = []
        self.handlers[event_type].append(handler)

    async def process_queue(self) -> None:
        while True:
            if self.priority_queue:
                _, event = heapq.heappop(self.priority_queue)
                if event.type in self.handlers:
                    for handler in self.handlers[event.type]:
                        await handler(event)
            await asyncio.sleep(0.01)

    async def start(self) -> None:
        asyncio.create_task(self.process_queue())
        async for message in self.subscriber:
            event_data = message.value
            event = Event.parse_raw(event_data)
            if event.type in self.handlers:
                for handler in self.handlers[event.type]:
                    await handler(event)

async def demo_handler(event: Event) -> None:
    print(f"Handled event: {event.type} with data: {event.data}")

async def main():
    bus = EventBus()
    await bus.subscribe("test_event", demo_handler)
    asyncio.create_task(bus.start())
    event = Event(type="test_event", priority=2, data={"message": "Hello World"})
    await bus.publish(event)
    await asyncio.sleep(1)

if __name__ == "__main__":
    asyncio.run(main())

import asyncio
import sqlite3
import numpy as np
from typing import List, Dict, Any, Optional
import mem0
from sqlite_vec import VecSQLite
import aiohttp
import os
from pydantic import BaseModel

class MemoryEntry(BaseModel):
    id: str
    content: str
    embedding: List[float]

class MemoryStore:
    def __init__(self):
        self.mem0 = mem0.Memory()
        self.db = VecSQLite(":memory:")
        self.db.execute("CREATE VIRTUAL TABLE memories USING vec0(embedding float[2048])")
        self.api_key = os.getenv("NVIDIA_EMBED_API_KEY")
        if not self.api_key:
            raise ValueError("NVIDIA_EMBED_API_KEY not set")

    async def get_embedding(self, text: str) -> List[float]:
        url = "https://integrate.api.nvidia.com/v1/embeddings"
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        data = {
            "model": "nvidia/llama-nemotron-embed-1b-v2",
            "input": [text]
        }
        try:
            async with aiohttp.ClientSession() as session:
                async with session.post(url, headers=headers, json=data) as response:
                    if response.status == 200:
                        result = await response.json()
                        return result["data"][0]["embedding"]
                    else:
                        raise Exception(f"API error: {response.status}")
        except Exception as e:
            print(f"Error getting embedding: {e}")
            return [0.0] * 2048

    async def add_memory(self, content: str, user_id: str = "default") -> None:
        embedding = await self.get_embedding(content)
        self.mem0.add(content, user_id=user_id)
        self.db.execute("INSERT INTO memories(rowid, embedding) VALUES (?, ?)", (None, np.array(embedding)))

    async def search_memory(self, query: str, user_id: str = "default") -> List[str]:
        embedding = await self.get_embedding(query)
        results = self.db.execute(
            "SELECT rowid, distance FROM memories WHERE embedding MATCH ? ORDER BY distance LIMIT 5",
            (np.array(embedding),)
        ).fetchall()
        memories = []
        for rowid, _ in results:
            # Retrieve content from mem0 or store it properly; for simplicity, assume mem0 holds it
            # In real impl, need to map rowid to content
            memories.append(f"Memory {rowid}")
        return memories

async def main():
    store = MemoryStore()
    await store.add_memory("This is a test memory.")
    results = await store.search_memory("test")
    print(f"Search results: {results}")

if __name__ == "__main__":
    asyncio.run(main())
# FILE: orchestrator.py
import asyncio
from typing import TypedDict, Annotated
from langgraph import StateGraph, START, END
from langgraph.checkpoint.memory import MemorySaver
from langgraph.types import Command
from agent_base import PlannerAgent, ExecutorAgent

class State(TypedDict):
    task: str
    plan: str
    action: str
    observation: str
    improvement: str

async def plan_node(state: State) -> Command:
    planner = PlannerAgent()
    plan = await planner.plan(state["task"])
    return Command(update={"plan": plan}, goto="act")

async def act_node(state: State) -> Command:
    executor = ExecutorAgent()
    action = await executor.execute(state["plan"])
    return Command(update={"action": action}, goto="observe")

async def observe_node(state: State) -> Command:
    observation = f"Observed action: {state['action']}"
    return Command(update={"observation": observation}, goto="improve")

async def improve_node(state: State) -> Command:
    improvement = f"Improved based on: {state['observation']}"
    return Command(update={"improvement": improvement}, goto=END)

def build_graph():
    graph = StateGraph(State)
    graph.add_node("plan", plan_node)
    graph.add_node("act", act_node)
    graph.add_node("observe", observe_node)
    graph.add_node("improve", improve_node)
    graph.add_edge(START, "plan")
    return graph.compile(checkpointer=MemorySaver())

async def main():
    graph = build_graph()
    result = await graph.ainvoke({"task": "Test task"}, {"configurable": {"thread_id": "1"}})
    print(f"Final state: {result}")

if __name__ == "__main__":
    asyncio.run(main())
# FILE: agent_base.py
import asyncio
from abc import ABC, abstractmethod
from typing import Dict, Any
from crewai import Agent, Task, Crew  # Assuming CrewAI is installed and imported

class BaseAgent(ABC):
    def __init__(self, name: str, role: str):
        self.agent = Agent(
            role=role,
            goal="Perform tasks effectively",
            backstory="An AI agent in the team.",
            verbose=True
        )

    @abstractmethod
    async def run(self, input_data: Dict[str, Any]) -> str:
        pass

class PlannerAgent(BaseAgent):
    def __init__(self):
        super().__init__("Planner", "Planner")

    async def plan(self, task: str) -> str:
        # Simulate planning
        return f"Plan for: {task}"

    async def run(self, input_data: Dict[str, Any]) -> str:
        return await self.plan(input_data["task"])

class ExecutorAgent(BaseAgent):
    def __init__(self):
        super().__init__("Executor", "Executor")

    async def execute(self, plan: str) -> str:
        # Simulate execution
        return f"Executed: {plan}"

    async def run(self, input_data: Dict[str, Any]) -> str:
        return await self.execute(input_data["plan"])

async def main():
    planner = PlannerAgent()
    executor = ExecutorAgent()
    plan = await planner.plan("Build a house")
    action = await executor.execute(plan)
    print(f"Plan: {plan}, Action: {action}")

if __name__ == "__main__":
    asyncio.run(main())
