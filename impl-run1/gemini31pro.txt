# FILE: event_bus.py
import asyncio
import time
import uuid
from typing import Any, Callable, Dict, List, Optional
from pydantic import BaseModel, Field

class Event(BaseModel):
    """Typed event model with correlation ID and priority."""
    id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    correlation_id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    event_type: str
    payload: Dict[str, Any]
    priority: int = 10  # Lower number = higher priority
    timestamp: float = Field(default_factory=time.time)

    def __lt__(self, other: "Event") -> bool:
        if self.priority == other.priority:
            return self.timestamp < other.timestamp
        return self.priority < other.priority

class EventBus:
    """Async event bus with pub/sub and priority queue."""
    def __init__(self):
        self._subscribers: Dict[str, List[Callable[[Event], Any]]] = {}
        self._queue: asyncio.PriorityQueue = asyncio.PriorityQueue()
        self._running: bool = False
        self._worker_task: Optional[asyncio.Task] = None

    def subscribe(self, event_type: str, callback: Callable[[Event], Any]) -> None:
        """Subscribe a callback to a specific event type."""
        if event_type not in self._subscribers:
            self._subscribers[event_type] = []
        self._subscribers[event_type].append(callback)

    async def publish(self, event: Event) -> None:
        """Publish an event to the priority queue."""
        await self._queue.put((event.priority, event.timestamp, event))

    async def start(self) -> None:
        """Start the event bus worker loop."""
        self._running = True
        self._worker_task = asyncio.create_task(self._process_events())
        print("[EventBus] Started.")

    async def stop(self) -> None:
        """Stop the event bus gracefully."""
        self._running = False
        if self._worker_task:
            self._worker_task.cancel()
            try:
                await self._worker_task
            except asyncio.CancelledError:
                pass
        print("[EventBus] Stopped.")

    async def _process_events(self) -> None:
        """Background task to route events to subscribers."""
        while self._running:
            try:
                _, _, event = await self._queue.get()
                callbacks = self._subscribers.get(event.event_type, [])
                
                # Execute all callbacks concurrently
                tasks = []
                for cb in callbacks:
                    if asyncio.iscoroutinefunction(cb):
                        tasks.append(asyncio.create_task(cb(event)))
                    else:
                        cb(event)
                
                if tasks:
                    await asyncio.gather(*tasks, return_exceptions=True)
                    
                self._queue.task_done()
            except asyncio.CancelledError:
                break
            except Exception as e:
                print(f"[EventBus] Error processing event: {e}")

# ==========================================
# Demo Execution

if __name__ == "__main__":
    async def demo():
        bus = EventBus()
        await bus.start()

        async def memory_handler(event: Event):
            print(f"-> [Memory Handler] Received {event.event_type} | Data: {event.payload} | Priority: {event.priority}")
            await asyncio.sleep(0.1)

        async def agent_handler(event: Event):
            print(f"-> [Agent Handler] Received {event.event_type} | Data: {event.payload} | Priority: {event.priority}")

        bus.subscribe("memory_saved", memory_handler)
        bus.subscribe("task_created", agent_handler)

        # Publish out of order to test priority (lower number = higher priority)
        await bus.publish(Event(event_type="memory_saved", payload={"item": "context_A"}, priority=10))
        await bus.publish(Event(event_type="task_created", payload={"task": "urgent_fix"}, priority=1))
        await bus.publish(Event(event_type="memory_saved", payload={"item": "context_B"}, priority=5))

        # Wait for processing
        await asyncio.sleep(0.5)
        await bus.stop()

    asyncio.run(demo())

# FILE: memory_store.py
import os
import sqlite3
import asyncio
import aiohttp
import sqlite_vec
from typing import List, Dict, Any, Optional
from mem0 import Memory

class NvidiaEmbedder:
    """NVIDIA Embeddings API Client (2048-dim)."""
    def __init__(self, api_key: Optional[str] = None):
        self.api_key = api_key or os.getenv("NVIDIA_EMBED_API_KEY")
        self.endpoint = "https://integrate.api.nvidia.com/v1/embeddings"
        self.model = "nvidia/llama-nemotron-embed-1b-v2"

    async def get_embedding(self, text: str) -> List[float]:
        if not self.api_key:
            print("[NvidiaEmbedder] Warning: No API key found. Returning dummy 2048-dim vector.")
            return [0.01] * 2048

        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        payload = {
            "input": [text],
            "model": self.model,
            "input_type": "query"
        }
        
        async with aiohttp.ClientSession() as session:
            try:
                async with session.post(self.endpoint, headers=headers, json=payload) as resp:
                    resp.raise_for_status()
                    data = await resp.json()
                    return data["data"][0]["embedding"]
            except Exception as e:
                print(f"[NvidiaEmbedder] Network error: {e}")
                return [0.0] * 2048

class SqliteVecStore:
    """Long-term persistence using sqlite-vec."""
    def __init__(self, db_path: str = "long_term_memory.db"):
        self.db_path = db_path
        self._init_db()

    def _init_db(self):
        self.conn = sqlite3.connect(self.db_path, check_same_thread=False)
        self.conn.enable_load_extension(True)
        sqlite_vec.load(self.conn)
        
        cursor = self.conn.cursor()
        # Create virtual table for 2048-dim vectors
        cursor.execute("""
            CREATE VIRTUAL TABLE IF NOT EXISTS vec_memory USING vec0(
                embedding float[2048]
            );
        """)
        # Metadata table
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS memory_meta (
                rowid INTEGER PRIMARY KEY,
                content TEXT,
                metadata TEXT
            );
        """)
        self.conn.commit()

    async def add_memory(self, embedding: List[float], content: str, metadata: str = "{}") -> int:
        def _insert():
            cursor = self.conn.cursor()
            cursor.execute("INSERT INTO vec_memory(embedding) VALUES (?)", (sqlite_vec.serialize_float32(embedding),))
            rowid = cursor.lastrowid
            cursor.execute("INSERT INTO memory_meta(rowid, content, metadata) VALUES (?, ?, ?)", (rowid, content, metadata))
            self.conn.commit()
            return rowid
        return await asyncio.to_thread(_insert)

    async def search(self, query_embedding: List[float], limit: int = 5) -> List[Dict[str, Any]]:
        def _query():
            cursor = self.conn.cursor()
            query = """
                SELECT m.rowid, m.content, m.metadata, distance
                FROM vec_memory v
                JOIN memory_meta m ON v.rowid = m.rowid
                WHERE embedding MATCH ? AND k = ?
                ORDER BY distance
            """
            cursor.execute(query, (sqlite_vec.serialize_float32(query_embedding), limit))
            return [{"id": r[0], "content": r[1], "metadata": r[2], "distance": r[3]} for r in cursor.fetchall()]
        return await asyncio.to_thread(_query)

class MemoryManager:
    """Unified memory manager combining Mem0 (Short-term) and sqlite-vec (Long-term)."""
    def __init__(self):
        # Mem0 uses local ephemeral/graph memory by default
        self.short_term = Memory() 
        self.long_term = SqliteVecStore()
        self.embedder = NvidiaEmbedder()

    async def store(self, content: str, metadata: Dict[str, Any] = None):
        """Stores in both Mem0 (graph context) and sqlite-vec (long-term vector)."""
        metadata = metadata or {}
        
        # 1. Store in Mem0
        self.short_term.add(content, user_id="agent_team", metadata=metadata)
        
        # 2. Store in Long-Term Sqlite-vec
        embedding = await self.embedder.get_embedding(content)
        await self.long_term.add_memory(embedding, content, str(metadata))
        print(f"[MemoryManager] Stored: '{content[:30]}...' in both systems.")

    async def retrieve(self, query: str) -> Dict[str, Any]:
        """Retrieves from both systems."""
        # Mem0 search
        st_results = self.short_term.search(query, user_id="agent_team")
        
        # Sqlite-vec search
        query_emb = await self.embedder.get_embedding(query)
        lt_results = await self.long_term.search(query_emb, limit=2)
        
        return {"short_term": st_results, "long_term": lt_results}

# ==========================================
# Demo Execution
# ==========================================
if __name__ == "__main__":
    async def demo():
        # Requires NVIDIA_EMBED_API_KEY environment variable for real vectors
        manager = MemoryManager()
        
        print("Storing memories...")
        await manager.store("The user prefers Python over JavaScript.", {"topic": "preferences"})
        await manager.store("The server IP is 192.168.1.100.", {"topic": "infrastructure"})
        
        print("\nRetrieving memories for query: 'What is the server address?'")
        results = await manager.retrieve("What is the server address?")
        
        print("\n--- Short Term (Mem0) ---")
        for res in results["short_term"]:
            print(res)
            
        print("\n--- Long Term (sqlite-vec) ---")
        for res in results["long_term"]:
            print(f"Content: {res['content']} | Distance: {res['distance']:.4f}")

    asyncio.run(demo())
