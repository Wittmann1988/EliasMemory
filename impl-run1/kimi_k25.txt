import logging

class Priority(IntEnum):
    LOW = 1
    NORMAL = 2
    HIGH = 3
    CRITICAL = 4

@dataclass
class Event:
    type: str
    payload: Dict[str, Any]
    correlation_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    timestamp: datetime = field(default_factory=datetime.utcnow)
    priority: Priority = Priority.NORMAL
    source: Optional[str] = None
    
    def to_json(self) -> str:
        return json.dumps({
            "type": self.type,
            "payload": self.payload,
            "correlation_id": self.correlation_id,
            "timestamp": self.timestamp.isoformat(),
            "priority": self.priority.value,
            "source": self.source
        })

class EventBus:
    def __init__(self, redis_url: Optional[str] = None):
        self._subscribers: Dict[str, Set[Callable[[Event], asyncio.Future]]] = {}
        self._queue: asyncio.PriorityQueue = asyncio.PriorityQueue()
        self._running = False
        self._task: Optional[asyncio.Task] = None
        self.redis_url = redis_url
        
    async def start(self):
        if not self._running:
            self._running = True
            self._task = asyncio.create_task(self._process_queue())
            logger.info("EventBus started")
    
    async def stop(self):
        self._running = False
        if self._task:
            self._task.cancel()
            try:
                await self._task
            except asyncio.CancelledError:
                pass
        logger.info("EventBus stopped")
    
    async def publish(self, event: Event) -> None:
        # Priority queue: lower number = higher priority in Python's PriorityQueue
        # We negate priority so higher enum values come out first
        await self._queue.put((-event.priority.value, event.timestamp.timestamp(), event))
        logger.debug(f"Published event {event.type} with corr_id {event.correlation_id}")
        
        # If Redis is configured, publish there too (stub for now)
        if self.redis_url:
            pass  # Redis implementation would go here
    
    def subscribe(self, event_type: str, handler: Callable[[Event], asyncio.Future]) -> None:
        if event_type not in self._subscribers:
            self._subscribers[event_type] = set()
        self._subscribers[event_type].add(handler)
        logger.info(f"Handler subscribed to {event_type}")
    
    def unsubscribe(self, event_type: str, handler: Callable[[Event], asyncio.Future]) -> None:
        if event_type in self._subscribers:
            self._subscribers[event_type].discard(handler)
    
    async def _process_queue(self):
        while self._running:
            try:
                # Wait for items with timeout to allow checking _running
                try:
                    _, _, event = await asyncio.wait_for(self._queue.get(), timeout=1.0)
                except asyncio.TimeoutError:
                    continue
                
                # Dispatch to subscribers
                handlers = self._subscribers.get(event.type, set()).copy()
                # Also dispatch to wildcard subscribers
                handlers.update(self._subscribers.get("*", set()))
                
                if handlers:
                    await asyncio.gather(
                        *[self._safe_handler(handler, event) for handler in handlers],
                        return_exceptions=True
                    )
            except Exception as e:
                logger.error(f"Error processing event queue: {e}")
    
    async def _safe_handler(self, handler: Callable[[Event], asyncio.Future], event: Event):
        try:
            await handler(event)
        except Exception as e:
            logger.error(f"Handler error for event {event.type}: {e}")

            "Content-Type": "application/json"

            "model": self.model,
            "input": texts,
            "encoding_format": "float"

                    self.endpoint, 
                    headers=headers,

                    timeout=aiohttp.ClientTimeout(total=30)
                ) as response:
                    if response.status != 200:
                        error_text = await response.text()
                        raise Exception(f"NVIDIA API error {response.status}: {error_text}")

import asyncio
from typing import TypedDict, Annotated, Sequence, List, Dict, Any, Optional
from dataclasses import dataclass, field
import operator
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.memory import MemorySaver
from langgraph.prebuilt import ToolNode
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class AgentState(TypedDict):
    messages: Annotated[Sequence[Dict[str, Any]], operator.add]
    task: str
    plan: Optional[str]
    actions: List[Dict[str, Any]]
    observations: List[Dict[str, Any]]
    reflection: Optional[str]
    next_step: Optional[str]
    correlation_id: str

class LangGraphOrchestrator:
    def __init__(self, memory_store=None, event_bus=None):
        self.memory = memory_store
        self.event_bus = event_bus
        self.graph = self._build_graph()
        self.checkpointer = MemorySaver()
        
    def _build_graph(self) -> StateGraph:
        workflow = StateGraph(AgentState)
        
        # Add nodes
        workflow.add_node("plan", self._plan_node)
        workflow.add_node("act", self._act_node)
        workflow.add_node("observe", self._observe_node)
        workflow.add_node("improve", self._improve_node)
        
        # Add edges
        workflow.set_entry_point("plan")
        workflow.add_edge("plan", "act")
        workflow.add_edge("act", "observe")
        
        # Conditional edge from observe
        workflow.add_conditional_edges(
            "observe",
            self._should_continue,
            {
                "improve": "improve",
                "end": END
            }
        )
        workflow.add_edge("improve", END)
        
        return workflow.compile(checkpointer=self.checkpointer)
    
    async def _plan_node(self, state: AgentState) -> AgentState:
        logger.info(f"Planning for task: {state['task']}")
        
        # Simulate planning logic
        plan = f"Plan for: {state['task']}\n1. Analyze requirements\n2. Execute steps\n3. Validate results"
        
        # Query memory for context
        if self.memory:
            try:
                context = await self.memory.search(state['task'])
                if context:
                    plan += f"\nContext from memory: {len(context)} relevant items found"
            except Exception as e:
                logger.error(f"Memory query error in planning: {e}")
        
        new_state = state.copy()
        new_state["plan"] = plan
        new_state["messages"] = list(state["messages"]) + [{"role": "planner", "content": plan}]
        
        # Publish event
        if self.event_bus:
            from event_bus import Event, Priority
            await self.event_bus.publish(Event(
                type="agent.plan.complete",
                payload={"task": state["task"], "plan": plan},
                correlation_id=state["correlation_id"],
                priority=Priority.NORMAL
            ))
        
        return new_state
    
    async def _act_node(self, state: AgentState) -> AgentState:
        logger.info(f"Executing actions for task: {state['task']}")
        
        # Simulate action execution
        actions = [
            {"tool": "search", "input": "relevant info", "status": "completed"},
            {"tool": "compute", "input": "process data", "status": "completed"}
        ]
        
        new_state = state.copy()
        new_state["actions"] = actions
        new_state["messages"] = list(state["messages"]) + [{"role": "actor", "content": f"Executed {len(actions)} actions"}]
        
        return new_state
    
    async def _observe_node(self, state: AgentState) -> AgentState:
        logger.info(f"Observing results for task: {state['task']}")
        
        # Simulate observation
        observations = [
            {"metric": "accuracy", "value": 0.95},
            {"metric": "latency", "value": "120ms"}
        ]
        
        new_state = state.copy()
        new_state["observations"] = observations
        new_state["messages"] = list(state["messages"]) + [{"role": "observer", "content": str(observations)}]
        
        # Determine if improvement is needed
        should_improve = any(obs["value"] < 0.9 for obs in observations if isinstance(obs.get("value"), (int, float)))
        new_state["next_step"] = "improve" if should_improve else "end"
        
        return new_state
    
    async def _improve_node(self, state: AgentState) -> AgentState:
        logger.info(f"Improving strategy for task: {state['task']}")
        
        reflection = f"Reflection: Based on observations {state['observations']}, optimizing approach..."
        
        new_state = state.copy()
        new_state["reflection"] = reflection
        new_state["messages"] = list(state["messages"]) + [{"role": "improver", "content": reflection}]
        
        # Store reflection in memory
        if self.memory:
            try:
                await self.memory.store(
                    f"Task: {state['task']}\nReflection: {reflection}",
                    {"type": "reflection", "task": state["task"]}
                )
            except Exception as e:
                logger.error(f"Memory storage error: {e}")
        
        return new_state
    
    def _should_continue(self, state: AgentState) -> str:
        return state.get("next_step", "end")
    
    async def run(self, task: str, thread_id: str = "default") -> Dict[str, Any]:
        initial_state = AgentState(
            messages=[],
            task=task,
            plan=None,
            actions=[],
            observations=[],
            reflection=None,
            next_step=None,
            correlation_id=thread_id
        )
        
        config = {"configurable": {"thread_id": thread_id}}
        
        # Run the graph
        result = await self.graph.ainvoke(initial_state, config)
        return result
