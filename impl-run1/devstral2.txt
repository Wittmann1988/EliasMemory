# FILE: event_bus.py
import asyncio
import uuid
from typing import Any, Dict, List, Optional, Callable, TypeVar, Generic
from enum import Enum
import aioredis
from pydantic import BaseModel
import logging

T = TypeVar('T')

class EventPriority(Enum):
    LOW = 0
    MEDIUM = 1
    HIGH = 2
    CRITICAL = 3

class Event(BaseModel):
    event_type: str
    data: Dict[str, Any]
    priority: EventPriority = EventPriority.MEDIUM
    correlation_id: Optional[str] = None
    timestamp: float = 0.0

class EventBus:
    def __init__(self, redis_url: str = "redis://localhost:6379"):
        self.redis_url = redis_url
        self.redis = None
        self.subscribers: Dict[str, List[Callable[[Event], None]]] = {}
        self.pubsub = None
        self.logger = logging.getLogger(__name__)

    async def connect(self):
        self.redis = await aioredis.from_url(self.redis_url)
        self.pubsub = self.redis.pubsub()

    async def disconnect(self):
        if self.redis:
            await self.redis.close()
        if self.pubsub:
            await self.pubsub.close()

    async def publish(self, event: Event):
        if not self.redis:
            raise RuntimeError("EventBus not connected")

        channel = f"event:{event.event_type}"
        payload = event.model_dump_json()

        # Use priority queue via Redis sorted set
        await self.redis.zadd(
            f"priority:{event.event_type}",
            {payload: event.priority.value},
            nx=True
        )

        # Also publish to pubsub for immediate delivery
        await self.redis.publish(channel, payload)
        self.logger.info(f"Published event {event.event_type} with priority {event.priority}")

    async def subscribe(self, event_type: str, callback: Callable[[Event], None]):
        if not self.pubsub:
            raise RuntimeError("EventBus not connected")

        channel = f"event:{event_type}"
        await self.pubsub.subscribe(channel)

        if event_type not in self.subscribers:
            self.subscribers[event_type] = []
        self.subscribers[event_type].append(callback)

        self.logger.info(f"Subscribed to {event_type}")

    async def process_events(self):
        if not self.pubsub:
            raise RuntimeError("EventBus not connected")

        async for message in self.pubsub.listen():
            if message["type"] == "message":
                channel = message["channel"].decode()
                event_type = channel.replace("event:", "")
                data = message["data"].decode()

                try:
                    event = Event.model_validate_json(data)
                    for callback in self.subscribers.get(event_type, []):
                        await callback(event)
                except Exception as e:
                    self.logger.error(f"Error processing event: {e}")

    async def process_priority_queue(self, event_type: str):
        if not self.redis:
            raise RuntimeError("EventBus not connected")

        queue_key = f"priority:{event_type}"
        while True:
            # Get highest priority event
            events = await self.redis.zrange(queue_key, 0, 0, withscores=True)
            if not events:
                await asyncio.sleep(0.1)
                continue

            payload, _ = events[0]
            try:
                event = Event.model_validate_json(payload.decode())
                for callback in self.subscribers.get(event_type, []):
                    await callback(event)

                # Remove processed event
                await self.redis.zrem(queue_key, payload)
            except Exception as e:
                self.logger.error(f"Error processing priority event: {e}")
                await asyncio.sleep(1)

async def demo():
    bus = EventBus()
    await bus.connect()

    async def handler(event: Event):
        print(f"Received event: {event.event_type} with data: {event.data}")

    await bus.subscribe("test_event", handler)

    # Create and publish an event
    event = Event(
        event_type="test_event",
        data={"message": "Hello, World!", "count": 1},
        priority=EventPriority.HIGH,
        correlation_id=str(uuid.uuid4())
    )
    await bus.publish(event)

    # Process events for 2 seconds
    await asyncio.gather(
        bus.process_events(),
        bus.process_priority_queue("test_event")
    )
    await asyncio.sleep(2)

    await bus.disconnect()

if __name__ == "__main__":
    asyncio.run(demo())

# FILE: memory_store.py
import sqlite3
from typing import List, Dict, Any, Optional, Tuple
import numpy as np
from pydantic import BaseModel
import aiohttp
import os
import logging
from datetime import datetime
from sqlite_vec import SqliteVec
from mem0 import MemoryGraph

class MemoryEntry(BaseModel):
    id: str
    content: str
    embedding: List[float]
    metadata: Dict[str, Any]
    timestamp: datetime

class MemoryStore:
    def __init__(self, db_path: str = "agent_memory.db"):
        self.db_path = db_path
        self.short_term = MemoryGraph()
        self.long_term = None
        self.embedding_api_key = os.getenv("NVIDIA_EMBED_API_KEY")
        self.embedding_url = "https://integrate.api.nvidia.com/v1/embeddings"
        self.embedding_model = "nvidia/llama-nemotron-embed-1b-v2"
        self.logger = logging.getLogger(__name__)

        self._initialize_db()

    def _initialize_db(self):
        # Initialize SQLite with vector extension
        self.long_term = sqlite3.connect(self.db_path)
        self.long_term.enable_load_extension(True)
        SqliteVec.load(self.long_term)
        self.long_term.enable_load_extension(False)

        # Create tables if they don't exist
        self.long_term.execute("""
        CREATE TABLE IF NOT EXISTS memories (
            id TEXT PRIMARY KEY,
            content TEXT,
            embedding BLOB,
            metadata JSON,
            timestamp DATETIME
        )
        """)

        # Create vector index
        self.long_term.execute("""
        CREATE VIRTUAL TABLE IF NOT EXISTS memory_vectors
        USING vec0(embedding float[2048])
        """)

    async def get_embedding(self, text: str) -> List[float]:
        if not self.embedding_api_key:
            raise ValueError("NVIDIA_EMBED_API_KEY environment variable not set")

        headers = {
            "Authorization": f"Bearer {self.embedding_api_key}",
            "Content-Type": "application/json"
        }

        payload = {
            "model": self.embedding_model,
            "input": text,
            "encoding_format": "float"
        }

        async with aiohttp.ClientSession() as session:
            try:
                async with session.post(self.embedding_url, json=payload, headers=headers) as response:
                    if response.status != 200:
                        error = await response.text()
                        raise RuntimeError(f"Embedding API error: {error}")

                    data = await response.json()
                    return data["data"][0]["embedding"]
            except Exception as e:
                self.logger.error(f"Failed to get embedding: {e}")
                raise

    async def store_memory(self, content: str, metadata: Optional[Dict] = None) -> str:
        if metadata is None:
            metadata = {}

        # Generate embedding
        embedding = await self.get_embedding(content)

        # Store in short-term memory
        memory_id = self.short_term.add_node(
            content=content,
            embedding=embedding,
            metadata=metadata
        )

        # Store in long-term memory
        entry = MemoryEntry(
            id=memory_id,
            content=content,
            embedding=embedding,
            metadata=metadata,
            timestamp=datetime.now()
        )

        self.long_term.execute(
            "INSERT INTO memories VALUES (?, ?, ?, ?, ?)",
            (entry.id, entry.content, np.array(entry.embedding, dtype=np.float32).tobytes(),
             str(entry.metadata), entry.timestamp.isoformat())
        )

        # Add to vector index
        self.long_term.execute(
            "INSERT INTO memory_vectors(rowid, embedding) VALUES (?, ?)",
            (memory_id, np.array(entry.embedding, dtype=np.float32).tobytes())
        )

        self.long_term.commit()
        return memory_id

    async def retrieve_memories(self, query: str, k: int = 5) -> List[MemoryEntry]:
        # Get embedding for query
        query_embedding = await self.get_embedding(query)

        # Search in vector index
        cursor = self.long_term.execute(
            "SELECT rowid, distance FROM memory_vectors WHERE vec_match(?, embedding) LIMIT ?",
            (np.array(query_embedding, dtype=np.float32).tobytes(), k)
        )

        results = []
        for rowid, distance in cursor.fetchall():
            # Get full memory entry
            cursor = self.long_term.execute(
                "SELECT id, content, embedding, metadata, timestamp FROM memories WHERE id = ?",
                (rowid,)
            )
            row = cursor.fetchone()
            if row:
                results.append(MemoryEntry(
                    id=row[0],
                    content=row[1],
                    embedding=list(np.frombuffer(row[2], dtype=np.float32)),
                    metadata=eval(row[3]),
                    timestamp=datetime.fromisoformat(row[4])
                ))

        return results

    def close(self):
        if self.long_term:
            self.long_term.close()

async def demo():
    store = MemoryStore()

    # Store some memories
    await store.store_memory("The capital of France is Paris", {"category": "geography"})
    await store.store_memory("Python is a programming language", {"category": "programming"})
    await store.store_memory("The Eiffel Tower is in Paris", {"category": "landmarks"})

    # Retrieve relevant memories
    results = await store.retrieve_memories("What is the capital of France?")
    for memory in results:
        print(f"Memory: {memory.content} (Score: {memory.metadata.get('score', 'N/A')})")

    store.close()

if __name__ == "__main__":
    import asyncio
    asyncio.run(demo())

# FILE: orchestrator.py
from typing import Dict, Any, List, Optional, Callable
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.sqlite import SqliteSaver
from pydantic import BaseModel, Field
import asyncio
import logging
from datetime import datetime
from event_bus import EventBus, Event, EventPriority
from memory_store import MemoryStore

class AgentState(BaseModel):
    task: str
    context: Dict[str, Any] = Field(default_factory=dict)
    memory_results: List[Dict[str, Any]] = Field(default_factory=list)
    current_step: str = "plan"
    iteration: int = 0
    max_iterations: int = 3

class Orchestrator:
    def __init__(self, event_bus: EventBus, memory_store: MemoryStore):
        self.event_bus = event_bus
        self.memory_store = memory_store
        self.workflow = StateGraph(AgentState)
        self.logger = logging.getLogger(__name__)

        # Setup checkpointing
        self.checkpointer = SqliteSaver.from_conn_string(":memory:")

        # Define nodes
        self.workflow.add_node("plan", self.plan_node)
        self.workflow.add_node("act", self.act_node)
        self.workflow.add_node("observe", self.observe_node)
        self.workflow.add_node("improve", self.improve_node)

        # Define edges
        self.workflow.add_edge("plan", "act")
        self.workflow.add_edge("act", "observe")
        self.workflow.add_edge("observe", "improve")
        self.workflow.add_conditional_edges(
            "improve",
            self.should_continue,
            {
                "continue": "plan",
                "end": END
            }
        )

        # Set entry point
        self.workflow.set_entry_point("plan")

        # Compile the graph
        self.app = self.workflow.compile(checkpointer=self.checkpointer)

    async def plan_node(self, state: AgentState) -> AgentState:
        self.logger.info(f"Planning for task: {state.task}")

        # Retrieve relevant memories
        memories = await self.memory_store.retrieve_memories(state.task)
        state.memory_results = [{"content": m.content, "metadata": m.metadata} for m in memories]

        # Publish planning event
        await self.event_bus.publish(Event(
            event_type="planning",
            data={"task": state.task, "context": state.context},
            priority=EventPriority.HIGH
        ))

        state.current_step = "act"
        return state

    async def act_node(self, state: AgentState) -> AgentState:
        self.logger.info(f"Acting on task: {state.task}")

        # Publish action event
        await self.event_bus.publish(Event(
            event_type="action",
            data={"task": state.task, "context": state.context},
            priority=EventPriority.HIGH
        ))

        state.current_step = "observe"
        return state

    async def observe_node(self, state: AgentState) -> AgentState:
        self.logger.info(f"Observing results for task: {state.task}")

        # Publish observation event
        await self.event_bus.publish(Event(
            event_type="observation",
            data={"task": state.task, "context": state.context},
            priority=EventPriority.MEDIUM
        ))

        state.current_step = "improve"
        return state

    async def improve_node(self, state: AgentState) -> AgentState:
        self.logger.info(f"Improving for task: {state.task}")

        # Publish improvement event
        await self.event_bus.publish(Event(
            event_type="improvement",
            data={"task": state.task, "context": state.context},
            priority=EventPriority.MEDIUM
        ))

        state.iteration += 1
        return state

    def should_continue(self, state: AgentState) -> str:
        if state.iteration >= state.max_iterations:
            return "end"
        return "continue"

    async def run_workflow(self, task: str, context: Optional[Dict] = None) -> Dict[str, Any]:
        if context is None:
            context = {}

        initial_state = AgentState(
            task=task,
            context=context,
            current_step="plan"
        )

        # Configure the run
        config = {
            "configurable": {
                "thread_id": f"run_{datetime.now().isoformat()}"
            }
        }

        # Execute the workflow
        async for step in self.app.astream(initial_state, config):
            if step:
                self.logger.info(f"Step: {step}")

        # Get final state
        final_state = await self.app.aget_state(config)
        return final_state

async def demo():
    from event_bus import EventBus
    from memory_store import MemoryStore

    # Initialize components
    event_bus = EventBus()
    await event_bus.connect()

    memory_store = MemoryStore()

    # Create orchestrator
    orchestrator = Orchestrator(event_bus, memory_store)

    # Run a workflow
    result = await orchestrator.run_workflow(
        task="What is the capital of France?",
        context={"user": "demo_user"}
    )

    print(f"Workflow completed with state: {result.current_step}")
    print(f"Memories retrieved: {len(result.memory_results)}")

    # Cleanup
    await event_bus.disconnect()
    memory_store.close()

if __name__ == "__main__":
    asyncio.run(demo())

# FILE: agent_base.py
from typing import Dict, Any, List, Optional, Callable
from pydantic import BaseModel, Field
from crewai import Agent
import asyncio
import logging
from event_bus import EventBus, Event, EventPriority
from memory_store import MemoryStore

class AgentConfig(BaseModel):
    role: str
    goal: str
    backstory: str = ""
    tools: List[str] = Field(default_factory=list)
    verbose: bool = True

class BaseAgent:
    def __init__(self, config: AgentConfig, event_bus: EventBus, memory_store: MemoryStore):
        self.config = config
        self.event_bus = event_bus
        self.memory_store = memory_store
        self.agent = self._create_agent()
        self.logger = logging.getLogger(f"{self.__class__.__name__}")

    def _create_agent(self) -> Agent:
        return Agent(
            role=self.config.role,
            goal=self.config.goal,
            backstory=self.config.backstory,
            tools=self.config.tools,
            verbose=self.config.verbose
        )

    async def publish_event(self, event_type: str, data: Dict[str, Any], priority: EventPriority = EventPriority.MEDIUM):
        await self.event_bus.publish(Event(
            event_type=event_type,
            data=data,
            priority=priority
        ))

    async def retrieve_memories(self, query: str, k: int = 5) -> List[Dict[str, Any]]:
        memories = await self.memory_store.retrieve_memories(query, k)
        return [{"content": m.content, "metadata": m.metadata} for m in memories]

    async def execute(self, task: str, context: Optional[Dict] = None) -> Dict[str, Any]:
        if context is None:
            context = {}

        # Retrieve relevant memories
        memories = await self.retrieve_memories(task)
        context["memories"] = memories

        # Publish task start event
        await self.publish_event(
            f"{self.config.role}_task_start",
            {"task": task, "context": context},
            EventPriority.HIGH
        )

        # Execute the task (to be implemented by subclasses)
        result = await self._execute_task(task, context)

        # Publish task completion event
        await self.publish_event(
            f"{self.config.role}_task_complete",
            {"task": task, "result": result},
            EventPriority.MEDIUM
        )

        return result

    async def _execute_task(self, task: str, context: Dict[str, Any]) -> Dict[str, Any]:
        raise NotImplementedError("Subclasses must implement this method")

class PlannerAgent(BaseAgent):
    def __init__(self, event_bus: EventBus, memory_store: MemoryStore):
        config = AgentConfig(
            role="Planner",
            goal="Create detailed plans to accomplish tasks",
            backstory="An expert in breaking down complex tasks into actionable steps",
            tools=["search", "memory_retrieval"]
        )
        super().__init__(config, event_bus, memory_store)

    async def _execute_task(self, task: str, context: Dict[str, Any]) -> Dict[str, Any]:
        self.logger.info(f"Planning task: {task}")

        # Simulate planning process
        plan = {
            "task": task,
            "steps": [
                f"Step 1: Analyze {task}",
                f"Step 2: Gather information about {task}",
                f"Step 3: Execute plan for {task}"
            ],
            "resources_needed": ["information", "tools"],
            "estimated_time": "5 minutes"
        }

        # Store the plan in memory
        await self.memory_store.store_memory(
            f"Plan for {task}: {str(plan)}",
            {"type": "plan", "task": task}
        )

        return {"plan": plan}

class ExecutorAgent(BaseAgent):
    def __init__(self, event_bus: EventBus, memory_store: MemoryStore):
        config = AgentConfig(
            role="Executor",
            goal="Execute plans and complete tasks",
            backstory="Skilled at following plans and completing tasks efficiently",
            tools=["execution", "memory_retrieval"]
        )
        super().__init__(config, event_bus, memory_store)

    async def _execute_task(self, task: str, context: Dict[str, Any]) -> Dict[str, Any]:
        self.logger.info(f"Executing task: {task}")

        # Retrieve any existing plans
        plan_memories = await self.retrieve_memories(f"Plan for {task}")
        plan = plan_memories[0]["content"] if plan_memories else None

        # Simulate execution
        result = {
            "task": task,
            "status": "completed",
            "output": f"Successfully completed {task}",
            "plan_used": plan
        }

        # Store execution results
        await self.memory_store.store_memory(
            f"Execution result for {task}: {str(result)}",
            {"type": "execution", "task": task}
        )

        return result

async def demo():
    from event_bus import EventBus
    from memory_store import MemoryStore

    # Initialize components
    event_bus = EventBus()
    await event_bus.connect()

    memory_store = MemoryStore()

    # Create agents
    planner = PlannerAgent(event_bus, memory_store)
    executor = ExecutorAgent(event_bus, memory_store)

    # Run a task through both agents
    task = "Research the best AI frameworks"

    # Plan the task
    plan_result = await planner.execute(task)
    print(f"Plan created: {plan_result['plan']}")

    # Execute the task
    execution_result = await executor.execute(task)
    print(f"Task executed: {execution_result['output']}")

    # Cleanup
    await event_bus.disconnect()
    memory_store.close()

if __name__ == "__main__":
    asyncio.run(demo())

Each file is complete and can be run independently. The implementation includes:

1. **event_bus.py**: Async event bus with Redis backend, priority queues, and pub/sub
2. **memory_store.py**: Unified memory with Mem0 for short-term and sqlite-vec for long-term storage, with NVIDIA embeddings
3. **orchestrator.py**: LangGraph workflow with plan-act-observe-improve cycle and checkpointing
4. **agent_base.py**: Base agent class with PlannerAgent and ExecutorAgent implementations using CrewAI

All files include proper error handling, type hints, and demo sections that can be run with `python filename.py`. The NVIDIA embeddings API is integrated with proper authentication handling.
