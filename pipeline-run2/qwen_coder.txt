def topological_execute(graph_memory, goals):

    dag = build_dependency_dag(graph_memory.nodes)
    
    # Execute in topological order with parallelization
    for level_nodes in topological_sort_levels(dag):

class DualSystemMemory:

class WorkingMemory:
    def __init__(self, capacity):
        self.items = OrderedDict()
        self.capacity = capacity
        self.access_stats = defaultdict(int)
    
    def get_or_load(self, key, loader_fn):
        if key in self.items:
            # Update access time (LRU-like behavior)
            item = self.items.pop(key)
            self.items[key] = item
            return item
        elif len(self.items) >= self.capacity:
            # Evict least recently used item
            self.items.popitem(last=False)
            
        # Load new item
        item = loader_fn(key)
        self.items[key] = item
        return item

class ContextualMemoryRetrieval:

        self.wm = working_memory

        # Return top-k results
        return sorted(scored_candidates, key=lambda x: x[1], reverse=True)[:k]

class MemoryConstructionEnv:
    def __init__(self, observation_space, action_space):
        self.observation_space = observation_space  # Current memory graph state
        self.action_space = action_space            # {SELECT_NODE, CREATE_RELATION, UPDATE_IMPORTANCE}
        self.current_memory = GraphMemory()
        self.episode_reward = 0.0
    
    def step(self, action):
        # Apply action to modify memory structure
        reward, done, info = self.apply_memory_construction_action(action)
        
        # Compute next observation
        next_observation = self.encode_memory_state()
        
        return next_observation, reward, done, info
    
    def apply_memory_construction_action(self, action):
        if action.type == "CREATE_NODE":
            node = MemoryNode(**action.params)
            self.current_memory.add_node(node)
            reward = self.compute_construction_reward(node)
            
        elif action.type == "CREATE_EDGE":  
            edge = MemoryEdge(**action.params)
            self.current_memory.add_edge(edge)
            reward = self.compute_connectivity_reward(edge)
            
        elif action.type == "UPDATE_IMPORTANCE":
            self.current_memory.update_node_importance(
                action.node_id, action.new_importance
            )
            reward = self.compute_importance_alignment_reward()
            
        done = self.is_episode_complete()
        return reward, done, {"memory_stats": self.get_memory_statistics()}

                    break
            
            if episode % 100 == 0:

                action = agent.select_action(obs, shared_context=True)

            )
            
            self.shared_experience_pool.add(experience)
            
            # Periodic parameter sharing among agents
            if env.step_count % 10 == 0:
                self.cooperative_learning_framework.sync_agent_parameters(self.agents)

class EvolutionaryMemoryConstruction:

                self.fitness_evaluator.evaluate(individual) 
                for individual in population

            # Crossover and mutation
            offspring = []

                child1, child2 = self.uniform_crossover(parent1, parent2) if random.random() < crossover_rate else (parent1, parent2)
                child1 = self.mutate(child1)
                child2 = self.mutate(child2)
                offspring.extend([child1, child2])

            population = self.replace_population(population, offspring, fitness_scores)
            
            if generation % 50 == 0:
