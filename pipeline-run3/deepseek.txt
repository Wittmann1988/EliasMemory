# core/event_bus.py
import asyncio
import json
from dataclasses import dataclass, field
from typing import Any, Dict, List, Callable, Optional
from datetime import datetime
import msgpack
from enum import Enum

class EventPriority(Enum):
    SYSTEM_CRITICAL = 0
    TASK_EXECUTION = 1
    MEMORY_UPDATE = 2
    LEARNING_SIGNAL = 3
    MONITORING = 4

@dataclass
class Event:
    """Unified event structure"""
    event_id: str
    event_type: str
    source: str  # Agent ID or system component
    target: Optional[str] = None  # Optional specific recipient
    priority: EventPriority = EventPriority.TASK_EXECUTION
    timestamp: datetime = field(default_factory=datetime.now)
    payload: Dict[str, Any] = field(default_factory=dict)
    metadata: Dict[str, Any] = field(default_factory=dict)
    requires_ack: bool = False
    ack_received: bool = False
    
    def to_bytes(self) -> bytes:
        """Serialize for inter-process communication"""
        data = {
            'event_id': self.event_id,
            'event_type': self.event_type,
            'source': self.source,
            'priority': self.priority.value,
            'timestamp': self.timestamp.isoformat(),
            'payload': self.payload,
            'metadata': self.metadata
        }
        return msgpack.packb(data)
    
    @classmethod
    def from_bytes(cls, data: bytes):
        """Deserialize from bytes"""
        decoded = msgpack.unpackb(data)
        return cls(
            event_id=decoded['event_id'],
            event_type=decoded['event_type'],
            source=decoded['source'],
            priority=EventPriority(decoded['priority']),
            timestamp=datetime.fromisoformat(decoded['timestamp']),
            payload=decoded['payload'],
            metadata=decoded['metadata']
        )

class EventBus:
    """Multi-platform optimized event bus with QoS"""
    
    def __init__(self, platform: str = "desktop"):
        self.platform = platform
        self.subscribers: Dict[str, List[Callable]] = {}
        self.event_queue = asyncio.PriorityQueue(maxsize=10000)
        self.dead_letter_queue = asyncio.Queue()
        self.ack_timeout = 5.0  # seconds
        
        # Platform-specific optimizations
        if platform == "termux":
            self.max_queue_size = 1000  # Lower memory footprint
            self.batch_size = 10
        elif platform == "jetson":
            self.max_queue_size = 10000
            self.batch_size = 100
        else:
            self.max_queue_size = 50000
            self.batch_size = 500
            
        self._processing_tasks = []
        
    async def publish(self, event: Event):
        """Publish event with priority queuing"""
        if self.event_queue.qsize() >= self.max_queue_size:
            # Apply backpressure strategy
            await self._apply_backpressure()
        
        priority_weight = event.priority.value
        await self.event_queue.put((priority_weight, event))
        
        # Wait for acknowledgment if required
        if event.requires_ack:
            try:
                await asyncio.wait_for(
                    self._wait_for_ack(event.event_id),
                    timeout=self.ack_timeout
                )
            except asyncio.TimeoutError:
                await self.dead_letter_queue.put(event)
                return False
        return True
    
    async def subscribe(self, event_type: str, callback: Callable):
        """Subscribe to specific event types"""
        if event_type not in self.subscribers:
            self.subscribers[event_type] = []
        self.subscribers[event_type].append(callback)
        
    async def _event_processor(self):
        """Core event processing loop"""
        while True:
            try:
                _, event = await self.event_queue.get()
                
                # Dispatch to subscribers
                if event.event_type in self.subscribers:
                    tasks = []
                    for callback in self.subscribers[event.event_type]:
                        if event.target is None or event.target == callback.__name__:
                            tasks.append(callback(event))
                    
                    # Process in parallel or sequence based on platform
                    if self.platform == "termux":
                        for task in tasks:
                            await task  # Sequential for stability
                    else:
                        await asyncio.gather(*tasks, return_exceptions=True)
                
                self.event_queue.task_done()
                
            except Exception as e:
                print(f"Event processing error: {e}")
                await self.dead_letter_queue.put(event)
