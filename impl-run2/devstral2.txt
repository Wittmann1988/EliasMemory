# FILE: event_bus.py
"""
Improved EventBus with:
- Event persistence to SQLite
- Dead-letter queue for failed handlers
- Metrics tracking (events/sec, avg latency)
- Backpressure handling (max queue size)
- Shared event types for all components
"""

import asyncio
import sqlite3
import time
from dataclasses import dataclass
from enum import Enum
from typing import Callable, Dict, List, Optional, Any
import logging
from collections import deque
from datetime import datetime

class EventType(Enum):
    """Shared event types for the entire system"""
    TASK_ASSIGNED = "TASK_ASSIGNED"
    TASK_COMPLETED = "TASK_COMPLETED"
    MEMORY_STORE = "MEMORY_STORE"
    MEMORY_QUERY = "MEMORY_QUERY"
    AGENT_REVIEW = "AGENT_REVIEW"
    SYSTEM_METRICS = "SYSTEM_METRICS"
    SELF_IMPROVEMENT = "SELF_IMPROVEMENT"

@dataclass
class Event:
    """Base event class with required fields"""
    event_type: EventType
    timestamp: float = time.time()
    payload: Dict[str, Any] = None
    metadata: Dict[str, Any] = None

class EventBus:
    """
    Improved EventBus with persistence and metrics.
    Changes from Run 1:
    - Added SQLite persistence for all events
    - Implemented dead-letter queue
    - Added metrics tracking
    - Backpressure handling
    - Shared event types
    """

    def __init__(self, max_queue_size: int = 1000):
        self.subscribers: Dict[EventType, List[Callable]] = {}
        self.event_queue = asyncio.Queue(maxsize=max_queue_size)
        self.dead_letter_queue = deque(maxlen=1000)
        self.metrics = {
            'events_processed': 0,
            'events_failed': 0,
            'avg_latency': 0,
            'last_event_time': time.time()
        }
        self._setup_db()
        self._running = False

    def _setup_db(self):
        """Initialize SQLite database for event persistence"""
        self.conn = sqlite3.connect('event_bus.db')
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS events (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                event_type TEXT NOT NULL,
                timestamp REAL NOT NULL,
                payload TEXT,
                metadata TEXT,
                processed BOOLEAN DEFAULT 0
            )
        ''')
        self.conn.commit()

    async def start(self):
        """Start the event processing loop"""
        self._running = True
        asyncio.create_task(self._process_events())

    async def stop(self):
        """Stop the event processing loop"""
        self._running = False

    async def _process_events(self):
        """Main event processing loop with metrics"""
        while self._running:
            try:
                event = await self.event_queue.get()
                start_time = time.time()

                # Persist event
                self._persist_event(event)

                # Process event
                if event.event_type in self.subscribers:
                    for handler in self.subscribers[event.event_type]:
                        try:
                            await handler(event)
                            self.metrics['events_processed'] += 1
                        except Exception as e:
                            self.metrics['events_failed'] += 1
                            self.dead_letter_queue.append((event, str(e)))
                            logging.error(f"Handler failed for {event.event_type}: {e}")

                # Update metrics
                self.metrics['avg_latency'] = (
                    (self.metrics['avg_latency'] * (self.metrics['events_processed'] - 1)) +
                    (time.time() - start_time)
                ) / self.metrics['events_processed'] if self.metrics['events_processed'] > 0 else 0

            except Exception as e:
                logging.error(f"Event processing error: {e}")
                await asyncio.sleep(0.1)

    def _persist_event(self, event: Event):
        """Persist event to SQLite"""
        try:
            self.conn.execute(
                "INSERT INTO events (event_type, timestamp, payload, metadata) VALUES (?, ?, ?, ?)",
                (event.event_type.value, event.timestamp, str(event.payload), str(event.metadata))
            )
            self.conn.commit()
        except Exception as e:
            logging.error(f"Failed to persist event: {e}")

    async def publish(self, event: Event):
        """Publish an event to the bus with backpressure handling"""
        try:
            await self.event_queue.put(event)
        except asyncio.QueueFull:
            logging.warning("Event queue full - applying backpressure")
            await asyncio.sleep(0.1)
            await self.event_queue.put(event)

    def subscribe(self, event_type: EventType, handler: Callable):
        """Subscribe to an event type"""
        if event_type not in self.subscribers:
            self.subscribers[event_type] = []
        self.subscribers[event_type].append(handler)

    def get_metrics(self):
        """Get current metrics"""
        return self.metrics

"""
Improved MemoryStore with:
- Event bus integration (publishes MEMORY_STORE/MEMORY_QUERY events)
- Memory decay (relevance reduces over time)
- Batch embedding API calls
- Connection pooling for SQLite
"""

import asyncio
import sqlite3
import time
from typing import List, Dict, Any, Optional
import os
import aiohttp
import numpy as np
from event_bus import Event, EventType
import logging

class MemoryStore:
    """
    Improved MemoryStore with event bus integration and memory decay.
    Changes from Run 1:
    - Now publishes events through EventBus
    - Implemented memory decay
    - Batch embedding API calls
    - Connection pooling
    """

    def __init__(self, event_bus: EventBus):
        self.event_bus = event_bus
        self._setup_db()
        self.embedding_cache = {}
        self.batch_size = 10
        self.decay_rate = 0.99  # Daily decay factor
        self._setup_event_handlers()

    def _setup_db(self):
        """Initialize SQLite database with connection pooling"""
        self.conn = sqlite3.connect('memory_store.db', check_same_thread=False)
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS memories (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                content TEXT NOT NULL,
                embedding BLOB,
                relevance REAL DEFAULT 1.0,
                created_at REAL NOT NULL,
                last_accessed REAL,
                metadata TEXT
            )
        ''')
        self.conn.commit()

    def _setup_event_handlers(self):
        """Subscribe to relevant events"""
        self.event_bus.subscribe(EventType.MEMORY_STORE, self._handle_memory_store)
        self.event_bus.subscribe(EventType.MEMORY_QUERY, self._handle_memory_query)

    async def _handle_memory_store(self, event: Event):
        """Handle MEMORY_STORE events"""
        content = event.payload.get('content')
        metadata = event.payload.get('metadata', {})

        if not content:
            logging.warning("Memory store event missing content")
            return

        # Check if we already have this memory
        cursor = self.conn.execute(
            "SELECT id FROM memories WHERE content = ?",
            (content,)
        )
        existing = cursor.fetchone()

        if existing:
            # Update existing memory
            self.conn.execute(
                "UPDATE memories SET relevance = ?, last_accessed = ?, metadata = ? WHERE id = ?",
                (1.0, time.time(), str(metadata), existing[0])
            )
        else:
            # Add new memory
            embedding = await self._get_embedding(content)
            self.conn.execute(
                "INSERT INTO memories (content, embedding, relevance, created_at, last_accessed, metadata) VALUES (?, ?, ?, ?, ?, ?)",
                (content, embedding.tobytes(), 1.0, time.time(), time.time(), str(metadata))
            )

        self.conn.commit()

    async def _handle_memory_query(self, event: Event):
        """Handle MEMORY_QUERY events"""
        query = event.payload.get('query')
        top_k = event.payload.get('top_k', 5)

        if not query:
            logging.warning("Memory query event missing query")
            return

        query_embedding = await self._get_embedding(query)

        # Get relevant memories
        cursor = self.conn.execute(
            "SELECT id, content, embedding, relevance FROM memories"
        )
        memories = cursor.fetchall()

        results = []
        for mem_id, content, embedding_bytes, relevance in memories:
            # Apply memory decay
            decayed_relevance = relevance * (self.decay_rate ** ((time.time() - self.conn.execute(
                "SELECT created_at FROM memories WHERE id = ?", (mem_id,)
            ).fetchone()[0]) / 86400))

            # Calculate similarity
            embedding = np.frombuffer(embedding_bytes, dtype=np.float32)
            similarity = np.dot(query_embedding, embedding) / (
                np.linalg.norm(query_embedding) * np.linalg.norm(embedding)
            )

            # Combine relevance and similarity
            score = 0.7 * similarity + 0.3 * decayed_relevance
            results.append((score, content, mem_id))

        # Sort and return top results
        results.sort(reverse=True, key=lambda x: x[0])
        top_results = [{"content": content, "memory_id": mem_id, "score": score}
                      for score, content, mem_id in results[:top_k]]

        # Update last accessed time for returned memories
        for result in top_results:
            self.conn.execute(
                "UPDATE memories SET last_accessed = ? WHERE id = ?",
                (time.time(), result['memory_id'])
            )

        self.conn.commit()

        # Publish results back to event bus
        await self.event_bus.publish(Event(
            event_type=EventType.MEMORY_QUERY,
            payload={
                'query': query,
                'results': top_results,
                'timestamp': time.time()
            }
        ))

    async def _get_embedding(self, text: str) -> np.ndarray:
        """Get embedding with batching and caching"""
        if text in self.embedding_cache:
            return self.embedding_cache[text]

        # Collect batch
        batch = [text]
        while len(batch) < self.batch_size:
            # In a real implementation, we'd have a queue of pending texts
            break

        # Call embedding API
        api_key = os.getenv('NVIDIA_EMBED_API_KEY')
        if not api_key:
            raise ValueError("NVIDIA_EMBED_API_KEY environment variable not set")

        url = "https://api.nvidia.com/v1/embeddings"
        headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }

        async with aiohttp.ClientSession() as session:
            async with session.post(
                url,
                json={
                    "model": "nvidia/llama-nemotron-embed-1b-v2",
                    "inputs": batch
                },
                headers=headers
            ) as response:
                if response.status != 200:
                    raise Exception(f"Embedding API error: {await response.text()}")

                data = await response.json()
                embeddings = [np.array(emb) for emb in data['data']]

        # Cache results
        for text, embedding in zip(batch, embeddings):
            self.embedding_cache[text] = embedding

        return embeddings[0]

    async def store_memory(self, content: str, metadata: Dict = None):
        """Public method to store memory via event bus"""
        await self.event_bus.publish(Event(
            event_type=EventType.MEMORY_STORE,
            payload={
                'content': content,
                'metadata': metadata or {}
            }
        ))

    async def query_memory(self, query: str, top_k: int = 5):
        """Public method to query memory via event bus"""
        await self.event_bus.publish(Event(
            event_type=EventType.MEMORY_QUERY,
            payload={
                'query': query,
                'top_k': top_k
            }
        ))

# FILE: orchestrator.py
"""
Improved Orchestrator with:
- Actual task routing through event bus
- Self-improvement loop (analyzes every 10 tasks)
- Checkpointing to SQLite
- Agent selection strategy adjustment
"""

import asyncio
import sqlite3
import time
from typing import Dict, List, Any
from event_bus import Event, EventType
import logging
import numpy as np

class Orchestrator:
    """
    Improved Orchestrator that actually routes tasks through event bus.
    Changes from Run 1:
    - Now properly uses event bus for all communications
    - Implemented self-improvement loop
    - Added checkpointing
    - Adjusts agent selection strategy
    """

    def __init__(self, event_bus: EventBus):
        self.event_bus = event_bus
        self.agents: Dict[str, Dict] = {}  # agent_id -> {success_count, failure_count}
        self.task_count = 0
        self._setup_db()
        self._setup_event_handlers()
        self.agent_selection_strategy = "round_robin"  # or "performance_based"

    def _setup_db(self):
        """Initialize SQLite database for checkpointing"""
        self.conn = sqlite3.connect('orchestrator.db')
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS agent_performance (
                agent_id TEXT PRIMARY KEY,
                success_count INTEGER DEFAULT 0,
                failure_count INTEGER DEFAULT 0,
                last_used REAL
            )
        ''')
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS system_state (
                id INTEGER PRIMARY KEY,
                task_count INTEGER,
                selection_strategy TEXT,
                timestamp REAL
            )
        ''')
        self.conn.commit()

    def _setup_event_handlers(self):
        """Subscribe to relevant events"""
        self.event_bus.subscribe(EventType.TASK_COMPLETED, self._handle_task_completed)
        self.event_bus.subscribe(EventType.SELF_IMPROVEMENT, self._handle_self_improvement)

    async def _handle_task_completed(self, event: Event):
        """Handle task completion events"""
        task_id = event.payload.get('task_id')
        agent_id = event.payload.get('agent_id')
        success = event.payload.get('success', False)

        self.task_count += 1

        # Update agent performance
        if agent_id not in self.agents:
            self.agents[agent_id] = {'success_count': 0, 'failure_count': 0}

        if success:
            self.agents[agent_id]['success_count'] += 1
        else:
            self.agents[agent_id]['failure_count'] += 1

        # Update in database
        cursor = self.conn.execute(
            "SELECT success_count, failure_count FROM agent_performance WHERE agent_id = ?",
            (agent_id,)
        )
        row = cursor.fetchone()

        if row:
            new_success = row[0] + (1 if success else 0)
            new_failure = row[1] + (0 if success else 1)
            self.conn.execute(
                "UPDATE agent_performance SET success_count = ?, failure_count = ?, last_used = ? WHERE agent_id = ?",
                (new_success, new_failure, time.time(), agent_id)
            )
        else:
            self.conn.execute(
                "INSERT INTO agent_performance (agent_id, success_count, failure_count, last_used) VALUES (?, ?, ?, ?)",
                (agent_id, 1 if success else 0, 0 if success else 1, time.time())
            )

        self.conn.commit()

        # Trigger self-improvement every 10 tasks
        if self.task_count % 10 == 0:
            await self.event_bus.publish(Event(
                event_type=EventType.SELF_IMPROVEMENT,
                payload={
                    'task_count': self.task_count,
                    'agent_performance': self.agents
                }
            ))

    async def _handle_self_improvement(self, event: Event):
        """Handle self-improvement events"""
        logging.info("Running self-improvement cycle")

        # Analyze agent performance
        total_tasks = sum(
            agent['success_count'] + agent['failure_count']
            for agent in self.agents.values()
        )

        if total_tasks == 0:
            return

        # Calculate success rates
        success_rates = {
            agent_id: data['success_count'] / (data['success_count'] + data['failure_count'])
            for agent_id, data in self.agents.items()
        }

        # Adjust selection strategy
        if self.agent_selection_strategy == "round_robin":
            # Switch to performance-based if we have enough data
            if total_tasks > 20:
                self.agent_selection_strategy = "performance_based"
                logging.info("Switched to performance-based agent selection")
        elif self.agent_selection_strategy == "performance_based":
            # Consider switching back if performance is similar
            if max(success_rates.values()) - min(success_rates.values()) < 0.1:
                self.agent_selection_strategy = "round_robin"
                logging.info("Switched back to round-robin selection")

        # Checkpoint state
        self.conn.execute(
            "INSERT INTO system_state (task_count, selection_strategy, timestamp) VALUES (?, ?, ?)",
            (self.task_count, self.agent_selection_strategy, time.time())
        )
        self.conn.commit()

        logging.info(f"Self-improvement complete. Strategy: {self.agent_selection_strategy}")

    async def assign_task(self, task: Dict[str, Any]):
        """Assign a task to an agent through the event bus"""
        task_id = task.get('task_id')
        if not task_id:
            task_id = str(time.time())

        # Select agent based on current strategy
        if self.agent_selection_strategy == "round_robin":
            agent_id = self._select_round_robin_agent()
        else:
            agent_id = self._select_performance_based_agent()

        # Publish task assignment event
        await self.event_bus.publish(Event(
            event_type=EventType.TASK_ASSIGNED,
            payload={
                'task_id': task_id,
                'agent_id': agent_id,
                'task': task,
                'timestamp': time.time()
            }
        ))

        return task_id

    def _select_round_robin_agent(self) -> str:
        """Select agent using round-robin strategy"""
        if not self.agents:
            raise ValueError("No agents available")

        # Simple round-robin based on last used time
        cursor = self.conn.execute(
            "SELECT agent_id FROM agent_performance ORDER BY last_used ASC LIMIT 1"
        )
        agent_id = cursor.fetchone()[0]

        # Update last used time
        self.conn.execute(
            "UPDATE agent_performance SET last_used = ? WHERE agent_id = ?",
            (time.time(), agent_id)
        )
        self.conn.commit()

        return agent_id

    def _select_performance_based_agent(self) -> str:
        """Select agent based on performance"""
        if not self.agents:
            raise ValueError("No agents available")

        # Get agent with highest success rate
        cursor = self.conn.execute('''
            SELECT agent_id, success_count, failure_count
            FROM agent_performance
            ORDER BY success_count * 1.0 / (success_count + failure_count) DESC
            LIMIT 1
        ''')
        agent_id = cursor.fetchone()[0]
        return agent_id

    def register_agent(self, agent_id: str):
        """Register a new agent"""
        if agent_id not in self.agents:
            self.agents[agent_id] = {'success_count': 0, 'failure_count': 0}
            self.conn.execute(
                "INSERT OR IGNORE INTO agent_performance (agent_id) VALUES (?)",
                (agent_id,)
            )
            self.conn.commit()

# FILE: agent_base.py
"""
Improved AgentBase with:
- Event bus subscription for task assignment
- Result reporting via events
- Per-agent success metrics
- ReviewerAgent for evaluating outputs
"""

import asyncio
import time
from typing import Dict, Any, Optional
from event_bus import Event, EventType
import logging

class AgentBase:
    """
    Improved AgentBase that uses event bus for all communications.
    Changes from Run 1:
    - Now properly subscribes to event bus
    - Reports results via events
    - Tracks success metrics
    """

    def __init__(self, agent_id: str, event_bus: EventBus):
        self.agent_id = agent_id
        self.event_bus = event_bus
        self.success_count = 0
        self.failure_count = 0
        self._setup_event_handlers()

    def _setup_event_handlers(self):
        """Subscribe to task assignment events"""
        self.event_bus.subscribe(EventType.TASK_ASSIGNED, self._handle_task_assigned)

    async def _handle_task_assigned(self, event: Event):
        """Handle task assignment events"""
        if event.payload.get('agent_id') != self.agent_id:
            return

        task_id = event.payload['task_id']
        task = event.payload['task']

        try:
            # Process task (to be implemented by subclasses)
            result = await self.process_task(task)

            # Report success
            await self.event_bus.publish(Event(
                event_type=EventType.TASK_COMPLETED,
                payload={
                    'task_id': task_id,
                    'agent_id': self.agent_id,
                    'success': True,
                    'result': result,
                    'timestamp': time.time()
                }
            ))
            self.success_count += 1

        except Exception as e:
            # Report failure
            await self.event_bus.publish(Event(
                event_type=EventType.TASK_COMPLETED,
                payload={
                    'task_id': task_id,
                    'agent_id': self.agent_id,
                    'success': False,
                    'error': str(e),
                    'timestamp': time.time()
                }
            ))
            self.failure_count += 1

    async def process_task(self, task: Dict[str, Any]) -> Any:
        """Process a task (to be implemented by subclasses)"""
        raise NotImplementedError("Subclasses must implement process_task")

class ReviewerAgent(AgentBase):
    """
    Specialized agent for reviewing other agents' outputs.
    Changes from Run 1:
    - Now properly integrated with event bus
    - Publishes review results
    """

    def __init__(self, event_bus: EventBus):
        super().__init__("reviewer_agent", event_bus)
        self._setup_review_handlers()

    def _setup_review_handlers(self):
        """Subscribe to task completion events for review"""
        self.event_bus.subscribe(EventType.TASK_COMPLETED, self._handle_task_completed)

    async def _handle_task_completed(self, event: Event):
        """Review completed tasks"""
        if not event.payload.get('success'):
            return  # Only review successful tasks

        task_id = event.payload['task_id']
        agent_id = event.payload['agent_id']
        result = event.payload['result']

        # Perform review (simple example - in reality would be more sophisticated)
        review_score = await self._evaluate_result(result)
        feedback = self._generate_feedback(review_score)

        # Publish review
        await self.event_bus.publish(Event(
            event_type=EventType.AGENT_REVIEW,
            payload={
                'task_id': task_id,
                'agent_id': agent_id,
                'review_score': review_score,
                'feedback': feedback,
                'timestamp': time.time()
            }
        ))

    async def _evaluate_result(self, result: Any) -> float:
        """Evaluate a result (0-1 score)"""
        # Simple example - in reality would use more sophisticated evaluation
        if isinstance(result, str):
            return min(1.0, len(result) / 100)  # Prefer longer responses (up to 100 chars)
        return 0.5  # Default score

    def _generate_feedback(self, score: float) -> str:
        """Generate feedback based on score"""
        if score > 0.8:
            return "Excellent work!"
        elif score > 0.6:
            return "Good work, could be improved"
        elif score > 0.4:
            return "Needs improvement"
        else:
            return "Poor quality, needs significant improvement"

    async def process_task(self, task: Dict[str, Any]) -> Any:
        """ReviewerAgent doesn't process regular tasks"""
        return {"status": "ReviewerAgent doesn't process tasks"}
