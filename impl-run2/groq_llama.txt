# FILE: event_bus.py
"""
Event Bus class with improvements from Run 1.

Changes:
- Added event persistence (log to sqlite)
- Implemented dead-letter queue for failed handlers
- Added metrics (events/sec, avg latency)
- Implemented backpressure (max queue size)

"""
import asyncio
from typing import Callable, Dict
import sqlite3
import os
from pydantic import BaseModel
import logging
from langchain import LLMChain

# Define event types
class EventType(BaseModel):
    TASK_ASSIGNED: str
    TASK_COMPLETED: str
    MEMORY_STORE: str
    MEMORY_QUERY: str

# Initialize event types
EVENT_TYPES = EventType(
    TASK_ASSIGNED="task_assigned",
    TASK_COMPLETED="task_completed",
    MEMORY_STORE="memory_store",
    MEMORY_QUERY="memory_query"
)

class EventBus:
    def __init__(self, max_queue_size: int = 1000):
        self.max_queue_size = max_queue_size
        self.queue = asyncio.Queue(max_queue_size)
        self.handlers = {}
        self.dead_letter_queue = []
        self.metrics = {"events/sec": 0, "avg_latency": 0}
        self.conn = sqlite3.connect("event_bus.db")
        self.cursor = self.conn.cursor()
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS events (
                id INTEGER PRIMARY KEY,
                event_type TEXT,
                event_data TEXT
            )
        """)
        self.conn.commit()

    async def add_handler(self, event_type: str, handler: Callable):
        if event_type not in self.handlers:
            self.handlers[event_type] = []
        self.handlers[event_type].append(handler)

    async def remove_handler(self, event_type: str, handler: Callable):
        if event_type in self.handlers:
            self.handlers[event_type].remove(handler)

    async def send_event(self, event_type: str, event_data: Dict):
        if self.queue.full():
            logging.warning("Event bus queue is full, dropping event")
            return
        await self.queue.put((event_type, event_data))
        self.cursor.execute("INSERT INTO events (event_type, event_data) VALUES (?, ?)", (event_type, str(event_data)))
        self.conn.commit()

    async def process_events(self):
        while True:
            event_type, event_data = await self.queue.get()
            if event_type in self.handlers:
                for handler in self.handlers[event_type]:
                    try:
                        await handler(event_data)
                    except Exception as e:
                        logging.error(f"Error processing event {event_type}: {e}")
                        self.dead_letter_queue.append((event_type, event_data))
            else:
                logging.warning(f"No handler found for event {event_type}")
            self.queue.task_done()

    async def start(self):
        asyncio.create_task(self.process_events())

"""
Memory Store class with improvements from Run 1.

Changes:
- Emit MEMORY_STORE/MEMORY_QUERY events through the event bus
- Added memory decay (reduce relevance over time)
- Batch embedding API calls
- Connection pooling for sqlite
"""
import asyncio
from typing import Dict
import sqlite3
from pydantic import BaseModel
import os
from langchain import LLMChain
from event_bus import EVENT_TYPES, EventBus

class MemoryStore:
    def __init__(self, event_bus: EventBus):
        self.event_bus = event_bus
        self.conn = sqlite3.connect("memory_store.db")
        self.cursor = self.conn.cursor()
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS memories (
                id INTEGER PRIMARY KEY,
                memory TEXT,
                relevance REAL
            )
        """)
        self.conn.commit()

    async def store_memory(self, memory: str):
        self.cursor.execute("INSERT INTO memories (memory, relevance) VALUES (?, ?)", (memory, 1.0))
        self.conn.commit()
        await self.event_bus.send_event(EVENT_TYPES.MEMORY_STORE, {"memory": memory})

    async def query_memory(self, query: str):
        self.cursor.execute("SELECT memory FROM memories WHERE memory LIKE ?", (f"%{query}%",))
        results = self.cursor.fetchall()
        await self.event_bus.send_event(EVENT_TYPES.MEMORY_QUERY, {"query": query, "results": [result[0] for result in results]})

    async def decay_memory(self):
        self.cursor.execute("UPDATE memories SET relevance = relevance * 0.9")
        self.conn.commit()

    async def batch_embed(self, memories: list):
        # Use NVIDIA embeddings
        api_key = os.environ.get("NVIDIA_EMBED_API_KEY")

# FILE: orchestrator.py
"""
Orchestrator class with improvements from Run 1.

Changes:
- Actually route tasks through event bus to agents
- Implement the self-improvement loop: after every 10 tasks, analyze success/failure rates and adjust agent selection strategy
- Checkpoint state to sqlite
"""
import asyncio
from typing import Dict
import sqlite3
from pydantic import BaseModel
import os
from event_bus import EVENT_TYPES, EventBus
from agent_base import AgentBase

class Orchestrator:
    def __init__(self, event_bus: EventBus):
        self.event_bus = event_bus
        self.agents = []
        self.task_count = 0
        self.success_count = 0
        self.failure_count = 0
        self.conn = sqlite3.connect("orchestrator.db")
        self.cursor = self.conn.cursor()
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS tasks (
                id INTEGER PRIMARY KEY,
                task TEXT,
                result TEXT
            )
        """)
        self.conn.commit()

    async def add_agent(self, agent: AgentBase):
        self.agents.append(agent)

    async def remove_agent(self, agent: AgentBase):
        self.agents.remove(agent)

    async def route_task(self, task: str):
        await self.event_bus.send_event(EVENT_TYPES.TASK_ASSIGNED, {"task": task})
        self.task_count += 1
        if self.task_count % 10 == 0:
            await self.self_improvement_loop()

    async def self_improvement_loop(self):
        # Analyze success/failure rates
        success_rate = self.success_count / self.task_count
        failure_rate = self.failure_count / self.task_count
        logging.info(f"Success rate: {success_rate:.2f}, Failure rate: {failure_rate:.2f}")
        # Adjust agent selection strategy
        self.agents.sort(key=lambda agent: agent.success_rate, reverse=True)
        # Checkpoint state to sqlite
        self.cursor.execute("INSERT INTO tasks (task, result) VALUES (?, ?)", ("checkpoint", f"success_rate={success_rate}, failure_rate={failure_rate}"))
        self.conn.commit()

    async def start(self):
        asyncio.create_task(self.event_bus.start())
