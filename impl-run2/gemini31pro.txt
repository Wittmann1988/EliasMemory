```python
# FILE: event_bus.py
import asyncio
import json
import sqlite3
import time
from enum import Enum
from typing import Callable, Dict, List, Any, Optional
from pydantic import BaseModel, Field

class EventType(str, Enum):
    """Run 2 Improvements: Centralized event schema to fix Run 1 inconsistencies."""
    TASK_REQUEST = "TASK_REQUEST"
    TASK_COMPLETE = "TASK_COMPLETE"
    TASK_FAILED = "TASK_FAILED"
    MEMORY_STORE = "MEMORY_STORE"
    MEMORY_QUERY = "MEMORY_QUERY"
    MEMORY_RESULT = "MEMORY_RESULT"
    EVALUATION_REQUEST = "EVALUATION_REQUEST"
    EVALUATION_RESULT = "EVALUATION_RESULT"
    METRICS_UPDATE = "METRICS_UPDATE"

class Event(BaseModel):
    """Run 2 Improvements: Pydantic model for strict event typing."""
    id: str = Field(default_factory=lambda: str(time.time()))
    type: EventType
    source: str
    target: Optional[str] = None
    payload: Dict[str, Any]
    timestamp: float = Field(default_factory=time.time)

class EventBus:
    """
    Run 2 Improvements: 
    - Added SQLite event persistence (log).
    - Added Dead-Letter Queue (DLQ) for failed handlers.
    - Added backpressure via bounded asyncio.Queue(maxsize).
    - Added metrics tracking (events/sec, latency).
    - Added timeouts and error handling.
    """
    def __init__(self, db_path: str = "event_bus.db", max_size: int = 1000):
        self.queue = asyncio.Queue(maxsize=max_size)
        self.subscribers: Dict[EventType, List[Callable]] = {e: [] for e in EventType}
        self.db_path = db_path
        self._init_db()
        
        # Metrics
        self.processed_count = 0
        self.start_time = time.time()
        self.total_latency = 0.0

    def _init_db(self):
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS event_log (
                    id TEXT PRIMARY KEY, type TEXT, source TEXT, 
                    payload TEXT, timestamp REAL
                )
            """)
            conn.execute("""
                CREATE TABLE IF NOT EXISTS dlq (
                    event_id TEXT, error TEXT, timestamp REAL
                )
            """)

    def subscribe(self, event_type: EventType, callback: Callable):
        self.subscribers[event_type].append(callback)

    async def publish(self, event: Event, timeout: float = 2.0):
        try:
            # Backpressure: wait_for enforces a timeout if queue is full
            await asyncio.wait_for(self.queue.put(event), timeout=timeout)
            
            # Persist event
            def _save():
                with sqlite3.connect(self.db_path) as conn:
                    conn.execute(
                        "INSERT INTO event_log VALUES (?, ?, ?, ?, ?)",
                        (event.id, event.type.value, event.source, json.dumps(event.payload), event.timestamp)
                    )
            await asyncio.to_thread(_save)
        except asyncio.TimeoutError:
            print(f"[EventBus] ERROR: Queue full, dropped event {event.id}")

    async def start(self):
        self._worker_task = asyncio.create_task(self._worker())

    async def stop(self):
        self._worker_task.cancel()

    async def _worker(self):
        while True:
            try:
                event = await self.queue.get()
                process_start = time.time()
                
                # Calculate queue latency
                latency = process_start - event.timestamp
                self.total_latency += latency

                handlers = self.subscribers.get(event.type, [])
                for handler in handlers:
                    try:
                        # Timeout for individual handlers
                        await asyncio.wait_for(handler(event), timeout=5.0)
                    except Exception as e:
                        print(f"[EventBus] Handler error for {event.type}: {e}")
                        self._send_to_dlq(event, str(e))

                self.processed_count += 1
                self.queue.task_done()
            except asyncio.CancelledError:
                break
            except Exception as e:
                print(f"[EventBus] Critical worker error: {e}")

    def _send_to_dlq(self, event: Event, error_msg: str):
        def _save_dlq():
            with sqlite3.connect(self.db_path) as conn:
                conn.execute(
                    "INSERT INTO dlq VALUES (?, ?, ?)",
                    (event.id, error_msg, time.time())
                )
        asyncio.create_task(asyncio.to_thread(_save_dlq))

    def get_metrics(self) -> Dict[str, float]:
        elapsed = time.time() - self.start_time
        avg_latency = (self.total_latency / self.processed_count) if self.processed_count > 0 else 0
        events_per_sec = self.processed_count / elapsed if elapsed > 0 else 0
        return {
            "events_per_sec": round(events_per_sec, 2),
            "avg_latency_sec": round(avg_latency, 4),
            "queue_size": self.queue.qsize()
        }
```

```python
                metadata TEXT,
                timestamp REAL,
                embedding TEXT -- Stored as JSON string for the mock
            )

        conn.close()

        }

                batch = texts[i:i+batch_size]
                payload = {"input": batch, "model": self.model, "input_type": "query"}

                        "https://integrate.api.nvidia.com/v1/embeddings",

                        all_embeddings.extend(embeddings)

                    print(f"[MemoryStore] Embedding API Error: {e}")
                    # Fallback on error
                    all_embeddings.extend([[0.0] * 2048 for _ in batch])
                    
        return all_embeddings

    async def store(self, text: str, metadata: Dict[str, Any]):

        conn = await self.pool.get()

        finally:
            await self.pool.put(conn)

        await self.event_bus.publish(Event(
            type=EventType.MEMORY_STORE,
            source="MemoryStore",
            payload={"text_preview": text[:50], "status": "stored"}
        ))

    async def query(self, text: str, top_k: int = 3) -> List[Dict]:
        await self.event_bus.publish(Event(
            type=EventType.MEMORY_QUERY,
            source="MemoryStore",
            payload={"query": text}
        ))
        
        embeddings = await self._get_embeddings([text])
        
        conn = await self.pool.get()
        results = []
        try:
            def _search():
                # Mocking vec_search and decay
                cur = conn.execute("SELECT id, text, timestamp FROM memories ORDER BY id DESC LIMIT ?", (top_k*2,))
                return cur.fetchall()
            rows = await asyncio.to_thread(_search)
            
            current_time = time.time()
            decay_rate = 0.0001 # Decay over time
            
            for row in rows:
                mem_id, mem_text, timestamp = row
                age = current_time - timestamp
                # Memory Decay logic: reduce relevance exponentially based on age
                base_score = 0.9 # Mock similarity
                decayed_score = base_score * math.exp(-decay_rate * age)

            # Sort by decayed score
            results = sorted(results, key=lambda x: x["score"], reverse=True)[:top_k]
            
        finally:
            await self.pool.put(conn)
            
        await self.event_bus.publish(Event(
            type=EventType.MEMORY_RESULT,
            source="MemoryStore",
            payload={"results_count": len(results)}
        ))
        return results
```

```python
    - Report results back via TASK_COMPLETE/TASK_FAILED events.

        self.capabilities = capabilities
        self.event_bus = event_bus
        self.memory_store = memory_store

        self.tasks_completed = 0
        self.tasks_failed = 0
        
        self.event_bus.subscribe(EventType.TASK_REQUEST, self._handle_task_request)

    async def _handle_task_request(self, event: Event):
        # Only process if targeted at this agent
        if event.target != self.name:
            return
            
        task_data = event.payload
        try:
            result = await self.execute(task_data)
            self.tasks_completed += 1
            
            await self.event_bus.publish(Event(
                type=EventType.TASK_COMPLETE,
                source=self.name,
                payload={"task_id": task_data["id"], "result": result}
            ))
            
            # Request evaluation from Reviewer
            await self.event_bus.publish(Event(
                type=EventType.EVALUATION_REQUEST,
                source=self.name,
                target="Reviewer",

            ))
            
        except Exception as e:
            self.tasks_failed += 1
            await self.event_bus.publish(Event(
                type=EventType.TASK_FAILED,
```

# FILE: orchestrator.py
import asyncio
import sqlite3
import json
from typing import Dict, List
from event_bus import EventBus, Event, EventType
from memory_store import MemoryStore

class Orchestrator:
    """
    Run 2 Improvements:
    - ACTUALLY routes tasks through event bus to agents.
    - Implemented Self-Improvement Loop: Analyzes success/failure every 10 tasks.
    - Checkpoints routing weights and state to SQLite.
    """
    def __init__(self, event_bus: EventBus, memory_store: MemoryStore, agent_names: List[str], db_path: str = "orchestrator.db"):
        self.event_bus = event_bus
        self.memory_store = memory_store
        self.db_path = db_path
        self.agent_names = [name for name in agent_names if name != "Reviewer"]
        
        # State & Routing Weights
        self.task_count = 0
        self.agent_stats = {name: {"success": 0, "fail": 0, "weight": 1.0} for name in self.agent_names}
        self.active_tasks = set()
        
        self._init_db()
        self._load_checkpoint()

        # Subscriptions
        self.event_bus.subscribe(EventType.TASK_COMPLETE, self._handle_success)
        self.event_bus.subscribe(EventType.TASK_FAILED, self._handle_failure)
        self.event_bus.subscribe(EventType.EVALUATION_RESULT, self._handle_eval)

    def _init_db(self):
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS routing_weights (
                    agent TEXT PRIMARY KEY, success INT, fail INT, weight REAL
                )
            """)

    def _load_checkpoint(self):
        with sqlite3.connect(self.db_path) as conn:
            cur = conn.execute("SELECT agent, success, fail, weight FROM routing_weights")
            for row in cur.fetchall():
                agent, s, f, w = row
                if agent in self.agent_stats:
                    self.agent_stats[agent] = {"success": s, "fail": f, "weight": w}

    def _checkpoint_state(self):
        with sqlite3.connect(self.db_path) as conn:
            for agent, stats in self.agent_stats.items():
                conn.execute("""
                    INSERT INTO routing_weights (agent, success, fail, weight) 
                    VALUES (?, ?, ?, ?)
                    ON CONFLICT(agent) DO UPDATE SET 
                    success=excluded.success, fail=excluded.fail, weight=excluded.weight
                """, (agent, stats["success"], stats["fail"], stats["weight"]))

    def _select_best_agent(self) -> str:
        # Select agent with highest weight
        return max(self.agent_stats.items(), key=lambda x: x[1]["weight"])[0]

    async def submit_task(self, task_id: str, description: str):
        target_agent = self._select_best_agent()
        self.active_tasks.add(task_id)
        
        print(f"[Orchestrator] Routing Task {task_id} to {target_agent} (Weight: {self.agent_stats[target_agent]['weight']:.2f})")
        
        await self.event_bus.publish(Event(
            type=EventType.TASK_REQUEST,
            source="Orchestrator",
            target=target_agent,
            payload={"id": task_id, "description": description}
        ))

    async def _handle_success(self, event: Event):
        agent = event.source
        if agent in self.agent_stats:
            self.agent_stats[agent]["success"] += 1
        self._mark_task_done(event.payload.get("task_id"))

    async def _handle_failure(self, event: Event):
        agent = event.source
        if agent in self.agent_stats:
            self.agent_stats[agent]["fail"] += 1
        self._mark_task_done(event.payload.get("task_id"))

    def _mark_task_done(self, task_id: str):
        if task_id in self.active_tasks:
            self.active_tasks.remove(task_id)
            self.task_count += 1
            self._check_self_improvement()

    async def _handle_eval(self, event: Event):
        # Further adjust weights based on reviewer score
        agent = event.payload["agent"]
        score = event.payload["score"]
        if agent in self.agent_stats:
            # Minor adjustment based on quality
            self.agent_stats[agent]["weight"] += (score - 0.5
