            try:
                if asyncio.iscoroutinefunction(handler):
                    await asyncio.wait_for(handler(event), timeout=timeout)
                else:
                    handler(event)

@dataclass
class Task:
    """
    CHANGED FROM RUN 1: Complete task schema with tracking.
    Run 1 had incomplete task definitions.

    description: str
    created_at: datetime = field(default_factory=datetime.utcnow)

    completed_at: Optional[datetime] = None

    max_retries: int = 3

            'task_id': self.task_id,

            'status': self.status,
            'result': self.result,
            'error': self.error,
            'started_at': self.started_at.isoformat() if self.started_at else None,
            'completed_at': self.completed_at.isoformat() if self.completed_at else None,
            'retry_count': self.retry_count,
            'max_retries': self.max_retries

    - Proper async/await with timeouts and error handling
    """
    
    def __init__(self, db_path: str = "orchestrator.db", improvement_interval: int = 10):
        self.db_path = db_path
        self.improvement_interval = improvement_interval
        self._event_bus = get_event_bus()
        self._memory_store = get_memory_store()
        self._agent_registry = get_agent_registry()
        self._tasks: Dict[str, Task] = {}
        self._task_queue: asyncio.Queue = asyncio.Queue()
        self._running = False
        self._tasks_completed = 0
        self._tasks_failed = 0
        self._improvement_logs: List[ImprovementLog] = []
        self._agent_weights: Dict[str, float] = {}
        self._init_db()
        self._load_state()
        self._setup_event_handlers()
    
    def _init_db(self):
        """Initialize sqlite database for state persistence."""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS tasks (
                task_id TEXT PRIMARY KEY,
                task_json TEXT,
                status TEXT,
                created_at TEXT
            )
        ''')
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS improvements (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                timestamp TEXT,
                log_json TEXT
            )
        ''')
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS agent_weights (
                agent_id TEXT PRIMARY KEY,
                weight REAL,
                updated_at TEXT
            )
        ''')
        conn.commit()
        conn.close()
    
    def _load_state(self):
        """Load state from database on startup."""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # Load tasks
        cursor.execute('SELECT task_json FROM tasks WHERE status IN (?, ?)', ('pending', 'running'))
        for row in cursor.fetchall():
            task_data = json.loads(row[0])
            self._tasks[task_data['task_id']] = Task(**task_data)
        
        # Load agent weights
        cursor.execute('SELECT agent_id, weight FROM agent_weights')
        for row in cursor.fetchall():
            self._agent_weights[row[0]] = row[1]
        
        conn.close()
        print(f"[ORCHESTRATOR] Loaded {len(self._tasks)} pending tasks")
    
    def _save_state(self):
        """Save state to database."""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # Save tasks
        for task in self._tasks.values():
            cursor.execute('''
                INSERT OR REPLACE INTO tasks (task_id, task_json, status, created_at)
                VALUES (?, ?, ?, ?)
            ''', (
                task.task_id,
                json.dumps(task.to_dict()),
                task.status,
                task.created_at.isoformat()
            ))
        
        # Save agent weights
        now = datetime.utcnow().isoformat()
        for agent_id, weight in self._agent_weights.items():
            cursor.execute('''
                INSERT OR REPLACE INTO agent_weights (agent_id, weight, updated_at)
                VALUES (?, ?, ?)
            ''', (agent_id, weight, now))
        
        conn.commit()
        conn.close()
    
    def _setup_event_handlers(self):
        """
        CHANGED FROM RUN 1: Subscribe to all relevant events through event bus.
        Run 1 had no event subscription in orchestrator.
        """
        self._event_bus.subscribe(EventType.TASK_COMPLETED, self._on_task_completed)
        self._event_bus.subscribe(EventType.TASK_FAILED, self._on_task_failed)
        self._event_bus.subscribe(EventType.AGENT_METRICS, self._on_agent_metrics)
    
    async def _on_task_completed(self, event: Event):
        """Handle task completion event."""
        task_id = event.payload.get('task_id')
        if task_id and task_id in self._tasks:
            task = self._tasks[task_id]
            task.status = 'completed'
            task.completed_at = datetime.utcnow()
            task.result = event.payload.get('result')
            self._tasks_completed += 1
            
            # Store result in memory
            await self._memory_store.store(
                content=f"Task completed: {task.description}",
                metadata={'task_id': task_id, 'result': str(task.result)},
                memory_id=f"task_result_{task_id}"
            )
            
            # Check for improvement cycle
            await self._check_improvement_cycle()
            
            self._save_state()
    
    async def _on_task_failed(self, event: Event):
        """Handle task failure event."""
        task_id = event.payload.get('task_id')
        if task_id and task_id in self._tasks:
            task = self._tasks[task_id]
            task.status = 'failed'
            task.error = event.payload.get('error')
            task.completed_at = datetime.utcnow()
            self._tasks_failed += 1

                task.retry_count += 1
                task.status = 'pending'
                task.assigned_agent = None
                await self._task_queue.put(task)
            else:
                self._save_state()
