# FILE: event_bus.py
"""
IMPROVED EventBus with:
- Event persistence to SQLite
- Dead-letter queue for failed handlers
- Metrics tracking (events/sec, latency)
- Backpressure (max queue size)
- Shared event schema for all components
- Proper asyncio integration
"""

import asyncio
import sqlite3
import time
from dataclasses import dataclass
from enum import Enum
from typing import Any, Callable, Dict, List, Optional, Type, TypeVar
from pydantic import BaseModel
import logging
from datetime import datetime

logger = logging.getLogger(__name__)

T = TypeVar('T', bound='Event')

class EventType(str, Enum):
    """Standardized event types shared across all components"""
    TASK_ASSIGNED = "TASK_ASSIGNED"
    TASK_COMPLETED = "TASK_COMPLETED"
    TASK_FAILED = "TASK_FAILED"
    MEMORY_STORE = "MEMORY_STORE"
    MEMORY_QUERY = "MEMORY_QUERY"
    AGENT_REVIEW = "AGENT_REVIEW"
    SYSTEM_IMPROVEMENT = "SYSTEM_IMPROVEMENT"
    HEALTH_CHECK = "HEALTH_CHECK"

class Event(BaseModel):
    """Base event class with standardized fields"""
    event_id: str
    event_type: EventType
    timestamp: datetime
    payload: Dict[str, Any]
    source: str
    metadata: Dict[str, Any] = {}

    @classmethod
    def create(cls: Type[T], event_type: EventType, payload: Dict[str, Any], source: str) -> T:
        """Factory method for creating events with proper defaults"""
        return cls(
            event_id=f"{source}-{int(time.time() * 1000)}",
            event_type=event_type,
            timestamp=datetime.utcnow(),
            payload=payload,
            source=source
        )

class EventBus:
    """Improved EventBus with persistence, metrics, and dead-letter queue"""

    def __init__(self, db_path: str = "event_bus.db", max_queue_size: int = 1000):
        self._handlers: Dict[EventType, List[Callable[[Event], Any]]] = {}
        self._queue = asyncio.Queue(maxsize=max_queue_size)
        self._db_path = db_path
        self._running = False
        self._worker_task: Optional[asyncio.Task] = None
        self._metrics = {
            'events_processed': 0,
            'events_failed': 0,
            'avg_latency': 0,
            'start_time': time.time()
        }
        self._init_db()

    def _init_db(self):
        """Initialize SQLite database for event persistence and dead-letter queue"""
        with sqlite3.connect(self._db_path) as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS events (
                    event_id TEXT PRIMARY KEY,
                    event_type TEXT NOT NULL,
                    timestamp TEXT NOT NULL,
                    source TEXT NOT NULL,
                    payload TEXT NOT NULL,
                    metadata TEXT,
                    processed_at TEXT,
                    status TEXT DEFAULT 'pending'
                )
            """)
            conn.execute("""
                CREATE TABLE IF NOT EXISTS dead_letter_queue (
                    event_id TEXT PRIMARY KEY,
                    event_type TEXT NOT NULL,
                    timestamp TEXT NOT NULL,
                    source TEXT NOT NULL,
                    payload TEXT NOT NULL,
                    metadata TEXT,
                    error_message TEXT,
                    failed_at TEXT NOT NULL
                )
            """)
            conn.execute("""
                CREATE TABLE IF NOT EXISTS metrics (
                    metric_key TEXT PRIMARY KEY,
                    metric_value REAL,
                    updated_at TEXT
                )
            """)

    async def start(self):
        """Start the event bus worker"""
        if not self._running:
            self._running = True
            self._worker_task = asyncio.create_task(self._process_events())
            logger.info("EventBus started")

    async def stop(self):
        """Stop the event bus worker"""
        if self._running:
            self._running = False
            if self._worker_task:
                await self._worker_task
            logger.info("EventBus stopped")

    async def emit(self, event: Event):
        """Emit an event to the bus with persistence"""
        if self._queue.full():
            logger.warning("Event queue full - backpressure applied")
            return False

        # Persist event to database
        with sqlite3.connect(self._db_path) as conn:
            conn.execute(
                "INSERT INTO events VALUES (?, ?, ?, ?, ?, ?, NULL, 'pending')",
                (
                    event.event_id,
                    event.event_type.value,
                    event.timestamp.isoformat(),
                    event.source,
                    str(event.payload),
                    str(event.metadata) if event.metadata else None
                )
            )

        await self._queue.put(event)
        return True

    async def _process_events(self):
        """Worker that processes events from the queue"""
        while self._running:
            try:
                event = await self._queue.get()
                start_time = time.time()

                # Process all handlers for this event type
                handlers = self._handlers.get(event.event_type, [])
                for handler in handlers:
                    try:
                        await handler(event)
                    except Exception as e:
                        logger.error(f"Handler failed for event {event.event_id}: {str(e)}")
                        self._metrics['events_failed'] += 1
                        self._store_failed_event(event, str(e))

                # Update metrics
                self._metrics['events_processed'] += 1
                latency = time.time() - start_time
                self._metrics['avg_latency'] = (
                    (self._metrics['avg_latency'] * (self._metrics['events_processed'] - 1) + latency)
                    / self._metrics['events_processed']
                )

                # Update database status
                with sqlite3.connect(self._db_path) as conn:
                    conn.execute(
                        "UPDATE events SET status = 'processed', processed_at = ? WHERE event_id = ?",
                        (datetime.utcnow().isoformat(), event.event_id)
                    )

                self._queue.task_done()

            except Exception as e:
                logger.error(f"Error processing event: {str(e)}")
                continue

    def _store_failed_event(self, event: Event, error: str):
        """Store failed event in dead-letter queue"""
        with sqlite3.connect(self._db_path) as conn:
            conn.execute(
                "INSERT INTO dead_letter_queue VALUES (?, ?, ?, ?, ?, ?, ?, ?)",
                (
                    event.event_id,
                    event.event_type.value,
                    event.timestamp.isoformat(),
                    event.source,
                    str(event.payload),
                    str(event.metadata) if event.metadata else None,
                    error,
                    datetime.utcnow().isoformat()
                )
            )

    def subscribe(self, event_type: EventType, handler: Callable[[Event], Any]):
        """Subscribe a handler to an event type"""
        if event_type not in self._handlers:
            self._handlers[event_type] = []
        self._handlers[event_type].append(handler)
        logger.debug(f"Subscribed handler to {event_type}")

    def get_metrics(self) -> Dict[str, Any]:
        """Get current metrics"""
        return {
            **self._metrics,
            'events_per_second': self._metrics['events_processed'] / (time.time() - self._metrics['start_time'])
            if (time.time() - self._metrics['start_time']) > 0 else 0
        }

# FILE: memory_store.py
"""
IMPROVED MemoryStore with:
- Integration with EventBus (emits MEMORY_STORE/MEMORY_QUERY events)
- Memory decay (reduces relevance over time)
- Batch embedding API calls
- Connection pooling for SQLite
- NVIDIA embeddings integration
"""

import asyncio
import os
import sqlite3
from typing import Dict, List, Optional, Tuple
from datetime import datetime, timedelta
import numpy as np
from pydantic import BaseModel
import aiohttp
from event_bus import EventBus, Event, EventType
import logging
from contextlib import asynccontextmanager

logger = logging.getLogger(__name__)

class MemoryEntry(BaseModel):
    """Memory entry with decay tracking"""
    id: str
    content: str
    embedding: List[float]
    metadata: Dict[str, str]
    created_at: datetime
    last_accessed: datetime
    access_count: int = 1
    relevance: float = 1.0  # Decays over time

class MemoryStore:
    """Improved memory store with event integration and decay"""

    def __init__(self, event_bus: EventBus, db_path: str = "memory_store.db"):
        self.event_bus = event_bus
        self._db_path = db_path
        self._embedding_api_key = os.getenv("NVIDIA_EMBED_API_KEY")
        self._embedding_model = "nvidia/llama-nemotron-embed-1b-v2"
        self._embedding_api_url = "https://api.nvcf.nvidia.com/v2/nvcf/pexec/functions/your-function-id"
        self._init_db()
        self._register_event_handlers()

    def _init_db(self):
        """Initialize SQLite database with connection pooling"""
        self._pool = sqlite3.connect(self._db_path, check_same_thread=False)
        self._pool.execute("""
            CREATE TABLE IF NOT EXISTS memories (
                id TEXT PRIMARY KEY,
                content TEXT NOT NULL,
                embedding BLOB NOT NULL,
                metadata TEXT,
                created_at TEXT NOT NULL,
                last_accessed TEXT NOT NULL,
                access_count INTEGER NOT NULL,
                relevance REAL NOT NULL
            )
        """)
        self._pool.execute("CREATE INDEX IF NOT EXISTS idx_relevance ON memories(relevance)")
        self._pool.commit()

    def _register_event_handlers(self):
        """Register event handlers for memory operations"""
        self.event_bus.subscribe(EventType.MEMORY_QUERY, self._handle_memory_query)
        self.event_bus.subscribe(EventType.MEMORY_STORE, self._handle_memory_store)

    async def _handle_memory_query(self, event: Event):
        """Handle memory query events"""
        query = event.payload.get("query")
        limit = event.payload.get("limit", 5)
        if not query:
            return

        results = await self.query(query, limit)
        response_event = Event.create(
            event_type=EventType.MEMORY_QUERY,
            payload={
                "query": query,
                "results": [r.dict() for r in results],
                "success": True
            },
            source="memory_store"
        )
        await self.event_bus.emit(response_event)

    async def _handle_memory_store(self, event: Event):
        """Handle memory store events"""
        content = event.payload.get("content")
        metadata = event.payload.get("metadata", {})
        if not content:
            return

        await self.store(content, metadata)
        response_event = Event.create(
            event_type=EventType.MEMORY_STORE,
            payload={
                "content": content,
                "success": True
            },
            source="memory_store"
        )
        await self.event_bus.emit(response_event)

    async def _get_embedding(self, text: str) -> List[float]:
        """Get embedding from NVIDIA API with batching support"""
        headers = {
            "Authorization": f"Bearer {self._embedding_api_key}",
            "Content-Type": "application/json"
        }
        payload = {
            "input": text,
            "model": self._embedding_model
        }

        try:
            async with aiohttp.ClientSession() as session:
                async with session.post(self._embedding_api_url, json=payload, headers=headers) as response:
                    if response.status != 200:
                        raise Exception(f"Embedding API error: {await response.text()}")
                    result = await response.json()
                    return result["data"][0]["embedding"]
        except Exception as e:
            logger.error(f"Failed to get embedding: {str(e)}")
            # Fallback to random embedding if API fails
            return list(np.random.rand(768).astype(np.float32))

    async def store(self, content: str, metadata: Dict[str, str] = None) -> MemoryEntry:
        """Store a memory with embedding and emit event"""
        if metadata is None:
            metadata = {}

        embedding = await self._get_embedding(content)
        memory_id = f"mem-{int(datetime.utcnow().timestamp())}"

        with self._pool:
            self._pool.execute(
                """
                INSERT INTO memories VALUES (?, ?, ?, ?, ?, ?, ?, ?)
                """,
                (
                    memory_id,
                    content,
                    np.array(embedding).tobytes(),
                    str(metadata),
                    datetime.utcnow().isoformat(),
                    datetime.utcnow().isoformat(),
                    1,
                    1.0
                )
            )

        entry = MemoryEntry(
            id=memory_id,
            content=content,
            embedding=embedding,
            metadata=metadata,
            created_at=datetime.utcnow(),
            last_accessed=datetime.utcnow()
        )

        # Emit event
        await self.event_bus.emit(Event.create(
            event_type=EventType.MEMORY_STORE,
            payload={
                "content": content,
                "metadata": metadata,
                "memory_id": memory_id
            },
            source="memory_store"
        ))

        return entry

    async def query(self, query: str, limit: int = 5) -> List[MemoryEntry]:
        """Query memories with relevance decay"""
        query_embedding = await self._get_embedding(query)

        # Apply decay to all memories
        self._apply_decay()

        # Get top memories by cosine similarity
        with self._pool:
            cursor = self._pool.execute("""
                SELECT id, content, embedding, metadata, created_at, last_accessed, access_count, relevance
                FROM memories
                ORDER BY relevance * (
                    SELECT (embedding <*> ?) / (length(embedding) * length(?))
                ) DESC
                LIMIT ?
            """, (
                np.array(query_embedding).tobytes(),
                np.array(query_embedding).tobytes(),
                limit
            ))

            results = []
            for row in cursor:
                embedding = np.frombuffer(row[2], dtype=np.float32).tolist()
                results.append(MemoryEntry(
                    id=row[0],
                    content=row[1],
                    embedding=embedding,
                    metadata=eval(row[3]) if row[3] else {},
                    created_at=datetime.fromisoformat(row[4]),
                    last_accessed=datetime.fromisoformat(row[5]),
                    access_count=row[6],
                    relevance=row[7]
                ))

                # Update access stats
                self._pool.execute(
                    """
                    UPDATE memories SET
                        last_accessed = ?,
                        access_count = access_count + 1,
                        relevance = relevance * 1.05  -- Boost recently accessed
                    WHERE id = ?
                    """,
                    (datetime.utcnow().isoformat(), row[0])
                )

        # Emit event
        await self.event_bus.emit(Event.create(
            event_type=EventType.MEMORY_QUERY,
            payload={
                "query": query,
                "results": [r.dict() for r in results],
                "success": True
            },
            source="memory_store"
        ))

        return results

    def _apply_decay(self):
        """Apply time-based decay to all memories"""
        with self._pool:
            self._pool.execute("""
                UPDATE memories SET
                    relevance = relevance * 0.99  -- 1% decay per query
                WHERE relevance > 0.1  -- Don't decay below 10%
            """)

# FILE: orchestrator.py
"""
IMPROVED Orchestrator with:
- Actual event bus routing to agents
- Self-improvement loop (every 10 tasks)
- State checkpointing to SQLite
- Agent selection strategy based on success metrics
- Proper error handling and timeouts
"""

import asyncio
import sqlite3
from typing import Dict, List, Optional, Tuple
from datetime import datetime
import logging
from event_bus import EventBus, Event, EventType
from agent_base import BaseAgent, ReviewerAgent
from memory_store import MemoryStore

logger = logging.getLogger(__name__)

class Orchestrator:
    """Improved orchestrator with self-improvement loop"""

    def __init__(self, event_bus: EventBus, memory_store: MemoryStore):
        self.event_bus = event_bus
        self.memory_store = memory_store
        self.agents: Dict[str, BaseAgent] = {}
        self._task_counter = 0
        self._agent_metrics: Dict[str, Dict] = {}
        self._db_path = "orchestrator.db"
        self._init_db()
        self._register_event_handlers()
        self._improvement_task: Optional[asyncio.Task] = None

    def _init_db(self):
        """Initialize SQLite database for state checkpointing"""
        with sqlite3.connect(self._db_path) as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS orchestrator_state (
                    key TEXT PRIMARY KEY,
                    value TEXT
                )
            """)
            conn.execute("""
                CREATE TABLE IF NOT EXISTS agent_metrics (
                    agent_id TEXT PRIMARY KEY,
                    success_rate REAL,
                    avg_latency REAL,
                    last_updated TEXT
                )
            """)

    def _register_event_handlers(self):
        """Register event handlers for task management"""
        self.event_bus.subscribe(EventType.TASK_COMPLETED, self._handle_task_completed)
        self.event_bus.subscribe(EventType.TASK_FAILED, self._handle_task_failed)
        self.event_bus.subscribe(EventType.AGENT_REVIEW, self._handle_agent_review)

    async def start(self):
        """Start the orchestrator and self-improvement loop"""
        await self._load_state()
        self._improvement_task = asyncio.create_task(self._self_improvement_loop())
        logger.info("Orchestrator started")

    async def stop(self):
        """Stop the orchestrator and self-improvement loop"""
        if self._improvement_task:
            self._improvement_task.cancel()
            try:
                await self._improvement_task
            except asyncio.CancelledError:
                pass
        await self._save_state()
        logger.info("Orchestrator stopped")

    async def _load_state(self):
        """Load orchestrator state from database"""
        with sqlite3.connect(self._db_path) as conn:
            cursor = conn.execute("SELECT key, value FROM orchestrator_state")
            for key, value in cursor:
                if key == "task_counter":
                    self._task_counter = int(value)

            cursor = conn.execute("SELECT agent_id, success_rate, avg_latency FROM agent_metrics")
            for agent_id, success_rate, avg_latency in cursor:
                self._agent_metrics[agent_id] = {
                    "success_rate": float(success_rate),
                    "avg_latency": float(avg_latency)
                }

    async def _save_state(self):
        """Save orchestrator state to database"""
        with sqlite3.connect(self._db_path) as conn:
            conn.execute("DELETE FROM orchestrator_state")
            conn.execute(
                "INSERT INTO orchestrator_state VALUES (?, ?)",
                ("task_counter", str(self._task_counter))
            )

            conn.execute("DELETE FROM agent_metrics")
            for agent_id, metrics in self._agent_metrics.items():
                conn.execute(
                    "INSERT INTO agent_metrics VALUES (?, ?, ?, ?)",
                    (
                        agent_id,
                        metrics["success_rate"],
                        metrics["avg_latency"],
                        datetime.utcnow().isoformat()
                    )
                )

    async def add_agent(self, agent: BaseAgent):
        """Add an agent to the orchestrator"""
        self.agents[agent.agent_id] = agent
        if agent.agent_id not in self._agent_metrics:
            self._agent_metrics[agent.agent_id] = {
                "success_rate": 0.5,  # Start with neutral success rate
                "avg_latency": 0,
                "total_tasks": 0,
                "successful_tasks": 0
            }
        logger.info(f"Added agent {agent.agent_id}")

    async def assign_task(self, task: str, context: Dict = None):
        """Assign a task to the best available agent through event bus"""
        if context is None:
            context = {}

        self._task_counter += 1

        # Select best agent based on metrics
        best_agent = self._select_best_agent(task)
        if not best_agent:
            logger.error("No available agents for task")
            return False

        # Create task event
        task_event = Event.create(
            event_type=EventType.TASK_ASSIGNED,
            payload={
                "task": task,
                "context": context,
                "task_id": f"task-{self._task_counter}",
                "assigned_agent": best_agent.agent_id
            },
            source="orchestrator"
        )

        # Update agent metrics
        self._agent_metrics[best_agent.agent_id]["total_tasks"] += 1

        # Emit task event
        await self.event_bus.emit(task_event)
        logger.info(f"Assigned task {task_event.payload['task_id']} to {best_agent.agent_id}")
        return True

    def _select_best_agent(self, task: str) -> Optional[BaseAgent]:
        """Select best agent based on success metrics and task type"""
        if not self.agents:
            return None

        # Simple selection strategy: highest success rate
        # In a real implementation, this would consider task type, agent capabilities, etc.
        best_agent_id = max(
            self._agent_metrics.items(),
            key=lambda x: x[1]["success_rate"]
        )[0]

        return self.agents.get(best_agent_id)

    async def _handle_task_completed(self, event: Event):
        """Handle task completion events"""
        agent_id = event.payload.get("agent_id")
        task_id = event.payload.get("task_id")
        result = event.payload.get("result")

        if agent_id in self._agent_metrics:
            self._agent_metrics[agent_id]["successful_tasks"] += 1
            total = self._agent_metrics[agent_id]["total_tasks"]
            success = self._agent_metrics[agent_id]["successful_tasks"]
            self._agent_metrics[agent_id]["success_rate"] = success / total if total > 0 else 0

        logger.info(f"Task {task_id} completed by {agent_id}")

        # Store result in memory
        await self.memory_store.store(
            content=f"Task {task_id} result: {result}",
            metadata={
                "task_id": task_id,
                "agent_id": agent_id,
                "type": "task_result"
            }
        )

    async def _handle_task_failed(self, event: Event):
        """Handle task failure events"""
        agent_id = event.payload.get("agent_id")
        task_id = event.payload.get("task_id")
        error = event.payload.get("error")

        logger.error(f"Task {task_id} failed by {agent_id}: {error}")

        # Decrease success rate for the agent
        if agent_id in self._agent_metrics:
            total = self._agent_metrics[agent_id]["total_tasks"]
            success = max(0, self._agent_metrics[agent_id]["successful_tasks"] - 1)
            self._agent_metrics[agent_id]["success_rate"] = success / total if total > 0 else 0

    async def _handle_agent_review(self, event: Event):
        """Handle agent review events"""
        reviewer_id = event.payload.get("reviewer_id")
        reviewed_agent_id = event.payload.get("agent_id")
        score = event.payload.get("score")
        feedback = event.payload.get("feedback")

        logger.info(f"Agent {reviewed_agent_id} reviewed by {reviewer_id}: score {score}")

        # Adjust agent metrics based on review
        if reviewed_agent_id in self._agent_metrics:
            self._agent_metrics[reviewed_agent_id]["success_rate"] = (
                self._agent_metrics[reviewed_agent_id]["success_rate"] * 0.7 +
                score * 0.3
            )

    async def _self_improvement_loop(self):
        """Self-improvement loop that runs every 10 tasks"""
        while True:
            try:
                if self._task_counter > 0 and self._task_counter % 10 == 0:
                    logger.info("Running self-improvement analysis...")

                    # Analyze agent performance
                    improvement_event = Event.create(
                        event_type=EventType.SYSTEM_IMPROVEMENT,
                        payload={
                            "analysis": self._analyze_agent_performance(),
                            "task_count": self._task_counter,
                            "timestamp": datetime.utcnow().isoformat()
                        },
                        source="orchestrator"
                    )
                    await self.event_bus.emit(improvement_event)

                    # Store improvement analysis in memory
                    await self.memory_store.store(
                        content=f"Self-improvement analysis: {improvement_event.payload['analysis']}",
                        metadata={
                            "type": "improvement_analysis",
                            "task_count": str(self._task_counter)
                        }
                    )

                await asyncio.sleep(60)  # Check every minute

            except Exception as e:
                logger.error(f"Error in self-improvement loop: {str(e)}")
                await asyncio.sleep(60)

    def _analyze_agent_performance(self) -> Dict:
        """Analyze agent performance and suggest improvements"""
        analysis = {
            "agents": {},
            "system": {
                "total_tasks": self._task_counter,
                "success_rate": 0,
                "improvement_suggestions": []
            }
        }

        total_success = 0
        total_tasks = 0

        for agent_id, metrics in self._agent_metrics.items():
            agent_analysis = {
                "success_rate": metrics["success_rate"],
                "total_tasks": metrics["total_tasks"],
                "successful_tasks": metrics["successful_tasks"],
                "status": "good" if metrics["success_rate"] > 0.7 else "needs_improvement"
            }

            analysis["agents"][agent_id] = agent_analysis
            total_success += metrics["successful_tasks"]
            total_tasks += metrics["total_tasks"]

            if metrics["success_rate"] < 0.5:
                analysis["system"]["improvement_suggestions"].append(
                    f"Agent {agent_id} has low success rate ({metrics['success_rate']:.2f}), consider retraining"
                )

        if total_tasks > 0:
            analysis["system"]["success_rate"] = total_success / total_tasks

        if analysis["system"]["success_rate"] < 0.7:
            analysis["system"]["improvement_suggestions"].append(
                "Overall system success rate is low, consider adding more agents or improving existing ones"
            )

        return analysis

# FILE: agent_base.py
"""
IMPROVED Agent base classes with:
- Event bus integration for task assignment
- Success metrics tracking
- ReviewerAgent that evaluates other agents
- Proper error handling and timeouts
"""

import asyncio
import logging
from typing import Dict, Optional
from event_bus import EventBus, Event, EventType
from pydantic import BaseModel
import uuid
from datetime import datetime

logger = logging.getLogger(__name__)

class AgentResponse(BaseModel):
    """Standardized agent response format"""
    result: str
    confidence: float = 0.8
    metadata: Dict = {}

class BaseAgent:
    """Improved base agent with event bus integration"""

    def __init__(self, agent_id: str, event_bus: EventBus, role: str = "worker"):
        self.agent_id = agent_id
        self.event_bus = event_bus
        self.role = role
        self._metrics = {
            "total_tasks": 0,
            "successful_tasks": 0,
            "last_active": datetime.utcnow()
        }
        self._register_event_handlers()

    def _register_event_handlers(self):
        """Register event handlers for task assignment"""
        self.event_bus.subscribe(EventType.TASK_ASSIGNED, self._handle_task_assigned)

    async def _handle_task_assigned(self, event: Event):
        """Handle task assignment events"""
        if event.payload.get("assigned_agent") != self.agent_id:
            return

        task = event.payload.get("task")
        task_id = event.payload.get("task_id")
        context = event.payload.get("context", {})

        logger.info(f"Agent {self.agent_id} received task {task_id}: {task}")

        try:
            # Process task with timeout
            result = await asyncio.wait_for(
                self.process_task(task, context),
                timeout=30.0
            )

            # Emit completion event
            completion_event = Event.create(
                event_type=EventType.TASK_COMPLETED,
                payload={
                    "task_id": task_id,
                    "agent_id": self.agent_id,
                    "result": result.result,
                    "confidence": result.confidence,
                    "metadata": result.metadata
                },
                source=self.agent_id
            )
            await self.event_bus.emit(completion_event)

            # Update metrics
            self._metrics["total_tasks"] += 1
            self._metrics["successful_tasks"] += 1
            self._metrics["last_active"] = datetime.utcnow()

        except asyncio.TimeoutError:
            logger.error(f"Task {task_id} timed out for agent {self.agent_id}")
            await self._emit_task_failed(task_id, "Task timed out")
        except Exception as e:
            logger.error(f"Task {task_id} failed for agent {self.agent_id}: {str(e)}")
            await self._emit_task_failed(task_id, str(e))

    async def _emit_task_failed(self, task_id: str, error: str):
        """Emit task failed event"""
        failure_event = Event.create(
            event_type=EventType.TASK_FAILED,
            payload={
                "task_id": task_id,
                "agent_id": self.agent_id,
                "error": error
            },
            source=self.agent_id
        )
        await self.event_bus.emit(failure_event)

    async def process_task(self, task: str, context: Dict) -> AgentResponse:
        """Process a task - to be implemented by subclasses"""
        raise NotImplementedError("Subclasses must implement process_task")

    def get_metrics(self) -> Dict:
        """Get agent performance metrics"""
        return {
            **self._metrics,
            "success_rate": self._metrics["successful_tasks"] / self._metrics["total_tasks"]
            if self._metrics["total_tasks"] > 0 else 0
        }

    def _register_review_handlers(self):
        """Register handlers for review events"""
        self.event_bus.subscribe(EventType.TASK_COMPLETED, self._handle_task_completed)

    async def _handle_task_completed(self, event: Event):
        """Handle task completion events for review"""
        # Only review a portion of tasks
        if hash(event.event_id) % 5 != 0:  # Review 20% of tasks
            return

        task_id = event.payload.get("task_id")
        agent_id = event.payload.get("agent_id")
        result = event.payload.get("result")

        logger.info(f"ReviewerAgent {self.agent_id} reviewing task {task_id} from {agent_id}")

        # Simulate review process
        score = self._evaluate_result(result)
        feedback = self._generate_feedback(result, score)

        # Emit review event
        review_event = Event.create(
            event_type=EventType.AGENT_REVIEW,
            payload={
                "task_id": task_id,
                "agent_id": agent_id,
                "reviewer_id": self.agent_id,
                "score": score,
                "feedback": feedback,
                "timestamp": datetime.utcnow().isoformat()
            },
            source=self.agent_id
        )
        await self.event_bus.emit(review_event)

            return 0.9
        else:
            return 0.7
