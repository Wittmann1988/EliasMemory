# FILE: event_bus.py
import asyncio
import sqlite3
import time
from collections import deque
from typing import Any, Dict, List, Callable, Optional
from pydantic import BaseModel
import os

# Shared event types for all modules
class EventType:
    TASK_ASSIGNED = "TASK_ASSIGNED"
    TASK_COMPLETED = "TASK_COMPLETED"
    MEMORY_STORE = "MEMORY_STORE"
    MEMORY_QUERY = "MEMORY_QUERY"
    AGENT_EVALUATION = "AGENT_EVALUATION"

class Event(BaseModel):
    type: str
    payload: Dict[str, Any]
    timestamp: float = None

    def __init__(self, **data):
        super().__init__(timestamp=time.time(), **data)

class EventBus:
    """
    IMPROVED EventBus from Run 1: Added event persistence to SQLite for replayability, dead-letter queue for failed handlers,
    metrics tracking (events/sec, avg latency), backpressure with max queue size to prevent overload.
    In Run 1, this was just a basic pub/sub; now it's production-ready with error handling and monitoring.
    """
    def __init__(self, db_path: str = "event_bus.db", max_queue_size: int = 1000):
        self.db_path = db_path
        self.max_queue_size = max_queue_size
        self.queue = deque(maxlen=max_queue_size)
        self.subscribers: Dict[str, List[Callable]] = {}
        self.dead_letter_queue: List[Event] = []
        self.metrics = {"events_processed": 0, "total_latency": 0.0, "start_time": time.time()}
        self._init_db()

    def _init_db(self):
        conn = sqlite3.connect(self.db_path)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS events (
                id INTEGER PRIMARY KEY,
                type TEXT,
                payload TEXT,
                timestamp REAL
            )
        """)
        conn.commit()
        conn.close()

    async def publish(self, event: Event):
        if len(self.queue) >= self.max_queue_size:
            self.dead_letter_queue.append(event)
            return
        self.queue.append(event)
        await self._persist_event(event)
        await self._process_event(event)

    async def _persist_event(self, event: Event):
        conn = sqlite3.connect(self.db_path)
        conn.execute("INSERT INTO events (type, payload, timestamp) VALUES (?, ?, ?)",
                     (event.type, str(event.payload), event.timestamp))
        conn.commit()
        conn.close()

    async def _process_event(self, event: Event):
        start = time.time()
        handlers = self.subscribers.get(event.type, [])
        for handler in handlers:
            try:
                await handler(event)
            except Exception as e:
                self.dead_letter_queue.append(event)
        latency = time.time() - start
        self.metrics["events_processed"] += 1
        self.metrics["total_latency"] += latency

    def subscribe(self, event_type: str, handler: Callable):
        if event_type not in self.subscribers:
            self.subscribers[event_type] = []
        self.subscribers[event_type].append(handler)

    def get_metrics(self) -> Dict[str, float]:
        elapsed = time.time() - self.metrics["start_time"]
        return {
            "events_per_sec": self.metrics["events_processed"] / elapsed if elapsed > 0 else 0,
            "avg_latency": self.metrics["total_latency"] / self.metrics["events_processed"] if self.metrics["events_processed"] > 0 else 0
        }

import asyncio
import sqlite3
import sqlite_vec
import time
from typing import List, Dict, Any
from concurrent.futures import ThreadPoolExecutor
from event_bus import EventBus, Event, EventType
import os
from nvidia import embeddings  # Assuming NVIDIA SDK or API client

class MemoryEntry(BaseModel):
    id: str
    content: str
    embedding: List[float]
    relevance: float
    timestamp: float

class MemoryStore:
    """
    IMPROVED MemoryStore from Run 1: Now emits MEMORY_STORE and MEMORY_QUERY events through the event bus for integration.
    Added memory decay (relevance halves every 24 hours), batch embedding API calls for efficiency, connection pooling for SQLite.
    In Run 1, it was isolated; now it's event-driven and self-maintaining.
    """
    def __init__(self, event_bus: EventBus, db_path: str = "memory.db", embed_model: str = "nvidia/llama-nemotron-embed-1b-v2"):
        self.event_bus = event_bus
        self.db_path = db_path
        self.embed_model = embed_model
        self.api_key = os.getenv("NVIDIA_EMBED_API_KEY")
        self.executor = ThreadPoolExecutor(max_workers=4)
        self._init_db()
        asyncio.create_task(self._decay_loop())

    def _init_db(self):
        conn = sqlite3.connect(self.db_path)
        sqlite_vec.init(conn)
        conn.execute("""
            CREATE VIRTUAL TABLE IF NOT EXISTS memories USING vec0(
                id TEXT,
                content TEXT,
                embedding FLOAT[1024],
                relevance REAL,
                timestamp REAL
            )
        """)
        conn.commit()
        conn.close()

    async def store(self, content: str):
        embedding = await self._batch_embed([content])
        entry = MemoryEntry(id=str(time.time()), content=content, embedding=embedding[0], relevance=1.0, timestamp=time.time())
        await asyncio.get_event_loop().run_in_executor(self.executor, self._store_db, entry)
        await self.event_bus.publish(Event(type=EventType.MEMORY_STORE, payload={"entry": entry.dict()}))

    async def query(self, query: str, top_k: int = 5) -> List[Dict[str, Any]]:
        query_emb = await self._batch_embed([query])
        results = await asyncio.get_event_loop().run_in_executor(self.executor, self._query_db, query_emb[0], top_k)
        await self.event_bus.publish(Event(type=EventType.MEMORY_QUERY, payload={"query": query, "results": results}))
        return results

    async def _batch_embed(self, texts: List[str]) -> List[List[float]]:
        # Batch API calls to NVIDIA embeddings
        client = embeddings.Client(api_key=self.api_key)
        response = await client.embed(model=self.embed_model, inputs=texts)
        return response.embeddings

    def _store_db(self, entry: MemoryEntry):
        conn = sqlite3.connect(self.db_path)
        conn.execute("INSERT INTO memories (id, content, embedding, relevance, timestamp) VALUES (?, ?, ?, ?, ?)",
                     (entry.id, entry.content, entry.embedding, entry.relevance, entry.timestamp))
        conn.commit()
        conn.close()

    def _query_db(self, query_emb: List[float], top_k: int) -> List[Dict[str, Any]]:
        conn = sqlite3.connect(self.db_path)
        results = conn.execute("""
            SELECT id, content, relevance, timestamp FROM memories
            ORDER BY vec_distance_L2(embedding, ?) LIMIT ?
        """, (query_emb, top_k)).fetchall()
        conn.close()
        return [{"id": r[0], "content": r[1], "relevance": r[2], "timestamp": r[3]} for r in results]

    async def _decay_loop(self):
        while True:
            await asyncio.sleep(86400)  # Every 24 hours
            await asyncio.get_event_loop().run_in_executor(self.executor, self._decay_db)

    def _decay_db(self):
        conn = sqlite3.connect(self.db_path)
        conn.execute("UPDATE memories SET relevance = relevance * 0.5 WHERE timestamp < ?", (time.time() - 86400,))
        conn.commit()
        conn.close()

import asyncio
import sqlite3
from typing import Dict, List, Any
from event_bus import EventBus, Event, EventType
from memory_store import MemoryStore
from agent_base import AgentBase, ReviewerAgent
from pydantic import BaseModel

class Task(BaseModel):
    id: str
    description: str

class Orchestrator:
    """
    IMPROVED Orchestrator from Run 1: Now actually routes tasks through the event bus to agents instead of direct calls.
    Implements self-improvement loop: after every 10 tasks, analyzes success/failure rates via event logs and adjusts agent selection strategy.
    Checkpoints state to SQLite. In Run 1, it was a stub; now it's dynamic and learning.
    """
    def __init__(self, event_bus: EventBus, memory_store: MemoryStore, agents: List[AgentBase]):
        self.event_bus = event_bus
        self.memory_store = memory_store
        self.agents = agents
        self.reviewer = ReviewerAgent(event_bus)
        self.task_count = 0
        self.success_rates: Dict[str, float] = {agent.name: 0.5 for agent in agents}
        self.strategy = "round_robin"  # Default
        self.db_path = "orchestrator.db"
        self._init_db()
        self.event_bus.subscribe(EventType.TASK_COMPLETED, self._handle_task_completion)

    def _init_db(self):
        conn = sqlite3.connect(self.db_path)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS checkpoints (
                id INTEGER PRIMARY KEY,
                state TEXT
            )
        """)
        conn.commit()
        conn.close()

    async def assign_task(self, task: Task):
        agent = self._select_agent()
        await self.event_bus.publish(Event(type=EventType.TASK_ASSIGNED, payload={"task": task.dict(), "agent": agent.name}))
        self.task_count += 1
        if self.task_count % 10 == 0:
            await self._self_improve()

    def _select_agent(self) -> AgentBase:
        if self.strategy == "best_performance":
            return max(self.agents, key=lambda a: self.success_rates[a.name])
        else:  # round_robin
            return self.agents[self.task_count % len(self.agents)]

    async def _self_improve(self):
        # Analyze logs (simplified: count completions vs failures from dead-letter or metrics)
        # Adjust strategy based on rates
        avg_rate = sum(self.success_rates.values()) / len(self.success_rates)
        if avg_rate < 0.6:
            self.strategy = "best_performance"
        else:
            self.strategy = "round_robin"
        print(f"Self-improvement: Switched to {self.strategy} strategy. Avg success: {avg_rate}")
        # Checkpoint
        state = {"task_count": self.task_count, "strategy": self.strategy, "rates": self.success_rates}
        conn = sqlite3.connect(self.db_path)
        conn.execute("INSERT INTO checkpoints (state) VALUES (?)", (str(state),))
        conn.commit()
        conn.close()

    async def _handle_task_completion(self, event: Event):
        agent_name = event.payload.get("agent")
        success = event.payload.get("success", True)
        self.success_rates[agent_name] = (self.success_rates[agent_name] + (1 if success else 0)) / 2  # Simple average

import asyncio
from abc import ABC, abstractmethod
from typing import Dict, Any
from event_bus import EventBus, Event, EventType
from pydantic import BaseModel

class AgentResult(BaseModel):
    task_id: str
    output: str
    success: bool

class AgentBase(ABC):
    """
    IMPROVED AgentBase from Run 1: Agents now subscribe to event bus for task assignment and report results via events.
    Tracks per-agent success metrics. Added ReviewerAgent that evaluates outputs via events.
    In Run 1, agents were not integrated; now they are event-driven.
    """
    def __init__(self, name: str, event_bus: EventBus):
        self.name = name
        self.event_bus = event_bus
        self.success_count = 0
        self.total_tasks = 0
        self.event_bus.subscribe(EventType.TASK_ASSIGNED, self._handle_assignment)

    @abstractmethod
    async def process_task(self, task: Dict[str, Any]) -> AgentResult:
        pass

    async def _handle_assignment(self, event: Event):
        if event.payload["agent"] == self.name:
            result = await self.process_task(event.payload["task"])
            self.total_tasks += 1
            if result.success:
                self.success_count += 1
            await self.event_bus.publish(Event(type=EventType.TASK_COMPLETED, payload={
                "agent": self.name, "task_id": result.task_id, "output": result.output, "success": result.success
            }))

class SimpleAgent(AgentBase):
    async def process_task(self, task: Dict[str, Any]) -> AgentResult:
        # Simulate processing
        await asyncio.sleep(0.1)
        return AgentResult(task_id=task["id"], output=f"Processed {task['description']}", success=True)

class ReviewerAgent:
    """
    NEW: ReviewerAgent evaluates other agents' outputs via AGENT_EVALUATION events.
    """
    def __init__(self, event_bus: EventBus):
        self.event_bus = event_bus
        self.event_bus.subscribe(EventType.TASK_COMPLETED, self._review)

    async def _review(self, event: Event):
        # Simple review: check if output contains "error"
        quality = 1.0 if "error" not in event.payload["output"].lower() else 0.5
        await self.event_bus.publish(Event(type=EventType.AGENT_EVALUATION, payload={
            "agent": event.payload["agent"], "task_id": event.payload["task_id"], "quality": quality
        }))

import asyncio
from event_bus import EventBus
from memory_store import MemoryStore
from orchestrator import Orchestrator, Task
from agent_base import SimpleAgent, ReviewerAgent

async def main():
    """
    Integration demo: Runs a full cycle with all 4 modules working together.
    """
    event_bus = EventBus()
    memory_store = MemoryStore(event_bus)
    agents = [SimpleAgent("Agent1", event_bus), SimpleAgent("Agent2", event_bus)]
    reviewer = ReviewerAgent(event_bus)  # Note: Reviewer is separate, not in agents list
    orchestrator = Orchestrator(event_bus, memory_store, agents)

    # Store some memory
    await memory_store.store("Sample knowledge")

    # Assign tasks
    for i in range(12):  # More than 10 to trigger self-improvement
        task = Task(id=str(i), description=f"Task {i}")
        await orchestrator.assign_task(task)
        await asyncio.sleep(0.1)  # Allow processing

    # Query memory
    results = await memory_store.query("knowledge")
    print(f"Memory query results: {results}")

    # Metrics
    metrics = event_bus.get_metrics()
    print(f"Event bus metrics: {metrics}")

if __name__ == "__main__":
    asyncio.run(main())
